 17 /**
 18  * In a mesh, the geometry is structured in units called cells (triangles).
 19  * This class provides operations on individual cells.
 20  * 
 21  * @class Provides cell by cell operations on models
 22  * @constructor 
 23  * @param {vxlModel} model the model for this mesh
 24  * @author Diego Cantor
 25  */
 26 function vxlMesh(model){
 27     
 28     this.cells = [];
 29     this.indices = [];
 30     this.normals = [];
 31     this.renderable = undefined;
 32     this.color = [0.8,0.8,0.8]; 
 33     
 34     this._init(model);
 35 };
 36 
 37 vxlMesh.prototype.setColor = function(color){
 38     this.color = color;
 39     
 40     
 41     if (this.renderable){
 42         this.updateRenderableColors();
 43     }
 44 }
 45 /**
 46  * Identifies the cells existing in the 
 47  * @private
 48  */
 49 vxlMesh.prototype._init = function(model){
 50     
 51     var ver = model.vertices;
 52     var ind = model.indices;
 53     
 54     var self = this;
 55     
 56     
 57     function initMesh(){
 58         var start = new Date().getTime();
 59         for(var i=0, L = ind.length; i<L; i+=3){ //for all indices
 60             idx  = ind[i];
 61             var triangle = [],x,y,z,idx;
 62 
 63             x = ver[idx*3];
 64             y = ver[idx*3 + 1];
 65             z = ver[idx*3 + 2];   
 66             triangle.push([x,y,z]);
 67 
 68             idx = ind[i+1];
 69             x = ver[idx*3];
 70             y = ver[idx*3 + 1];
 71             z = ver[idx*3 + 2];   
 72             triangle.push([x,y,z]);
 73 
 74             idx = ind[i+2];            
 75             x = ver[idx*3];
 76             y = ver[idx*3 + 1];
 77             z = ver[idx*3 + 2];   
 78             triangle.push([x,y,z]);
 79 
 80             self.indices.push([ind[i], ind[i+1], ind[i+2]]);
 81             self.cells.push(new vxlCell(triangle));
 82         }
 83         var col = [self.color[0], self.color[1], self.color[2]];
 84         for (var i=0, L =self.cells.length; i<L; i+=1){
 85             self.normals.push(self.cells[i]._normal);
 86             self.cells[i].color = col; //BEWARE FLOATARRAYS DONT TRANSLATE WELL
 87         }
 88         
 89         self.renderable = self._getRenderable();
 90         var elapsed = new Date().getTime() - start;
 91         console.info('Mesh ['+ model.name +'] generated in '+elapsed+ ' ms');
 92     };
 93     
 94     //because this operation is time consuming it is deferred here.
 95     //this causes that the reenderable object is not available in the renderer
 96     //until this op finishes.
 97    setTimeout(function(){initMesh()},0);
 98 };

101 /**
102  * Returns a renderable model of the mesh
103  * @private
104  */
105 vxlMesh.prototype._getRenderable = function(){
106     
107     var r = new vxlModel();
108     
109     r.colors = [];
110     r.pickingColors = [];
111     for(var i=0, count = this.cells.length; i<count; i +=1){
112             r.indices   = r.indices.concat([i*3, i*3+1, i*3+2]);
113             r.vertices  = r.vertices.concat(this.cells[i].getFlattenVertices());
114             for (var j = 0; j<3;j+=1){
115                 r.colors    = r.colors.concat(this.cells[i].color);
116                 r.pickingColors = r.pickingColors.concat(this.cells[i]._pickingColor);
117             }
118     }
119     
120     r.computeNormals();
121     return r;
122 };

134                 r.pickingColors = r.pickingColors.concat(this.cells[i]._pickingColor);
135             }
136     }
137 };

139 /**
140  * Determines if this mesh contains the cell indicated by the parameter cellUID
141  * @param {string} cellUID the unique identifier of a cell
142  */
143 vxlMesh.prototype.hasCell = function(cellUID){
144   for(var i=0, count= this.cells.length; i < count; i+=1){
145       if (this.cells[i].UID == cellUID){
146           return true;
147       }
148   } 
149   return false; 
150 };  
151 
152 /**
153  * Determines if this mesh contains the cell indicated by the parameter cellUID
154  * @param {string} cellUID the unique identifier of a cell
155  */
156 vxlMesh.prototype.getCell = function(cellUID){
157   for(var i=0, count= this.cells.length; i < count; i+=1){
158       if (this.cells[i].UID == cellUID){
159           return this.cells[i];
160       }
161   } 
162   return null; 
163 };
164 
165 
166 vxlMesh.prototype.removeCell = function(cellUID){
167   var idx = -1;
168   for(var i=0, count= this.cells.length; i < count; i+=1){
169       if (this.cells[i].UID == cellUID){
170           idx = i;
171           break;
172       }
173   }
174   if (idx !=-1) 
175    this.cells.splice(idx,1);
176 };

178 /**
179  * This is just an experimental method. Determines what cells are facing the camera. 
180  * May be this can be used for anything? I don't know! 
181  * Maybe to see through a surface??
182  *
183  * @param {Object} camera
184  * @param {Object} angle
185  */
186 vxlMesh.prototype.intersect = function(camera, angle){
187     
188     var ray = camera._normal;
189     
190     selection = [];
191 
192     for(var i=0;i<this.normals.length; i+=1){
193         var dp = Math.acos(vec3.dot(ray, this.normals[i])) * vxl.def.rad2deg;
194         if (Math.abs(dp) <= angle){
195             selection = selection.concat(this.indices[i]);
196         }  
197     }
198     return selection;
199 };
