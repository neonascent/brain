 16 /**
 17  * Creates a vxlLandmark object and associates it with a vxlCamera. 
 18  * 
 19  * 
 20  * @class Stores the camera state so it can be retrieved later 
 21  * @constructor this is the constructor doc
 22  * @param {vxlCamera} camera
 23  * @author Diego Cantor
 24  * @see vxlCamera
 25   */
 26 
 27 function vxlLandmark(name, camera) {
 28 	
 29 	if(!( camera instanceof vxlCamera)) {
 30 		alert('vxlLandmark needs a vxlCamera as argument');
 31 		return null;
 32 	}
 33 	
 34 	this.name             = name;
 35 	this.camera           = camera;
 36 	
 37 	var c = this.camera;
 38     
 39     this._fov             = c._fov;
 40     this.Z_NEAR           = c.Z_NEAR;    
 41     this.Z_FAR            = c.Z_FAR;
 42     
 43     this._matrix          = mat4.set(c._matrix, mat4.create());
 44     this._right           = vec3.set(c._right, vec3.create());
 45     this._up              = vec3.set(c._up, vec3.create());
 46     this._normal          = vec3.set(c._normal, vec3.create());   
 47     this._position        = vec3.set(c._position, vec3.create());
 48     this._focalPoint      = vec3.set(c._focalPoint, vec3.create());
 49     this._distanceVector  = vec3.set(c._distanceVector, vec3.create());
 50     
 51     this._azimuth       = c._azimuth;
 52     this._elevation     = c._elevation;
 53     this._roll          = c._roll;
 54     this._relAzimuth    = c._relAzimuth;
 55     this._relElevation  = c._relElevation;
 56     this._relRoll       = c._relRoll;
 57     this._dollyingStep  = c._dollyngStep; //dollying step
 58     this._distance      = c._distance;
 59     
 60     this._following     = c._following;
 61     this._trackingMode  = c._trackingMode;
 62 };
 63 
 64 
 65 /**
 66  * Updates the camera with the state stored in vxlLandmark.
 67  */
 68 vxlLandmark.prototype.retrieve = function() {
 69     
 70 	var c = this.camera;
 71 	
 72 	c._fov             = this._fov;
 73     c.Z_NEAR           = this.Z_NEAR;    
 74     c.Z_FAR            = this.Z_FAR;
 75     
 76     c._matrix          = mat4.set(this._matrix, mat4.create());
 77     c._right           = vec3.set(this._right, vec3.create());
 78     c._up              = vec3.set(this._up, vec3.create());
 79     c._normal          = vec3.set(this._normal, vec3.create());   
 80     c._position        = vec3.set(this._position, vec3.create());
 81     c._focalPoint      = vec3.set(this._focalPoint, vec3.create());
 82     c._distanceVector  = vec3.set(this._distanceVector, vec3.create());
 83     
 84     c._azimuth       = this._azimuth;
 85     c._elevation     = this._elevation;
 86     c._roll          = this._roll;
 87     c._relAzimuth    = this._relAzimuth;
 88     c._relElevation  = this._relElevation;
 89     c._relRoll       = this._relRoll;
 90     c._dollyingStep  = this._dollyngStep; //dollying step
 91     c._distance      = this._distance;
 92     
 93     c._following     = this._following;
 94     c._trackingMode  = this._trackingMode;
 95     
 96     c.refresh();
 97 };
