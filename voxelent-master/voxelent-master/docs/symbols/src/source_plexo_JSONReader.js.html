 17 vxlJSONReader.prototype               = new vxlDataStreamReader();
 18 vxlJSONReader.prototype.constructor   = vxlJSONReader;
 19 
 20 /**
 21  * Reads data periodically from a JSON file. The source can be a php that returns the JSON
 22  * or a JSON file in the server itself.
 23  * @extends vxlDataStreamReader
 24  * @class
 25  */
 26 function vxlJSONReader(){
 27     vxlDataStreamReader.call(this);
 28 }; 
 29 
 30 /**
 31  * @private
 32  */
 33 vxlJSONReader.prototype._requestData = function(){
 34     
 35     var self = this;
 36     
 37     var successHandler = function(){
 38         return function(json, textStatus){
 39             if (json == null) return;
 40              self.callback(json);
 41             
 42        }
 43     };
 44     
 45     var errorHandler = function(){
 46         return function(request, status, error){}
 47      };
 48 
 49     //executes the AJAX request
 50     var request  = $.ajax({
 51         url         : self.source,
 52         type        :"GET",
 53         dataType    :"json",
 54         success     : successHandler(),
 55         error       : errorHandler()
 56     });  
 57 };
