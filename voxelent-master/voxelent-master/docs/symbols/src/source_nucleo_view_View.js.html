 17 /**
 18 * Each HTML5 canvas is assigned a view object (vxlView) in Voxelent's Nucleo.
 19 * 
 20 * A vxlView provides the code to handle cameras, interaction and rendering capabilities, plus actor handling on the 
 21 * HTML5 canvas that otherwise would need to be written over and over for each application
 22 * if you were writing a WebGL app from scratch.
 23 * Luckily this is not the case. You have the awesome vxlView who takes care of all these little details for you.
 24 * @class the class that manages what it is rendered the HTML5 canvas on the web page 
 25 * @constructor
 26 * @param {String} canvasID id of the canvas in the DOM tree. That's all we need to setup a vxlView for you
 27 * @param {vxlScene} scene if this view is sharing a scene, this parameter corresponds to the scene being shared.
 28 * @param {Boolean} handleLayout if true or absent, the canvas will be resized dynamically
 29 * @author Diego Cantor
 30 */
 31 
 32 function vxlView(canvasID, scene, handleLayout){
 33 
 34 	//this.id = 0; //@TODO: Who handles this? Maybe one Scene has several Views?
 35 	
 36 	this.canvas = document.getElementById(canvasID);
 37 	
 38 	if (this.canvas == null){
 39 		alert('View: the canvas ' + canvasID+ ' does not exist.');
 40 		throw('View: the canvas ' + canvasID+ ' does not exist.');
 41 	}
 42 	
 43 	this.name = canvasID;
 44 	
 45     //View dimensions
 46 	this.width = this.canvas.width;
 47 	this.height = this.canvas.height;
 48 	
 49 	//Clear depth and background color
 50 	this.clearDepth = 1.0;
 51 	this.backgroundColor = vxl.def.view.background.slice(0);
 52 	
 53 	//Drag and drop
 54 	this.dragndrop = false;
 55 
 56 	//Create Renderer
 57 	this.renderer = new vxlRenderer(this);
 58 	this.setBackgroundColor(this.backgroundColor);
 59 	this.setClearDepth(this.clearDepth);
 60 	
 61 	//Create Camera Manager
 62 	this.cameraman = new vxlCameraManager(this);
 63 	
 64 	//Create View Interactor
 65 	this.interactor = new vxlTrackerInteractor();
 66 	
 67 	this.interactor.connectView(this);
 68 	
 69 	//Create Scene
 70 	if (scene != null && scene instanceof vxlScene){
 71 		this.scene = scene;
 72 	}
 73 	else if (vxl.c.scene != undefined){
 74 		this.scene = vxl.c.scene;
 75 	}
 76 	else{
 77 		this.scene = new vxlScene();
 78 	
 79 	}
 80 	//Add this view to the scene;
 81 	this.scene.views.push(this);
 82 	
 83 	if (vxl.c.view == undefined){
 84 		vxl.c.view = this;
 85 	}
 86 	
 87 	vxl.go.views.push(this);
 88 	
 89 	if (handleLayout == undefined){
 90 	    handleLayout = true;
 91 	}
 92 	
 93 	this.setAutoResize(handleLayout);
 94 	
 95 	this.UID = vxl.util.generateUID();
 96 
 97 
 98     var ntf = vxl.go.notifier;
 99     var e   = vxl.events;
100     ntf.publish(e.VIEW_NEW, this);
101     ntf.subscribe(e.DEFAULT_LUT_LOADED,this);
102     ntf.fire(e.VIEW_NEW, this);
103 
104 	vxl.go.console('View: the view '+ this.name+' has been created');
105 };

108 /**
109  * Handles the events to which a view is subscribed in Voxelent's Nucleo
110  * @param {String} event the name of the event
111  * @param {Object} source the origin of the event. Useful to do callbacks if necessary
112  */
113 vxlView.prototype.handleEvent = function(event, source){
114 	vxl.go.console('vxlView ['+this.name+']: receiving event '+event);
115 	if (event == vxl.events.DEFAULT_LUT_LOADED){
116 		this.scene.setLookupTable(vxl.def.lut.main);
117 	}
118 };

128 /**
129  * It creates a div around the canvas to handle resizing appropriately
130  * @private 
131  */
132 vxlView.prototype._wrapDiv = function(){
133     
134     $(this.canvas).css({
135         position: 'absolute',
136         height: 'auto',
137         bottom: 0,
138         top: 0,
139         left: 0,
140         right: 0,
141         margin: "5px"
142     });
143     
144     $(this.canvas).wrap('<div id='+this.name+'-wrapper />');
145     $('#'+this.name+'-wrapper').css({
146        width:'100%',
147        height:'100%' 
148     });
149     
150     this.canvas.focus();
151 }

153 /**
154  * Resizes the canvas so it fits its parent node in the DOM tree
155  */
156 vxlView.prototype.resize = function(){
157     var parent = this.canvas.parentNode;
158     this.width = $(parent).width()-5;       
159     this.height = $(parent).height()-5;
160     
161     if (this.width > window.innerWidth - 5){
162     	this.width = window.innerWidth -5;
163     }
164     
165     if (this.height > window.innerHeight - 5){
166     	this.height = window.innerHeight - 5;
167     }
168     
169     $(this.canvas).attr('width', this.width);
170     $(this.canvas).attr('height', this.height);
171 }
172 
173 /**
174  * Enables automatic resizing of the view when the browser window is resized
175  * @param flag enbles automatic resizing if true, disables it if false
176  */
177 vxlView.prototype.setAutoResize = function(flag){
178     
179     if (flag){
180          this._wrapDiv();
181          this.resize(); //first time
182         $(window).resize((function(self){return function(){self.resize();}})(this));
183     }
184     else{
185         $(window).unbind('resize');
186     }
187 };

190 /**
191  * Run non standard code (mozilla, webkit) 
192  *@private 
193  */
194 vxlView.prototype._runPrefixMethod = function(obj,method){
195     
196     var pfx = ["webkit", "moz", "ms", "o", ""];
197     var p = 0, m, t;
198 
199     while (p < pfx.length && !obj[m]) {
200         m = method;
201         if (pfx[p] == "") {
202             m = m.substr(0,1).toLowerCase() + m.substr(1);
203         }
204         m = pfx[p] + m;
205         t = typeof obj[m];
206         if (t != "undefined") {
207             pfx = [pfx[p]];
208             return (t == "function" ? obj[m]() : obj[m]);
209         }
210         p++;
211     }
212 };

214 /**
215  * Handles HTML5 Full screen for this view 
216  * @param {Boolean} flag set flag to true if fullscreen is wanted. Set it to false to exit fullscreen.
217  */
218 vxlView.prototype.fullscreen = function(flag){
219     if (flag == true){
220         this._runPrefixMethod(this.canvas, "RequestFullScreen");
221     }
222     else{
223         if (this._runPrefixMethod(document,"FullScreen") || 
224             this._runPrefixMethod(document,"isFullScreen")){
225             this._runPrefixMethod(document,"CancelFullScreen");
226         }    
227     }
228 
229 };

234  */

247 /**
248  * Resets the view
249  */
250 vxlView.prototype.reset = function(){
251 	this.stop();
252 	this.scene.reset();
253 	this.cameraman.reset();
254 	this.start();
255 };

257 /**
258  * Sets the view's background color
259  * @param {Number, Array, vec3} r it can be the red component, a 3-dimensional Array or a vec3 (glMatrix)
260  * @param {Number} g if r is a number, then this parameter corresponds to the green component
261  * @param {Number} b if r is a number, then this parameter corresponds to the blue component
262  
263  * @see vxlRenderer#clearColor
264  */
265 vxlView.prototype.setBackgroundColor = function(r,g,b){
266 	this.backgroundColor = vxl.util.createArr3(r,g,b); 
267 	this.renderer.clearColor(this.backgroundColor);
268 };
