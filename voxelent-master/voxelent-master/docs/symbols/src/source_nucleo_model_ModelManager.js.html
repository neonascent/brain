 16 /**
 17  * Creates vxlModel objects and assigns them to a vxlScene. 
 18  * 
 19  * vxlModelManager provides methods for loading remote and local models.
 20  * 
 21  * 
 22  * @class Controls the loading and network transfer of models into memory
 23  * @constructor
 24  * @author Diego Cantor
 25  */
 26 function vxlModelManager(){
 27 	this.toLoad = []; 
 28 	this.models = [];
 29 	
 30 	var e = vxl.events;
 31 	vxl.go.notifier.publish([
 32 	       e.MODELS_LOADING,
 33 	       e.MODEL_NEW,
 34 	       e.MODELS_LOADED
 35 	    ],this);
 36 	vxl.go.notifier.subscribe(e.READER_DONE, this);
 37 };
 38 
 39 /**
 40  * Handles voxelent events to which this model manager is subscribed to
 41  * @param {String} event the event that was fired 
 42  * @param {Object} source the object that fired the event 
 43  */
 44 vxlModelManager.prototype.handleEvent = function(event,source){
 45     var reader = source;
 46     var parts = reader.getParts();
 47     for (var i = 0; i < parts.length; i+=1){
 48         this.add(parts[i],parts[i].name,reader.scene);
 49     }
 50 }

 53 /**
 54  * Uses a JSON/Ajax mechanism to load models from a Web Server.
 55  * @param {String} uri The path to the file that will be loaded. 
 56  * @param {vxlScene} scene The scene that will create an actor for this model. This parameter is optional.
 57  */  
 58 vxlModelManager.prototype.load = function(uri, scene) {
 59     var manager = this;
 60     
 61     var filename = uri.replace(/^.*[\\\/]/, '');
 62     var modelname = filename.split('.')[0];
 63     var extension = filename.split('.')[1];
 64 	
 65 	var dtype = 'json';
 66 	
 67 	if (manager.isModelLoaded(modelname)) return;
 68 	
 69 	
 70 	if (extension == 'vtk'){
 71 	    dtype ='text';
 72 	}
 73 	else if (extension == 'json'){
 74 	    dtype = 'json'
 75 	}
 76 	else{
 77 	    alert ('vxlManager.load ERROR: Unknown filetype ['+extension+']');
 78 	    throw ('vxlManager.load ERROR: Unknown filetype ['+extension+']');
 79 	}
 80 	   
 81 	nocacheuri = uri +'?nocache=' + (new Date()).getTime();
 82 	
 83 	
 84 	vxl.go.console('ModelManager.load: Requesting '+uri+'...');
 85 	vxl.go.notifier.fire(vxl.events.MODELS_LOADING, this);
 86 	
 87 	var successHandler = function(manager,modelname,scene){
 88 	    switch(extension){
 89 	    case 'json':
 90     		return function(json, textStatus){
 91     		    json.uri = filename;
 92     		    json.path = vxl.util.getPath(uri)
 93     			manager.add(json,modelname,scene);
 94     		}
 95     		break;
 96     	case 'vtk':
 97 		    return function(data){
 98 		        reader = new vxlVTKReader(scene)
 99 		        reader.parseText(modelname, data)
100 		    }
101 		 }
102 	};
103 	
104 	var errorHandler = function(uri){
105 		return function(request, status, error){
106 			
107 			if(error.code = 1012){
108 				alert('The file '+uri+' could not be accessed. \n\n'+
109     			'Please make sure that the path is correct and that you have the right pemissions');
110 			}
111 			else{
112 				alert ('There was a problem loading the file '+uri+'. HTTP error code:'+request.status);
113 			}		
114 		}
115 	};
116 	
117 	var request  = $.ajax({
118 		url			: nocacheuri,
119 		type		:"GET",
120 		dataType	: dtype,
121 		success 	: successHandler(manager,modelname,scene),
122 		error		: errorHandler(uri)
123 	});    
124 };

126 /**
127  * Loads a list of models and assigns them to a scene
128  * @param {Array} list list of files to load 
129  * @param {vxlScene} scene scene to callback when the models are loaded 
130  */
131 vxlModelManager.prototype.loadList = function(list,scene){
132 	this.toLoad = list.slice(0); 
133 	vxl.go.console('ModelManager.loadList: models to load ->' + this.toLoad.length);
134    	for(var i=0;i<this.toLoad.length; i++){
135 		this.load(this.toLoad[i],scene);
136    	}
137 };
138 
139 /**
140  * 
141  * @param {Object} JSON_OBJECT the JSON object that contains the definition of the model
142  * @param {String} name name of the model to be created
143  * @param {vxlScene} scene the scene to be called back when the model is created
144  */
145 vxlModelManager.prototype.add = function(JSON_OBJECT,name,scene){
146 	
147 	var self = this;
148 	
149 	function scheduledAdd(){
150 	
151     	
152     	var model = new vxlModel(name, JSON_OBJECT);
153     	
154     	model.loaded = true;
155     	self.models.push(model);
156     	self.toLoad.splice(name,1); //removes it from the pending list if exists
157     	
158     	vxl.go.console('ModelManager: model '+model.name+' created.'); 
159     	
160     	if (scene != undefined && scene instanceof vxlScene){
161     		vxl.go.console('ModelManager: notifying the scene...');
162     		if (scene.loadingMode == vxl.def.model.loadingMode.LIVE){
163     			scene.createActor(model);
164     		}
165     		else if (scene.loadingMode == vxl.def.model.loadingMode.LATER){
166     			if(self.allLoaded()){
167     				scene.createActors(self.models);
168     			}
169     		}
170     		else if (scene.loadingMode == vxl.def.model.loadingMode.DETACHED){
171     			//do nothing
172     		}
173     	}
174         
175     	if(self.allLoaded()){ 
176     		vxl.go.notifier.fire(vxl.events.MODELS_LOADED, self);
177     	}
178     	else{
179     	    vxl.go.notifier.fire(vxl.events.MODEL_NEW, self);
180     	}
181 	}
182 	
183 	setTimeout(function(){scheduledAdd()},0);
184  };

186 /**
187  * It will delete all of the loaded models
188  */  
189 vxlModelManager.prototype.reset = function(){
190 	this.firstLoadedModel = false;
191 	for (var i=0; i <this.models.length; i++){
192 		this.models[i] = null;
193 	}
194 	
195 	this.models        = [];
196 	this.toLoad        = [];
197 };
198 
199 /**
200  * Checks if a model has been loaded yet
201  * @param {String} name the name of the model to check
202  */
203 vxlModelManager.prototype.isModelLoaded = function(name){
204 	for(var i=0;i<this.models.length;i++){
205 		if (this.models[i].name == name) return true;
206 	}
207 	return false;
208 };
209 
210 /**
211  * Returns true if all the models in the list passed to the method loadList have been loaded.
212  */
213 vxlModelManager.prototype.allLoaded = function(){
214 	return (this.toLoad.length == 0);
215 };

218 /**
219  * Returns the model if it has been loaded by this model manager, null otherwise.
220  * @param {String} name the name of the model to retrieve
221  */
222 vxlModelManager.prototype.getModelByName = function(name){
223  	for(var i=0, max = this.models.length; i<max; i+=1){
224 		if (this.models[i].name == name) return this.models[i];
225 	}
226 	return null;
227 };
