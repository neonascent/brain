 17 /**
 18  * Models are totally independent of views and of the rendering process
 19  * @class Represents a geometric object. A model is represented by an actor in a scene.
 20  * @constructor
 21  * @param {String} name the name for this model
 22  * @param {Object} JSON_OBJECT the JSON Object that defines this model (Optional)
 23  * @see vxlModel#load
 24  * @author Diego Cantor
 25  */
 26 function vxlModel(name, JSON_OBJECT){
 27 	this.name       = name;
 28 	this.indices 	= [];
 29 	this.vertices 	= [];
 30 	this.scalars 	= undefined;
 31 	this.diffuse	= undefined;
 32 	this.ambient    = undefined;
 33 	this.specular   = undefined;
 34 	this.shininness = undefined;
 35 	this.normals 	= undefined;
 36 	this.wireframe 	= undefined;
 37 	this.bb         = [0,0,0,0,0,0];
 38     this.centre     = [0,0,0,0,0,0];
 39 	this.mode       = vxl.def.actor.mode.SOLID
 40 	this.image      = undefined;
 41 	this.uri        = undefined;
 42 	this.colors     = undefined;
 43 	this.type       = vxl.def.model.type.SIMPLE;
 44 	this.renderable = undefined;
 45    
 46     if (JSON_OBJECT != undefined){
 47         this.load(this.name, JSON_OBJECT);
 48     }
 49     
 50     this.setType(this.type);
 51 };
 52 
 53 /**
 54  * Sets the type of model. A renderable object is created and associated
 55  * with this model in case the type is MESH or BIG_DATA
 56  * 
 57  * @param {vxl.def.model.type} type the type of model
 58  */
 59 vxlModel.prototype.setType = function(type){
 60     
 61     if (this.indices.max() < vxl.def.model.MAX_NUM_INDICES && 
 62         type == vxl.def.model.type.BIG_DATA){
 63             throw('The model is not big enough to be of BIG_DATA type. Max index found: '+this.indices.max());
 64         }
 65     
 66     this.type = type;  
 67 };

 77 /**
 78  * Populates this model with the JSON_OBJECT (JSON object)
 79  * @param {String} nm the name given to this model
 80  * @param {Object} JSON_OBJECT the JSON object that describes the model
 81  */
 82 vxlModel.prototype.load = function(nm,JSON_OBJECT){
 83 	this.name		= nm.replace(/\.[^/.]+$/, "")
 84 	if (JSON_OBJECT.name != null){ //if the name is defined in the JSON object, then use it
 85 		this.name = JSON_OBJECT.name;
 86 	}
 87 	
 88 	//Load all properties
 89 	for(i in JSON_OBJECT){
 90 		this[i] = JSON_OBJECT[i];
 91 	}
 92 	
 93 	this.update();
 94 
 95 };
 96 
 97 /**
 98  * Update model state when the model has changed (changes in vertices or indices) 
 99  */
100 vxlModel.prototype.update = function(){
101         //Now minimal checks
102     if (this.vertices == undefined){
103         alert('The model '+ this.name+' does not have vertices. Impossible to render!');
104     }
105     
106 
107     if(
108         (this.normals == undefined || 
109             (this.normals != undefined && this.normals.length == 0))
110         && this.indices != undefined)
111     {
112         this.computeNormals();
113     }
114     
115      
116     if (this.wireframe == undefined){
117         this.computeWireframeIndices();
118     }
119     
120     if (this.mode == undefined){
121         this.mode = vxl.def.actor.mode.SOLID;
122     }
123     
124     if (this.texture != undefined){
125         this.mode = vxl.def.actor.mode.TEXTURED;
126     }
127     
128     this.computeBoundingBox();
129     
130     
131     if (this.type == vxl.def.model.type.SIMPLE && this.indices.max() > vxl.def.model.MAX_NUM_INDICES){
132         this.setType(vxl.def.model.type.BIG_DATA);
133     }
134     
135 };

137 /**
138  * Calculates the normals for this model in case that the JSON object does not include them
139  * 
140  * 
141  */
142 vxlModel.prototype.computeNormals = function(){
143   //face normal calculation
144     var vs = this.vertices;
145 	var ind = this.indices;
146 	var x=0; 
147     var y=1;
148 	var z=2;
149 	
150 	var ns = [];
151 	for(var i=0;i<vs.length;i=i+3){ //for each index, initialize normal x, normal y, normal z
152 		ns[i+x]=0.0;
153 		ns[i+y]=0.0;
154 		ns[i+z]=0.0;
155 	}
156 	
157 	for(var i=0;i<ind.length;i=i+3){ //we work on triads of vertex to calculate normals so i = i+3 (i = indices index)
158 		var v1 = [];
159 		var v2 = [];
160 		var normal = [];	
161 		//p2 - p1
162 		v1[x] = vs[3*ind[i+2]+x] - vs[3*ind[i+1]+x];
163 		v1[y] = vs[3*ind[i+2]+y] - vs[3*ind[i+1]+y];
164 		v1[z] = vs[3*ind[i+2]+z] - vs[3*ind[i+1]+z];
165 		//p0 - p1
166 		v2[x] = vs[3*ind[i]+x] - vs[3*ind[i+1]+x];
167 		v2[y] = vs[3*ind[i]+y] - vs[3*ind[i+1]+y];
168 		v2[z] = vs[3*ind[i]+z] - vs[3*ind[i+1]+z];
169 		//cross product
170 		normal[x] = v1[y]*v2[z] - v1[z]*v2[y];
171 		normal[y] = v1[z]*v2[x] - v1[x]*v2[z];
172 		normal[z] = v1[x]*v2[y] - v1[y]*v2[x];
173 		
174 		for(j=0;j<3;j++){ //update the normals of the triangle
175 			ns[3*ind[i+j]+x] =  ns[3*ind[i+j]+x] + normal[x];
176 			ns[3*ind[i+j]+y] =  ns[3*ind[i+j]+y] + normal[y];
177 			ns[3*ind[i+j]+z] =  ns[3*ind[i+j]+z] + normal[z];
178 		}
179 	}
180 		
181 	//normalize the result
182 	for(var i=0;i<vs.length;i=i+3){ //the increment here is because each vertex occurs with an offset of 3 in the array (due to x, y, z contiguous values)
183 	
184 	    var nn=[];
185 		nn[x] = ns[i+x];
186 		nn[y] = ns[i+y];
187 		nn[z] = ns[i+z];
188 		
189 		var len = Math.sqrt((nn[x]*nn[x])+(nn[y]*nn[y])+(nn[z]*nn[z]));
190 		if (len == 0) len = 1.0;
191 		
192 		nn[x] = nn[x]/len;
193 		nn[y] = nn[y]/len;
194 		nn[z] = nn[z]/len;
195 		
196 		ns[i+x] = nn[x];
197 		ns[i+y] = nn[y];
198 		ns[i+z] = nn[z];
199 	}
200 	this.normals = ns;
201 };

203 /**
204  * Flips the normals 
205  */
206 vxlModel.prototype.flipNormals = function(){
207     var ns = this.normals;
208     
209     if (ns == undefined) return; //no normals
210     
211     for (var i=0; i<ns.length; i+=1){
212         ns[i] = -ns[i];
213     }
214 };

216 /**
217  * Generate the wireframe indices using the model indices
218  */  
219 vxlModel.prototype.computeWireframeIndices = function(){
220 	var ind = this.indices;
221     var wfi = [];
222 	var j = 0;
223 	for(var i=0; i<ind.length; i=i+3){
224 	   wfi[j] = ind[i];
225 	   wfi[j+1] = ind[i+1];
226 	   wfi[j+2] = ind[i+1];
227 	   wfi[j+3] = ind[i+2];
228 	   wfi[j+4] = ind[i+2];
229 	   wfi[j+5] = ind[i];
230 	   j = j+6;
231 	}
232 	this.wireframe = wfi;
233 };

236 /**
237  * Calculate the bounding box of this model and its centre
238  * 
239  */
240 vxlModel.prototype.computeBoundingBox = function(){	
241     
242     //This is the case with the scene toys
243     if (this.vertices.length == 0){
244         this.bb = [0,0,0,0,0,0];
245         this.centre = [0,0,0,0,0,0];
246         return;
247     }
248     
249 	var vs = this.vertices;
250 	var bbm  = [vs[0],vs[1],vs[2],vs[0],vs[1],vs[2]];
251 	
252 	var i = vs.length;  
253 	for(var i=0, N = vs.length;i<N;i=i+3){
254 		bbm[0] = Math.min(bbm[0],vs[i]);
255 		bbm[1] = Math.min(bbm[1],vs[i+1]);
256 		bbm[2] = Math.min(bbm[2],vs[i+2]);
257 		bbm[3] = Math.max(bbm[3],vs[i]);
258 		bbm[4] = Math.max(bbm[4],vs[i+1]);
259 		bbm[5] = Math.max(bbm[5],vs[i+2]);
260 	}
261 	
262 	
263 	var c = [0, 0, 0];
264      //computes the centre 
265     c[0] = (bbm[3] + bbm[0]) /2;
266     c[1] = (bbm[4] + bbm[1]) /2;
267     c[2] = (bbm[5] + bbm[2]) /2;
268     
269     
270     this.bb = bbm;  
271     this.centre = c;
272 };

274 /**
275  *Returns the bounding box vertices. This method is used by the rendering engine 
276  */
277 vxlModel.prototype.getBoundingBoxVertices = function(){
278     var b = this.bb;
279     return [
280         b[0], b[1], b[2],
281         b[0], b[4], b[2],
282         b[3], b[4], b[2],
283         b[3], b[1], b[2],
284         b[0], b[1], b[5],
285         b[0], b[4], b[5],
286         b[3], b[4], b[5],
287         b[3], b[1], b[5] 
288         ];
289 };
