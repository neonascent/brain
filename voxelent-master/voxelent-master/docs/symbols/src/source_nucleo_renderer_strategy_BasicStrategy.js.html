 21 /**
 22  * @constructor
 23  * @class
 24  * Implements a basic rendering strategy 
 25  * 
 26  * 
 27  * A vxlBasicStrategy object is selected by default as the strategy that an actor
 28  * will use for rendering.
 29  * 
 30  * A rendering strategy allows decoupling rendering specific code (i.e. code that access and 
 31  * 	communicates with the program) from the actor.
 32  * @author Diego Cantor
 33  * 
 34  */
 35 function vxlBasicStrategy(renderer){
 36 	vxlRenderStrategy.call(this,renderer);
 37 	this._gl_buffers  = {};
 38 	this._gl_textures = {};
 39 	
 40 	if (renderer){
 41 	var gl = this.renderer.gl;
 42 	
 43 	gl.enable(gl.BLEND);
 44 	gl.blendFunc(gl.SRC_ALPHA, gl.ONE_MINUS_SRC_ALPHA);
 45 	gl.enable(gl.DEPTH_TEST);
 46 	gl.depthFunc(gl.LESS);
 47 	gl.clearDepth(1.0);
 48 	gl.disable(gl.CULL_FACE);
 49     }	
 50 }

 52 /**
 53  * Implements basic allocation of memory. Creates the WebGL buffers for the actor
 54  * @param {vxlScene} scene the scene to allocate memory for
 55   */
 56 vxlBasicStrategy.prototype.allocate = function(scene){
 57     var elements = scene._actors.concat(scene.toys.list);
 58     var NUM = elements.length;
 59     
 60     for(var i = 0; i < NUM; i+=1){
 61         this._allocateActor(elements[i]);
 62     }
 63 };

 83 /**
 84  * Receives one actor and returns the GL buffers
 85  */
 86 vxlBasicStrategy.prototype._allocateActor = function(actor){
 87    
 88     if (this._gl_buffers[actor.UID] != undefined){
 89         if (actor.dirty){
 90             this._reallocateActor(actor); 
 91         }
 92         return;
 93     }
 94    	
 95 	var gl = this.renderer.gl;
 96 	var model = actor.model;
 97     var buffers = {};
 98 	
 99 	//Vertex Buffer
100 	buffers.vertex = gl.createBuffer();
101 	
102 	
103 	//Index Buffer
104 	if (model.indices != undefined){
105 		buffers.index = gl.createBuffer();
106 	}
107 	
108 	//Normals Buffer
109 	if (model.normals){
110 		buffers.normal = gl.createBuffer();
111 	}
112 	
113 	//Color Buffer for scalars
114 	if (model.scalars != undefined || model.colors != undefined){
115 		buffers.color = gl.createBuffer(); //we don't BIND values or use the buffers until the lut is loaded and available
116 	}
117 	
118 	//Wireframe Buffer 
119 	if (model.wireframe != undefined){
120 		buffers.wireframe = gl.createBuffer();
121 	}
122 	
123 	//Bounding box Buffer 
124     buffers.bb = gl.createBuffer();
125 	
126 	//Texture Coords Buffer
127 	if (model.texture){
128 	    buffers.texcoords = gl.createBuffer();
129 	}
130 	   
131     //Cleaning up
132     gl.bindBuffer(gl.ARRAY_BUFFER, null);
133     gl.bindBuffer(gl.ELEMENT_ARRAY_BUFFER, null);
134 	
135 	
136 	this._gl_buffers[actor.UID] = buffers;
137 	this._reallocateActor(actor);
138 };

141 /**
142  * Receives one actor and returns the GL buffers
143  */
144 vxlBasicStrategy.prototype._reallocateActor = function(actor){
145    
146     
147     vxlRenderStrategy
148     var gl = this.renderer.gl;
149     var model = actor.model;
150     var buffers = this._gl_buffers[actor.UID];
151     
152     gl.bindBuffer(gl.ARRAY_BUFFER, buffers.vertex);
153     gl.bufferData(gl.ARRAY_BUFFER, new Float32Array(model.vertices), gl.STATIC_DRAW);
154     
155     //Index Buffer
156     if (model.indices != undefined){    
157     //Cleaning up
158     gl.bindBuffer(gl.ARRAY_BUFFER, null);
159     gl.bindBuffer(gl.ELEMENT_ARRAY_BUFFER, null);
160         gl.bindBuffer(gl.ELEMENT_ARRAY_BUFFER, buffers.index);
161         gl.bufferData(gl.ELEMENT_ARRAY_BUFFER, new Uint16Array(model.indices), gl.STATIC_DRAW);
162     }
163     
164     //Normals Buffer
165     if (model.normals){
166         gl.bindBuffer(gl.ARRAY_BUFFER, buffers.normal);
167         gl.bufferData(gl.ARRAY_BUFFER, new Float32Array(model.normals), gl.STATIC_DRAW);
168     }
169     
170     //Color Buffer for scalars
171     if (model.scalars != undefined || model.colors != undefined){
172         buffers.color = gl.createBuffer(); //we don't BIND values or use the buffers until the lut is loaded and available
173     }
174     
175     //Wireframe Buffer 
176     if (model.wireframe != undefined){
177         gl.bindBuffer(gl.ELEMENT_ARRAY_BUFFER, buffers.wireframe);
178         gl.bufferData(gl.ELEMENT_ARRAY_BUFFER, new Uint16Array(model.wireframe), gl.STATIC_DRAW);
179     }   
180     //Cleaning up
181     gl.bindBuffer(gl.ARRAY_BUFFER, null);
182     gl.bindBuffer(gl.ELEMENT_ARRAY_BUFFER, null);
183     
184     //Bounding box Buffer 
185     gl.bindBuffer(gl.ELEMENT_ARRAY_BUFFER, buffers.bb);
186     gl.bufferData(gl.ELEMENT_ARRAY_BUFFER, new Uint16Array(vxlModel.BB_INDICES), gl.STATIC_DRAW);
187     
188     
189     //Texture Coords Buffer
190     if (model.texture){
191         gl.bindBuffer(gl.ARRAY_BUFFER, buffers.texcoords);
192         gl.bufferData(gl.ARRAY_BUFFER, new Float32Array(model.texcoords), gl.STATIC_DRAW);
193         
194         this._gl_textures[actor.UID] = gl.createTexture();
195         gl.bindTexture(gl.TEXTURE_2D, this._gl_textures[actor.UID]);
196         gl.texImage2D(gl.TEXTURE_2D, 0, gl.RGBA, gl.RGBA, gl.UNSIGNED_BYTE, actor.texture.image);
197         gl.generateMipmap(gl.TEXTURE_2D);
198         
199         gl.bindTexture(gl.TEXTURE_2D, null);
200     }
201     
202     
203     //Cleaning up
204     gl.bindBuffer(gl.ARRAY_BUFFER, null);
205     gl.bindBuffer(gl.ELEMENT_ARRAY_BUFFER, null);
206 };
207 /**
208  * Passes the matrices to the shading program
209  * @param {vxlActor} the actor 
210  * we will update each Model-View matrix of each renderer according to
211  * the actor position,scale and rotation.
212  * @private
213  */
214 vxlBasicStrategy.prototype._applyActorTransform = function(actor){
215     
216     var r		= this.renderer;
217     var trx 	= r.transforms
218     var	prg 	= r.prg;
219     var glsl 	= vxl.def.glsl;
220 
221     trx.push();
222         mat4.multiply(trx.mvMatrix, actor._matrix);
223 	    prg.setUniform(glsl.MODEL_VIEW_MATRIX,	r.transforms.mvMatrix);
224 
225 	    trx.calculateModelViewPerspective();
226         prg.setUniform(glsl.MVP_MATRIX, r.transforms.mvpMatrix);
227         
228         trx.calculateNormal(); 
229         prg.setUniform(glsl.NORMAL_MATRIX, r.transforms.nMatrix);
230     
231     trx.pop();
232  };
