 17 /**
 18  * 
 19  * A vxlCamera object simplifies WebGL programming by providing a simple object interface to the lower level
 20  * matrix manipulations that are required to view a 3D scene.
 21  * 
 22  * When moving and rotating the camera, such matrices are updated and Voxelent's Nucleo will use this information
 23  * in order to draw the scene accordingly to the camera position an orientation.
 24  * 
 25  * A vxlCamera object requires a vxlView to be created. Having said that, a vxlView can be associated with 
 26  * multiple cameras.
 27  * 
 28  * The azimuth, elevation and roll operations occur with respect to the standard coordinate system:
 29  * right =[1,0,0], up = [0,1,0] and normal 
 30  * of 
 31  * 
 32  * @class Like a real camera it moves around the 3D scene displaying the current point of view in the browser
 33  * @constructor Creates a vxlCamera. 
 34  * @param {vxlView} vw
 35  * @param {Object} t the type of camera 
 36  * @author Diego Cantor
 37  */
 38 function vxlCamera(vw,t) {
 39     this.UID            = vxl.util.generateUID(); //unique identification key
 40     this.view           = vw;
 41 
 42     this._fov           = vxl.def.camera.fov;
 43     this.Z_NEAR         = vxl.def.camera.near;    
 44     this.Z_FAR          = vxl.def.camera.far;
 45     
 46     this._matrix 	      = mat4.identity();
 47     this._right 		  = vec3.createFrom(1, 0, 0);
 48 	this._up              = vec3.createFrom(0, 1, 0);
 49 	this._forward          = vec3.createFrom(0, 0, 1);	
 50 	this._position        = vec3.createFrom(0, 0, 1);
 51 	this._focalPoint      = vec3.createFrom(0, 0, 0);
 52 	this._distanceVector  = vec3.createFrom(0, 0, 0);
 53     
 54     this._azimuth 	    = 0;
 55     this._elevation     = 0;
 56     this._roll          = 0;
 57 	this._relAzimuth    = 0;
 58     this._relElevation  = 0;
 59     this._relRoll       = 0;
 60 	this._dollyingStep  = 0; //dollying step
 61     this._distance      = 1;
 62     
 63     this._following     = undefined;
 64     this._trackingMode  = vxl.def.camera.tracking.DEFAULT;
 65     
 66     this.landmarks         = [];
 67 
 68 	if (t != undefined && t !=null){
 69         this.type = t;
 70     }
 71     else {
 72         this.type = vxl.def.camera.type.EXPLORING;
 73     }
 74     
 75 };

 77 /**
 78  * Establishes the type of camera
 79  * @param {vxl.def.camera.type} type the type of camera
 80  * @param {vxl.def.camera.tracking} trackingMode if the camera is of tracking type, the tracking mode can be set as 
 81  * an optional parameter here.
 82  * @see {vxl.def.camera}
 83  */
 84 vxlCamera.prototype.setType = function(type, trackingMode){
 85     
 86     if (type != vxl.def.camera.type.ORBITING && 
 87         type != vxl.def.camera.type.TRACKING &&
 88         type != vxl.def.camera.type.EXPLORING) {
 89         console.error('vxlCamera.setType ERROR type'+ type +' unknown. Setting camera to EXPLORING type.');
 90         this.type = vxl.def.camera.type.EXPLORING;
 91     }
 92     else {
 93         this.type = type
 94     }
 95     
 96     if (this.type  == vxl.def.camera.type.TRACKING && trackingMode != undefined){
 97         this.setTrackingMode(trackingMode);
 98     }
 99 };

101 /**
102  * Sets the tracking type of this camera when it follows an actor
103  * <p> to set the tracking type of the camera <code>myCamera</code> you should make sure that your camera is of tracking type with:
104  *  <code>myCamera.setType(vxl.def.camera.type.TRACKING)</code>.
105  *  For instance:
106  * </p>
107  *  <pre class='prettyprint'>
108  *  var actor = vxl.api.getActor('cone'); //from the current scene
109  *  var camera = vxl.c.camera;
110  *  camera.setType(vxl.def.camera.type.TRACKING);
111  *  camera.setTrackingMode(vxl.def.camera.tracking.ROTATIONAL);
112  *  camera.follow(actor);
113  * </pre>
114  * <p> a shorter way would be:</p>
115  * <pre class='prettyprint'>
116  *  var actor = vxl.api.getActor('cone'); //from the current scene
117  *  var camera = vxl.c.camera;
118  *  camera.setType(vxl.def.camera.type.TRACKING);
119  *  camera.follow(actor, vxl.def.camera.tracking.ROTATIONAL);
120  * </pre>
121  * @see vxlCamera#follow, vxlCamera#setTrackingMode
122  */
123 vxlCamera.prototype.setTrackingMode = function(mode){
124     if (this.type != vxl.def.camera.type.TRACKING){
125         alert("Impossible to set a tracking mode if the camera is not of tracking type");
126         throw("Impossible to set a tracking mode if the camera is not of tracking type");
127     }
128     if (mode == undefined) return;
129     this._trackingMode = mode;
130 };

132 /**
133  * Follows a given actor. If this operation is called on an ORBITING camera,
134  * the camera mode will change to be a TRACKING camera.
135  * 
136  * @param {vxlActor} actor actor to track
137  * @param {String} trackingType one of the possible values of <code>vxl.def.camera.tracking</code>
138  * @see {vxlCamera#setType, vxl.def.camera.tracking}
139  */
140 vxlCamera.prototype.follow = function(actor, trackingType){
141     this.setType(vxl.def.camera.type.TRACKING, trackingType);
142     if (actor == undefined || actor == null){
143         alert("vxlCamera.follow: Unable to follow undefined/null actor");
144         console.error("vxlCamera.follow: Unable to follow undefined/null actor");
145     }
146     this._following = actor;
147     actor.addTrackingCamera(this);
148 };

158 /**
159  * Updates the camera according to the current tracking mode. This method is a callback used by the 
160  * vxlActor class to notify the camera of any actor transformations. This is applicable of course only when
161  * this camera is following the actor passed as parameter here
162  * @param {vxlActor} actor the actor being followed by this camera
163  */
164 vxlCamera.prototype.updateWithActor = function(actor){
165     if (this._following != actor) return; //fail safe
166     
167     switch(this._trackingMode){
168         case vxl.def.camera.tracking.DEFAULT:
169                 break;
170         case vxl.def.camera.tracking.ROTATIONAL:
171         case vxl.def.camera.tracking.CINEMATIC:
172                 this.setFocalPoint(actor._position);
173                 break;
174         case vxl.def.camera.tracking.TRANSLATIONAL:
175                 this.translate(actor._translation);  
176                 this.setFocalPoint(actor._position);      
177                 break;
178     }
179 };

181  *<p>Sets the camera position in the scene
182  * This method has three parameters x,y,z which represent the coordinates for 
183  * the camera's position.
184  * </p>
185  * <p>
186  * This method takes into account the current focal point. The camera will look at the
187  * focal point after this operation. If you want to move the camera position and the focal point
188  * simultaneously, then use <code>vxlCamera.translate</code>.
189  * </p>
190  *  
191  *  * @param {Number, Array} x the x-coordinate. x can also be an Array [a,b,c] in this case the y and z parameters are discarded.
192  * @param {Number} y the y-coordinate
193  * @param {Number} z the z-coordinate
194  * @see{vxlCamera#translate}
195  */
196 vxlCamera.prototype.setPosition = function(x,y,z) {
197     this._setPosition(x,y,z);
198     this.setFocalPoint(this._focalPoint);
199 };

201 /**
202  * Looks at a given point in space (sets the focal point of this camera).
203  * 
204  * Note: If the camera is doing cinematic tracking, the up vector will be affected.
205  * @param {Number, Array, vec3} x it can be the x coordinate, a 3-dimensional Array or a vec3 (glMatrix)
206  * @param {Number} y if x is a number, then this parameter corresponds to the y-coordinate
207  * @param {Number} z if x is a number, then this parameter corresponds to the z-coordinate
208  */
209 vxlCamera.prototype.setFocalPoint = function(x,y,z) {
210     
211     var up = vec3.create([0,1,0]);
212     //var up = vec3.create(this._up);
213     this._focalPoint = vxl.util.createVec3(x,y,z);
214     
215     if (this._trackingMode == vxl.def.camera.tracking.CINEMATIC){
216         var d  = vec3.subtract(this._focalPoint, this._position, vec3.create());
217         var x  = d[0], y = d[1],  z = d[2],  r  = vec3.length(d);
218         var el =      Math.asin(y/r)  * vxl.def.rad2deg;
219         var az = 90 + Math.atan2(z,x) * vxl.def.rad2deg;
220         var m = mat4.identity();
221         mat4.rotateY(m, az * vxl.def.deg2rad);
222         mat4.rotateX(m, el * vxl.def.deg2rad);
223         up = mat4.multiplyVec3(m, [0,1,0], vec3.create());
224     }
225     
226     mat4.inverse(mat4.lookAt(this._position, this._focalPoint, up), this._matrix);
227     
228     this._getAxes();
229     this._getDistance();
230     this._getAngles();
231 };

233 /**
234  * Position the camera to a given distance from the current focal point
235  * @param {Number} d the distance
236  * @TODO: REVIEW COMMENTED CODE
237  */
238 vxlCamera.prototype.setDistance = function(d) {
239     
240     if(this._distance == d || d <0) {
241         return;
242     }
243 
244     this._distance = d;
245 
246     // Distance should be greater than .0002
247     if(this._distance < 0.0002) {
248         this._distance = 0.0002;
249         console.warn(" vxlCamera.setDistance WARN: Distance is set to minimum (0.0002)");
250     }
251     
252     this._dollyingStep = this._distance / 100;
253     
254     var pos = vec3.create();
255     
256     var d = this._distance;
257     var n = this._forward;
258     var f = this._focalPoint;
259     
260     pos[0] = d*n[0] + f[0];
261     pos[1] = d*n[1] + f[1];
262     pos[2] = d*n[2] + f[2];
263     
264     this._setPosition(pos);
265 };

343 /**
344  * Changes the azimuth and elevation with respect to the current camera axes
345  * @param {Number} azimuth the relative azimuth
346  * @param {Number} elevation the relative elevation
347  * @param {Number} roll the relative roll
348  */
349 vxlCamera.prototype.rotate = function(azimuth,elevation,roll){
350    
351     if (this.type == vxl.def.camera.type.EXPLORING){ 
352         
353         azimuth   = this._getAngle(azimuth);
354         elevation = this._getAngle(elevation);
355         roll      = this._getAngle(roll);
356         var rotX  = quat4.fromAngleAxis(elevation * vxl.def.deg2rad, [1,0,0]);
357         var rotY  = quat4.fromAngleAxis(azimuth   * vxl.def.deg2rad, [0,1,0]);
358         var rotZ  = quat4.fromAngleAxis(roll      * vxl.def.deg2rad, [0,0,1]); 
359         var rotQ = quat4.multiply(rotY, rotX, quat4.create());
360         rotQ = quat4.multiply(rotQ, rotZ, quat4.create());
361         var rotMatrix = quat4.toMat4(rotQ);
362         mat4.translate(this._matrix, [0,0,-this._distance]);
363         mat4.multiply(this._matrix, rotMatrix);
364         mat4.translate(this._matrix, [0,0,this._distance]);
365     }
366     else {
367         if (Math.abs(this._elevation + elevation) > 90) return; //don't allow
368         this._relElevation = this._getAngle(elevation);
369         this._relAzimuth = this._getAngle(azimuth);
370         this._relRoll = this._getAngle(roll);
371         
372         if (this.type == vxl.def.camera.type.TRACKING){
373             //this._relElevation = -this._relElevation;
374             //this._relAzimuth   = -this._relAzimuth;
375             //this._relRoll      = -this._relRoll;
376         }
377         
378         this._elevation += this._relElevation;
379         this._azimuth += this._relAzimuth;
380         this._roll += this._relRoll;
381         
382          this._computeMatrix();
383     }
384     
385        
386     this._getAxes();
387     if (this.type == vxl.def.camera.type.ORBITING || this.type == vxl.def.camera.type.EXPLORING){
388         this._getPosition();
389     }
390     else if (this.type == vxl.def.camera.type.TRACKING){
391         this._getFocalPoint();
392     }
393     
394    this._update();
395 };

398 /**
399  * Performs the dollying operation in the direction indicated by the camera normal axis.
400  * The dollying mechanism offered by a camera makes sure that the camera moves fast
401  * towards the object when the distance is large and slow when it is very close to the object.
402  * For that effect, every time that the new position  (after dollying) is calculated, the field dstep is computed.
403  * 
404  * @param {Number} value the dollying value 
405  * 
406  */
407 vxlCamera.prototype.dolly = function(value) {
408     
409 	var    n =  this._forward; 
410     var pos = vec3.create(this._position);
411     var step = value*this._dollyingStep;
412     pos[0] += step*n[0];
413     pos[1] += step*n[1];
414     pos[2] += step*n[2];
415     
416     this._setPosition(pos);
417     if (this.type == vxl.def.camera.type.ORBITING || this.type == vxl.def.camera.type.DYNAMIC){
418         this._getDistance();
419     }
420     else if (this.type == vxl.def.camera.type.TRACKING){
421         //move the focal point and keep the distance
422         vec3.add(pos, this._distanceVector, this._focalPoint);
423     }
424 };

547 /**

556             return;
557         }

574 vxlCamera.prototype.getLandmarks
575 /**
576  * This method sets the camera to a distance such that the area covered by the bounding box (parameter)
577  * is viewed.
578  * @param {vxlBoundingBox} bb the bounding box
579  */
580 vxlCamera.prototype._shot = function(bb){
581     
582     this.setElevation(0);
583     this.setAzimuth(0);
584     
585 	var maxDim = Math.max(bb[3] - bb[0], bb[4] - bb[1]);
586 	
587 	cc = [0,0,0];
588 
589 	cc[0] = (bb[3] + bb[0]) /2;
590 	cc[1] = (bb[4] + bb[1]) /2;
591 	cc[2] = (bb[5] + bb[2]) /2;
592 		
593 	cc[0] = Math.round(cc[0]*1000)/1000;
594 	cc[1] = Math.round(cc[1]*1000)/1000;
595 	cc[2] = Math.round(cc[2]*1000)/1000;
596 	
597 	if(maxDim != 0) {
598 		var d = 1.5 * maxDim / (Math.tan(this._fov * Math.PI / 180));
599 		this.setPosition([cc[0], cc[1], cc[2]+ d]);
600 	}
601 	
602 	this.setFocalPoint(cc);
603 };

642 /**
643  * Sets the  camera matrix
644  * @private
645  */
646 vxlCamera.prototype._computeMatrix = function(){
647     
648     mat4.identity(this._matrix);
649     
650     var rotX  = quat4.fromAngleAxis(this._elevation * vxl.def.deg2rad, [1,0,0]);
651     var rotY  = quat4.fromAngleAxis(this._azimuth   * vxl.def.deg2rad, [0,1,0]);
652     var rotZ  = quat4.fromAngleAxis(this._roll      * vxl.def.deg2rad, [0,0,1]); 
653     var rotQ = quat4.multiply(rotY, rotX, quat4.create());
654     rotQ = quat4.multiply(rotQ, rotZ, quat4.create());
655     var rotMatrix = quat4.toMat4(rotQ);
656     
657     if (this.type ==  vxl.def.camera.type.ORBITING || this.type == vxl.def.camera.type.EXPLORING){
658         mat4.translate(this._matrix, this._focalPoint);
659         mat4.multiply(this._matrix, rotMatrix);
660         mat4.translate(this._matrix, [0,0,this._distance]);
661     } 
662     else if(this.type ==  vxl.def.camera.type.TRACKING){
663         mat4.translate(this._matrix, this._position);
664         mat4.multiply(this._matrix, rotMatrix);
665     }
666 };

709 /**
710  * Recalculates axes based on the current matrix
711  * @private
712  */
713 vxlCamera.prototype._getAxes = function(){
714     var m       = this._matrix;
715     vec3.set(mat4.multiplyVec4(m, [1, 0, 0, 0]), this._right);
716     vec3.set(mat4.multiplyVec4(m, [0, 1, 0, 0]), this._up);
717     vec3.set(mat4.multiplyVec4(m, [0, 0, 1, 0]), this._forward);
718     vec3.normalize(this._right);
719     vec3.normalize(this._up);
720     vec3.normalize(this._forward);
721 };

723 /**
724  * Recalculates the position based on the current matrix
725  * Called only when camera is of ORBITING type
726  * @private
727  */
728 vxlCamera.prototype._getPosition = function(){
729     var m       = this._matrix;
730     vec3.set(mat4.multiplyVec4(m, [0, 0, 0, 1]), this._position);
731     this._getDistance();
732 };

734 /**
735  * Called only when camera is of TRACKING type
736  * @private
737  */
738 vxlCamera.prototype._getFocalPoint = function(){
739     mat4.multiplyVec3(mat4.toRotationMat(this._matrix), [0,0,-this._distance], this._distanceVector); 
740     vec3.add(this._position, this._distanceVector, this._focalPoint);
741     this._getDistance();               
742 };

744 /**
745  * Recalculates the distance variables based on the current state
746  * @private
747  */
748 vxlCamera.prototype._getDistance = function(){
749     this._distanceVector = vec3.subtract(this._focalPoint, this._position, vec3.create());
750     this._distance = vec3.length(this._distanceVector);
751     this._dollyingStep = this._distance / 100;
752 };

754 /**
755  * Recalculates euler angles based on the current state
756  * @private
757  */
758 vxlCamera.prototype._getAngles = function(){
759     //Recalculates angles  
760     var x = this._distanceVector[0], y = this._distanceVector[1],  z = this._distanceVector[2];
761     var r = vec3.length(this._distanceVector);
762     this._elevation = Math.asin(y/r)    * vxl.def.rad2deg;
763     this._azimuth   = Math.atan2(-x,-z) * vxl.def.rad2deg;
764 };

767 /**
768  * Recalculates the camera state from the camera matrix
769  * @private
770  */
771 vxlCamera.prototype._update = function(){
772     this._getAxes();
773     this._getPosition();
774     this._getDistance();
775     this._getAngles();
776 };

778 /**
779  * @private
780  */
781 vxlCamera.prototype._calculateAngles = function(){
782     var rotM = mat4.toMat3(this._matrix);
783     var Q = mat3.toQuat4(rotM);
784     var x = Q[0], y = Q[1], z = Q[2], w=Q[3];
785     
786     var roll  = Math.atan2(2 * (w * x + y * z),1 - 2 * (x * x + y * y)) * vxl.def.rad2deg;
787     var pitch = Math.asin (2 * (w * y - z * y)) * vxl.def.rad2deg;
788     var yaw   = Math.atan2(2 * (w * z + x * y), 1 - 2 * (y *y + z * z)) * vxl.def.rad2deg;
789     
790     console.info(' roll :' + vxl.util.format(roll,2));
791     console.info('pitch :' + vxl.util.format(pitch,2));
792     console.info('  yaw :' + vxl.util.format(yaw,2));
793 };

795 /**
796  * Returns an angle between 0 and 360 degrees
797  * @private
798  */
799 vxlCamera.prototype._getAngle = function(angle){
800    
801     if (angle == undefined){
802          return 0;
803     }
804     else if (angle > 360 || angle <-360) {
805         return angle % 360;
806     }
807     else return angle;
808 };
