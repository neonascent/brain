 17 /**
 18  * @class Represents the ESSL source code of each program
 19  */
 20 function vxlProgram(){
 21     this.ID = undefined;
 22     this.ATTRIBUTES = [];
 23     this.UNIFORMS = [];
 24     this.VERTEX_SHADER = "";
 25     this.FRAGMENT_SHADER = "";
 26     this.DEFAULTS = {};
 27 };
 28 
 29 vxlProgram.prototype.copy = function(prg){
 30     this.ID                 = prg.ID;
 31     this.ATTRIBUTES         = prg.ATTRIBUTES;
 32     this.UNIFORMS           = prg.UNIFORMS;
 33     this.VERTEX_SHADER      = prg.VERTEX_SHADER;
 34     this.FRAGMENT_SHADER    = prg.FRAGMENT_SHADER;
 35     this.DEFAULTS           = prg.DEFAULTS;
 36 };
 37 
 38 /**
 39  * Obtain the list of attributes and uniforms from the code
 40  */
 41 vxlProgram.prototype.introspect = function(){
 42     
 43     this.ATTRIBUTES = [];
 44     this.UNIFORMS = [];
 45     
 46     var code = this.VERTEX_SHADER.concat(this.FRAGMENT_SHADER);
 47     code = code.replace(/(\r\n|\n\r|\n)/gm,"");
 48     
 49     var uniforms   = code.match(/(uniform)\s*\w*\s*\w*/g);
 50     var attributes = code.match(/(attribute)\s*\w*\s*\w*/g);    
 51     
 52     if (uniforms.length == 0){
 53         throw new vxlProgramException("The code for the program "+this.ID+" does not contain any valid uniforms");
 54     }
 55     
 56     if (attributes.length == 0){
 57         throw new vxlProgramException("The code for the program "+this.ID+" does not contain any valid attributes");
 58     }
 59     
 60     for(var i=0, N = uniforms.length; i < N; i +=1){
 61         this.UNIFORMS.push(uniforms[i].substr(uniforms[i].lastIndexOf(" ")+1, uniforms[i].length));
 62     }
 63     
 64     for(var i=0, N = attributes.length; i < N; i +=1){
 65         this.ATTRIBUTES.push(attributes[i].substr(attributes[i].lastIndexOf(" ")+1, attributes[i].length));
 66     }
 67     
 68 };
 69 /**
 70  * Creates a program object from the ESSL scripts embedded in the DOM
 71  * @param {Object} id
 72  * @param {Object} vertexShaderId
 73  * @param {Object} fragmentShaderId
 74  */
 75 vxlProgram.createFromDOM = function(id, vertexShaderId,fragmentShaderId){
 76 
 77     var prg = new vxlProgram();
 78     
 79     prg.ID = id;
 80     var vsElement   = document.getElementById(vertexShaderId);
 81     var fsElement = document.getElementById(fragmentShaderId);
 82     
 83     if (vsElement == null || fsElement == null){
 84         throw new vxlProgramException("shaders don't exist");
 85     }
 86     
 87     prg.VERTEX_SHADER = vsElement.innerHTML;
 88     prg.FRAGMENT_SHADER = fsElement.innerHTML;
 89     
 90     prg.introspect();
 91     
 92     return prg;
 93     
 94 };

 97 vxlProgram.createFromJSON = function(json){
 98   
 99     var prg = new vxlProgram();
100      if (json.ID){
101        prg.ID = json.ID;
102      } //otherwise use the one defined in the constructor
103       
104      prg.VERTEX_SHADER = json.VERTEX_SHADER;
105      prg.FRAGMENT_SHADER = json.FRAGMENT_SHADER;
106      prg.DEFAULTS = json.DEFAULTS;      
107      prg.introspect();
108      
109      return prg;
110   
111 };
