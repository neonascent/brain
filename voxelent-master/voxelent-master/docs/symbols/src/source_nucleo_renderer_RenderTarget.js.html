 18 function vxlRenderTarget(renderer){
 19    
 20     this.canvas         = renderer.view.canvas;
 21     this.texture        = null;
 22     this.framebuffer    = null;
 23     this.renderbuffer   = null;
 24     this.gl             = renderer.gl;
 25     this.configure(); 
 26 };
 27 
 28 
 29 vxlRenderTarget.prototype.configure = function(){
 30     var width = this.canvas.width;
 31     var height = this.canvas.height;
 32     var gl = this.gl;
 33     
 34     //1. Init Picking Texture
 35     this.texture = gl.createTexture();
 36     gl.bindTexture(gl.TEXTURE_2D, this.texture);
 37     gl.texImage2D(gl.TEXTURE_2D, 0, gl.RGBA, width, height, 0, gl.RGBA, gl.UNSIGNED_BYTE, null);
 38     
 39     //2. Init Render Buffer
 40     this.renderbuffer = gl.createRenderbuffer();
 41     gl.bindRenderbuffer(gl.RENDERBUFFER, this.renderbuffer);
 42     gl.renderbufferStorage(gl.RENDERBUFFER, gl.DEPTH_COMPONENT16, width, height);
 43     
 44     
 45     //3. Init Frame Buffer
 46     this.framebuffer = gl.createFramebuffer();
 47     gl.bindFramebuffer(gl.FRAMEBUFFER, this.framebuffer);
 48     gl.framebufferTexture2D(gl.FRAMEBUFFER, gl.COLOR_ATTACHMENT0, gl.TEXTURE_2D, this.texture, 0);
 49     gl.framebufferRenderbuffer(gl.FRAMEBUFFER, gl.DEPTH_ATTACHMENT, gl.RENDERBUFFER, this.renderbuffer);
 50     
 51 
 52     //4. Clean up
 53     gl.bindTexture(gl.TEXTURE_2D, null);
 54     gl.bindRenderbuffer(gl.RENDERBUFFER, null);
 55     gl.bindFramebuffer(gl.FRAMEBUFFER, null);
 56 };
 57 
 58 vxlRenderTarget.prototype.update = function(){
 59     
 60     var gl = this.gl;
 61     var width = this.canvas.width;
 62     var height = this.canvas.height;
 63    
 64     gl.bindTexture(gl.TEXTURE_2D, this.texture);
 65     gl.texImage2D(gl.TEXTURE_2D, 0, gl.RGBA, width, height, 0, gl.RGBA, gl.UNSIGNED_BYTE, null);
 66     
 67     //2. Init Render Buffer
 68     gl.bindRenderbuffer(gl.RENDERBUFFER, this.renderbuffer);
 69     gl.renderbufferStorage(gl.RENDERBUFFER, gl.DEPTH_COMPONENT16, width, height);
 70 };
 71 
 72 /**
 73  * 
 74  * @param{Number} x
 75  * @param{Number} y
 76  */
 77 vxlRenderTarget.prototype.readPixel = function(x,y){
 78     
 79     var gl = this.gl;
 80     var readout = new Uint8Array(1 * 1 * 4);
 81     gl.bindFramebuffer(gl.FRAMEBUFFER, this.framebuffer);
 82     gl.readPixels(x,y,1,1,gl.RGBA,gl.UNSIGNED_BYTE,readout);
 83     gl.bindFramebuffer(gl.FRAMEBUFFER, null);
 84     return readout;
 85 }
