 17 /**
 18  * A cell is the minimum surface that can be selected on a mesh
 19  * @class Provides cell definitions
 20  * @constructor
 21  * @author Diego Cantor
 22  */
 23 function vxlCell(mesh, index, vertices, color){
 24     
 25     this.UID = vxl.util.generateUID();
 26     this.mesh  = mesh;
 27     this.index = index;
 28     this.vertices = vertices;
 29     this.color = color==undefined?[0.8,0.8,0.8]:color;
 30     this.normal = undefined; //cell normal
 31     this.position = [0,0,0];
 32     this._calculatePosition();
 33     this._calculateNormal();
 34     this._pickingColor  = vxl.go.picker.getColorFor(this);
 35 };
 36 
 37 vxlCell.prototype._calculatePosition = function(){
 38     this.position[0] = (this.vertices[0][0] + this.vertices[1][0] + this.vertices[2][0])/3;
 39     this.position[1] = (this.vertices[0][1] + this.vertices[1][1] + this.vertices[2][1])/3;
 40     this.position[2] = (this.vertices[0][2] + this.vertices[1][2] + this.vertices[2][2])/3;
 41 };
 42     
 43     
 44 /**
 45  * Calculates the cell normal
 46  * @private
 47  */
 48 vxlCell.prototype._calculateNormal = function(){
 49     var p1 = vec3.subtract(this.vertices[1], this.vertices[0], vec3.create());
 50     var p2 = vec3.subtract(this.vertices[2], this.vertices[0], vec3.create());
 51     this.normal =  vec3.normalize(vec3.cross(p1,p2,vec3.create()));
 52 };
 53 
 54 /**
 55  * Returns an unidimensional array with the vertex information
 56  * [[a,b,c],[d,e,f],[g,h,i]] --> [a,b,c,d,e,f,g,h,i]
 57  */
 58 vxlCell.prototype.getFlattenVertices = function(){
 59     var v = this.vertices;
 60     return [v[0][0],v[0][1],v[0][2],v[1][0],v[1][1],v[1][2],v[2][0],v[2][1],v[2][2]];
 61 };
 62 
 63 
 64 /**
 65  * Updates the cell color. 
 66  * @param {Object} r
 67  * @param {Object} g
 68  * @param {Object} b
 69  */
 70 vxlCell.prototype.setColor = function(r,g,b){
 71     this.color = vxl.util.createArr3(r,g,b);
 72     
 73     this.mesh._updateCellColor(this.index);
 74     
 75 }
