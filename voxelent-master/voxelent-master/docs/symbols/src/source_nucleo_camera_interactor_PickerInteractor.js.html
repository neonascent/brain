 19 function vxlPickerInteractor(camera){
 20 	
 21 	this.plist 					= [];
 22 	this.texture 				= null;
 23 	this.framebuffer 			= null;
 24 	this.renderbuffer 			= null;
 25 	this.hitPropertyCallback 	= null;
 26 	this.processHitsCallback 	= null;
 27 	this.addHitCallback 		= null;
 28 	this.removeHitCallback 		= null;
 29 	this.moveCallback 			= null;
 30 	this.picking 				= false;
 31 	
 32 	this.connectTo(camera);
 33 }

 35 vxlPickerInteractor.prototype.connectTo = function(camera){
 36 	vxlCameraInteractor.apply(this, camera);
 37 	if (this.camera){
 38 		this.configure();
 39 	}
 40 }

 42 vxlPickerInteractor.prototype.configure = function(){
 43 	
 44 	var gl = this.camera.view.renderer.gl;
 45 	
 46 	var width = 512*4; //@TODO get size from this.camera.view.canvas
 47 	var height = 512*4;
 48 	
 49 	//1. Init Picking Texture
 50 	this.texture = gl.createTexture();
 51 	gl.bindTexture(gl.TEXTURE_2D, this.texture);
 52 	gl.texParameteri(gl.TEXTURE_2D, gl.TEXTURE_MAG_FILTER, gl.LINEAR);
 53 	gl.texParameteri(gl.TEXTURE_2D, gl.TEXTURE_MIN_FILTER, gl.LINEAR_MIPMAP_NEAREST);
 54     gl.generateMipmap(gl.TEXTURE_2D);
 55     gl.texImage2D(gl.TEXTURE_2D, 0, gl.RGBA, width, height, 0, gl.RGBA, gl.UNSIGNED_BYTE, null);
 56 	
 57 	//2. Init Render Buffer
 58 	this.renderbuffer = gl.createRenderbuffer();
 59     gl.bindRenderbuffer(gl.RENDERBUFFER, this.renderbuffer);
 60     gl.renderbufferStorage(gl.RENDERBUFFER, gl.DEPTH_COMPONENT16, width, height);
 61 	
 62     
 63     //3. Init Frame Buffer
 64     this.framebuffer = gl.createFramebuffer();
 65 	gl.bindFramebuffer(gl.FRAMEBUFFER, this.framebuffer);
 66 	gl.framebufferTexture2D(gl.FRAMEBUFFER, gl.COLOR_ATTACHMENT0, gl.TEXTURE_2D, this.texture, 0);
 67     gl.framebufferRenderbuffer(gl.FRAMEBUFFER, gl.DEPTH_ATTACHMENT, gl.RENDERBUFFER, this.renderbuffer);
 68 	
 69 
 70 	//4. Clean up
 71 	gl.bindTexture(gl.TEXTURE_2D, null);
 72     gl.bindRenderbuffer(gl.RENDERBUFFER, null);
 73     gl.bindFramebuffer(gl.FRAMEBUFFER, null);
 74     
 75     //5. Assign random colors to object in current scene
 76     //for(var i=0, max = this.view.scene.actors.length; i<max; i+=1){
 77     	//this
 78    // }
 79 }

 82 vxlPickerInteractor.prototype.get2DCoords = function(ev){
 83 	var x, y, top = 0, left = 0, obj = this.camera.view.canvas;
 84 
 85 	while (obj && obj.tagName != 'BODY') {
 86 		top += obj.offsetTop;
 87 		left += obj.offsetLeft;
 88 		obj = obj.offsetParent;
 89 	}
 90     
 91     left += window.pageXOffset;
 92     top  += window.pageYOffset;
 93  
 94 	// return relative mouse position
 95 	x = ev.clientX - left;
 96 	y = c_height - (ev.clientY - top); //c_height is a global variable that we maintain in codeview.js
 97                                        //this variable contains the height of the canvas and it updates dynamically
 98                                        //as we resize the browser window.
 99 	
100 	return {x:x,y:y};
101 }

103 vxlPickerInteractor.prototype._compare = function(readout, color){
104     vxl.go.console('PickerInteractor: comparing object '+object.alias+' diffuse ('+Math.round(color[0]*255)+','+Math.round(color[1]*255)+','+Math.round(color[2]*255)+') == readout ('+ readout[0]+','+ readout[1]+','+ readout[2]+')');
105     return (Math.abs(Math.round(color[0]*255) - readout[0]) <= 1 &&
106 			Math.abs(Math.round(color[1]*255) - readout[1]) <= 1 && 
107 			Math.abs(Math.round(color[2]*255) - readout[2]) <= 1);
108 }

110 vxlPickerInteractor.prototype.find = function(coords){
111 	
112 	var gl = this.camera.view.renderer.gl;
113 	//read one pixel
114 	var readout = new Uint8Array(1 * 1 * 4);
115 	gl.bindFramebuffer(gl.FRAMEBUFFER, this.framebuffer);
116 	gl.readPixels(coords.x,coords.y,1,1,gl.RGBA,gl.UNSIGNED_BYTE,readout);
117 	gl.bindFramebuffer(gl.FRAMEBUFFER, null);
118 
119     var found = false;
120     var scene = this.view.scene;
121     if (this.hitPropertyCallback == undefined) {alert('The picker needs an object property to perform the comparison'); return;}
122 
123     for(var i = 0, max = scene.actors.length; i < max; i+=1){
124         var ob = scene.actors[i];
125         if (ob.name == 'floor') continue;
126                 
127         var property  = this.hitPropertyCallback(ob);
128     
129         if (this._compare(readout, property)){
130             var idx  = this.plist.indexOf(ob);
131             if (idx != -1){
132                 this.plist.splice(idx,1);
133                 if (this.removeHitCallback){
134                     this.removeHitCallback(ob); 
135                 }
136             }
137             else {
138                 this.plist.push(ob);
139                 if (this.addHitCallback){
140                     this.addHitCallback(ob); 
141                 }
142             }
143             found = true;
144             break;
145         }
146     }
147     draw();
148     return found;
149 };

151 vxlPickerInteractor.prototype.stop = function(){
152     if (this.processHitsCallback != null && this.plist.length > 0){
153         this.processHitsCallback(this.plist);
154     }
155     this.plist = [];
156 }
157 
158 vxlPickerInteractor.prototype.onMouseUp = function(ev){
159 	if(!ev.shiftKey){
160 		this.stop();
161 	}
162 }
163 
164 
165 vxlPickerInteractor.prototype.onMouseDown = function(ev){
166 	var coords = this.get2DCoords(ev);
167 	this.picking = this.find(coords);
168 	if(!this.picking){
169 		this.stop();
170 	}
171 }
172 
173 vxlPickerInteractor.prototype.onMouseMove = function(ev){
174 	this.lastX = this.x;
175 	this.lastY = this.y;
176 	this.x = ev.clientX;
177 	this.y = ev.clientY;
178 	var dx = this.x - this.lastX;
179 	var dy = this.y - this.lastY;
180 	if(this.picking && this.moveCallback){
181 		this.moveCallback(this, dx,dy);
182 	}
183 }
