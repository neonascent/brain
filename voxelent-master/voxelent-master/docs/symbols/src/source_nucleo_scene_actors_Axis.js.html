 18 vxl.def.model.axis = new vxlModel();
 19 vxl.def.model.axis.load('axis', {
 20                     	"vertices": [	-1, 0, 0, 	 1, 0, 0, 	 0,-2, 0,	 0, 2, 0,	 0, 0,-1,	 0, 0, 1	],

 23                     	});

 25 
 26 vxlAxis.prototype 				= new vxlActor();
 27 vxlAxis.prototype.constructor 	= vxlAxis;
 28 
 29 /**
 30  * @class
 31  * @constructor
 32  * @author Diego Cantor
 33  */
 34 function vxlAxis() {
 35 	vxlActor.call(this,vxl.def.model.axis);
 36 	this.centre 	= [0,0,0];
 37 	this.mode 		= vxl.def.actor.mode.WIREFRAME;
 38 	this.visible 	= false;
 39 	this.toy     	= true;
 40 };
 41 
 42 /**
 43 * Sets the centre of the axis actor in the scene
 44 */
 45 vxlAxis.prototype.setCentre = function (ctr){
 46     var x = ctr[0];
 47 	var y = ctr[1];
 48 	var z = ctr[2];
 49 	
 50 	this.centre[0] = x;
 51 	this.centre[1] = y;
 52 	this.centre[2] = z;
 53     
 54     var ver = this.model.vertices;
 55 	
 56 	ver[0] = x-1;
 57 	ver[1] = y;
 58 	ver[2] = z;
 59 	
 60 	ver[3] = x+1;
 61 	ver[4] = y;
 62 	ver[5] = z;
 63 	
 64 	ver[6] = x;
 65 	ver[7] = y-2;
 66 	ver[8] = z;
 67 	
 68 	ver[9] = x;
 69 	ver[10] = y+2;
 70 	ver[11] = z;
 71 	
 72 	ver[12] = x;
 73 	ver[13] = y;
 74 	ver[14] = z-1;
 75 	
 76 	ver[15] = x;
 77 	ver[16] = y;
 78 	ver[17] = z+1;
 79 };
