 16 /**
 17  * One of the main classes of Voxelent's Nucleo
 18  * A renderer object encapsulates most of the low level calls to WebGL.
 19  * It is here where Nucleo obtains a reference to the WebGL context.
 20  * 
 21  * It is also here here (and in the Actor class) where all the attributes and uniforms are passed
 22  * to the rendering program.
 23  * 
 24  * @class
 25  * @constructor
 26  * @author Diego Cantor
 27  */
 28 function vxlRenderer(vw){
 29     
 30 	this.view       	= vw;
 31 	this.renderRate 	= vxl.def.renderer.rate.NORMAL;
 32 	this.mode       	= vxl.def.renderer.mode.TIMER;
 33     this.gl         	= this.getWebGLContext();
 34     this.pm        	    = new vxlProgramManager(this.gl);
 35     this.transforms 	= new vxlTransforms(vw);
 36     this.fps            = 0;
 37     this.currentProgram = undefined;
 38     this.engine 		= new vxlRenderEngine(this);
 39     
 40     
 41     this._time          = 0;
 42     this._startDate     = 0;
 43     this._running       = false;
 44     this._clearColor    = undefined;
 45     this._renderTarget  = undefined;
 46     
 47     this._knownPrograms = {};
 48     this._enforce       = false; //to enforce the program or not
 49     this.setProgram(vxl.go.essl.lambert);
 50     
 51 }

 53 /**
 54  * Tries to obtain a WebGL context from the canvas associated with the view to which this
 55  * renderer belongs to.
 56  * @TODO: Review depth test and blending functions maybe these should be configurable.
 57  */
 58 vxlRenderer.prototype.getWebGLContext = function(){
 59 	
 60 	var WEB_GL_CONTEXT = null;
 61 	var canvas     = this.view.canvas;
 62 	this.width     = canvas.width;
 63 	this.height    = canvas.height;
 64 	
 65 	var names      = ["webgl", "experimental-webgl", "webkit-3d", "moz-webgl"];
 66 	
 67 	for (var i = 0; i < names.length; ++i) {
 68 		try {
 69 			WEB_GL_CONTEXT = canvas.getContext(names[i]);
 70 		} 
 71 		catch(e) {}
 72 		if (WEB_GL_CONTEXT) {
 73 			break;
 74 		}
 75 	}
 76 	if (WEB_GL_CONTEXT == null) {
 77 		//@TODO: print a nicer jquery  alert
 78 		alert("Sorry: WebGL is not available on this browser. Have you tried the newest version of Firefox, Chrome or Safari?"); 
 79 		return;
 80 	}
 81 	else {
 82 		this._initializeGLContext(WEB_GL_CONTEXT);
 83 		
 84 	}
 85     return WEB_GL_CONTEXT;
 86 };

 88 /**
 89  * Initializes the WebGL context
 90  *
 91  * @private 
 92  */
 93 vxlRenderer.prototype._initializeGLContext = function(gl){
 94     gl.enable(gl.DEPTH_TEST);
 95     gl.enable(gl.BLEND);
 96     gl.blendFunc(gl.SRC_ALPHA, gl.ONE_MINUS_SRC_ALPHA);
 97     gl.depthFunc(gl.LESS);
 98     gl.pixelStorei(gl.UNPACK_FLIP_Y_WEBGL, true);
 99 };

102 /**
103  * Tries to add a new program definition to this renderer
104  * @param {vxlProgram} program an instance of a vxlProgram object or one of its descendents
105  * @param {vxlEngine} engine (optional) a new engine
106  */
107 vxlRenderer.prototype.setProgram = function(program,engine,forceIt){
108     
109     
110     if (this._enforce && program.ID != this.currentProgram.ID){
111         throw new vxlRendererException("The current program is being enforced. Please use vxlRenderer.releaseProgram first");
112     }
113     
114     if (forceIt != undefined && forceIt == true){
115         this._enforce = true;
116     }
117     else{
118         this._enforce = false;
119     }
120     
121     
122     //Check if the program has be instantiated before
123     if (this._knownPrograms[program.ID] == undefined || this._enforce){
124         this._knownPrograms[program.ID]= program;
125     }
126     
127     var prg = this._knownPrograms[program.ID];
128     var pm = this.pm;
129 
130 	if (!pm.isRegistered(prg.ID)){
131 		pm.register(prg);
132 	}
133 	
134 	if (!pm.isLoaded(prg.ID)){
135 		pm.loadProgram(prg.ID);
136 	}
137 	
138 	pm.useProgram(prg.ID);
139 	pm.loadDefaults();
140 	
141 	this.currentProgram = prg;
142 	
143 	if (engine != undefined && engine != this.engine){
144 	   this.setEngine(engine);
145 	}
146 	
147 	
148 };

150 /**
151  * When the current program is being enforced by the renderer (see setProgram), any
152  * subsequent call to setProgram will be unsuccessful. So for instance, actors
153  * who want to use a different program to be rendered would not be able to do so.
154  * 
155  * This method releases the current program from being enforced 
156  */
157 vxlRenderer.prototype.releaseProgram = function(){
158     this._enforce = false;
159 }

161 /**
162  * Sets the current rendering engine. 
163  * 
164  * @param {vxlEngine} engine The engine to be used.  
165  */
166 vxlRenderer.prototype.setEngine = function(engine){
167     if (engine != null && engine != undefined){
168         this.engine = engine
169     }

172     }

176 /**
177  * Clears the rendering context
178  */
179 vxlRenderer.prototype.clear = function(){
180     this.gl.clearColor(this._clearColor[0], this._clearColor[1], this._clearColor[2], 1.0);
181 	this.gl.clear(this.gl.COLOR_BUFFER_BIT | this.gl.DEPTH_BUFFER_BIT);
182 	this.gl.viewport(0, 0, this.view.canvas.width, this.view.canvas.height); //@TODO: Think about dividing view ports for multi-view apps - March 19/2012
183 	this.transforms.calculatePerspective();
184 };

186 /**
187  * Sets the rendering mode. Options are in vxl.def.renderer.mode
188  * This method updates the rendering mode and tries to restart the rendering process
189  * @param {String} mode the mode to set
190  */
191 vxlRenderer.prototype.setMode = function(mode){
192     this.stop();
193     this.mode = mode;
194     this.start();   
195 };

197 /**
198  * Starts the renderer
199  */
200 vxlRenderer.prototype.start = function(){
201     
202     this._running = true;
203     this._startDate = new Date().getTime();
204     this._time  = 0;
205     
206 	if(this.mode == vxl.def.renderer.mode.TIMER){
207 		vxl.go.console('Renderer: starting rendering for view ['+this.view.name+'] at '+this.renderRate+ 'ms');
208 		this._timeUp();
209 		//this.timerID = setInterval((function(self) {return function() {self.render();}})(this),this.renderRate); 
210 	}
211 	else if(this.mode == vxl.def.renderer.mode.ANIMFRAME){
212 	    vxl.go.console('Renderer: starting rendering at the fastest speed',true);
213 		vxl.go.renderman.render();
214 	}
215 };

217 /**
218  * Implements a self adjusting timer
219  * @see <a href="http://www.sitepoint.com/creating-accurate-timers-in-javascript/">Creating accurate timers in JavasScript</a>
220  * @private   
221  */
222 vxlRenderer.prototype._timeUp = function(){
223     if (!this._running) return;
224     
225     this.render();
226     
227     if (this._time == this.renderRate * 100){  
228         this._time = 0;
229         this._startDate = new Date().getTime();
230     }
231     
232     this._time += this.renderRate;
233 
234     var diff = (new Date().getTime() - this._startDate) - this._time;
235     
236     if (diff > this.renderRate) diff = 0; //ignore it
237     
238     setTimeout((function(self){
239         return function(){
240             self._timeUp();
241         }
242     })(this), this.renderRate - diff);
243 }
244 /**
245  * Stops the renderer
246  */
247 vxlRenderer.prototype.stop = function(){
248 	if (this.mode == vxl.def.renderer.mode.TIMER){
249 		//clearInterval(this.timerID);
250 		this._running = false;
251 	}
252 	else if (this.mode == vxl.def.renderer.mode.ANIMFRAME){
253 		vxl.go.renderman.cancel();
254 	}
255 };

257 /**
258  * Sets the rendering rate in ms
259  * @param {Number} /test-baking.htmlrate the new rendering rate in milliseconds
260  */
261 vxlRenderer.prototype.setRenderRate = function(rate){ //rate in ms
262     
263     if (rate == undefined || rate <=0){ 
264         throw 'vxlRenderer.setRenderRate: the rate cannot be zero or undefined';
265     } 
266     
267     if (this.mode == vxl.def.renderer.mode.ANIMFRAME){
268         throw 'vxlRenderer.setRenderRate: if the mode is ANIMFRAME render rate is irrelevant';
269     }
270       
271 	this.stop();
272 	this.renderRate = rate;
273 	this.start();
274     
275     vxl.go.console('Renderer: view['+this.view.name+'], render rate = ' + rate,true);
276 
277 };

279 /**
280  * Sets the color used to clear the rendering context
281  * @param {Number, Array, vec3} r it can be the red component, a 3-dimensional Array or a vec3 (glMatrix)
282  * @param {Number} g if r is a number, then this parameter corresponds to the green component
283  * @param {Number} b if r is a number, then this parameter corresponds to the blue component
284  * @see vxlView#setBackgroundColor
285  */
286 vxlRenderer.prototype.clearColor = function(r,g,b){
287     var cc = vxl.util.createArr3(r,g,b);
288     this._clearColor = cc;
289 	this.gl.clearColor(cc[0], cc[1], cc[2], 1.0);
290 };

300 /**
301  * Renders the scene using the current engine
302  */
303 vxlRenderer.prototype.render = function(){
304     var engine = this.engine, scene = this.view.scene, start = undefined, elapsed = undefined;
305     
306     this.clear();                   //clear the canvas
307     engine.allocate(scene);	    //allocate memory for actors added since last rendering
308     
309     start = new Date().getTime();
310     engine.render(scene);
311     elapsed = new Date().getTime() - start;
312     
313     engine.deallocate(scene);     //deallocate memory if necessary
314     
315     // calculating FPS metric
316     if(elapsed >0){
317         this.fps = Math.round((this.fps * 0.80 + (1000.0/elapsed) * 0.2)* 100)/100;
318     }
319     
320 };

322 /**
323  * Reallocates the actors marked as dirty, without requiring rerendering. This mechanism allows
324  * to update the GL buffers for dirty actors. 
325  */
326 vxlRenderer.prototype.reallocate = function(){
