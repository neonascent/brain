 17 vxlExternalEngine.prototype = new vxlEngine();
 18 vxlExternalEngine.prototype.constructor = vxlExternalEngine;
 19 /**
 20  * 
 21  * @param {Object} allocate
 22  * @param {Object} render
 23  * @param {Object} deallocate
 24  */
 25 function vxlExternalEngine(renderer, allocate, render, deallocate){
 26     this.renderer = renderer;
 27     this.allocateCallback = allocate;
 28     this.renderCallback = render;
 29     this.deallocateCallback = deallocate;
 30 }
 31 
 32 
 33 vxlExternalEngine.prototype.allocate = function(scene){
 34     this.allocateCallback(this.renderer, scene);
 35 };
 36 
 37 
 38 vxlExternalEngine.prototype.render = function(scene){
 39     this.renderCallback(this.renderer, scene);  
 40 };
 41 
 42 vxlExternalEngine.prototype.deallocate = function(scene){
 43     this.deallocateCallback(this.renderer, scene);  
 44 };
