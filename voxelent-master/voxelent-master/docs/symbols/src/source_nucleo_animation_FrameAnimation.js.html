 16 /*
 17 * Idea: to use a lightweight pattern. A pool of vxlModels that are reused.
 18 * Every frame the information is copied to the buffers, instead of saving as many gl vbos as models
 19 */
 20 /**
 21  * Provides frame-to-frame animation
 22  * 
 23  * @class Manages a frame-to-frame animation
 24  * @constructor
 25  * @param map  JSON object where each property name is one frame and each property value
 26  * is a list of actors 
 27  * var map = {"frame1":["actor1","actor2"], "frame2":["actor3","actor4"]}
 28  * 
 29  */
 30 function vxlFrameAnimation(map){
 31 	this.scene             = null;
 32 
 33 	this.actorByFrameMap   = [];
 34 	this.activeFrame       = 1;
 35 	this.mark              = 1;
 36 	this._running          = false;
 37     this.frameCount        = 0;
 38     this.renderRate        = 500;
 39     
 40     this._startDate        = undefined;
 41     this._time             = 0;
 42     
 43     this._setup(map);
 44     if (vxl.c.animation == null) vxl.c.animation = this;
 45 };
 46 
 47 /**
 48  * The actor will appear in the indicated frame of this animation
 49  * @param {Number} frame the frame
 50  * @param {String} actorName the name of the actor. It must exist.
 51  */
 52 vxlFrameAnimation.prototype.addActorToFrame = function(frame,actorName){
 53 	if (typeof(this.actorByFrameMap[frame])=='undefined'){
 54 		this.actorByFrameMap[frame] = new Array();
 55 	}
 56 	if (this.actorByFrameMap[frame].indexOf(actorName) == -1){
 57 		this.actorByFrameMap[frame].push(actorName);
 58 	}
 59     if (frame>this.frameCount) this.frameCount = frame;
 60 };
 61 
 62 /**
 63  * Map is a JSON object where each property name is one frame and each property value
 64  * is a list of actors 
 65  * 
 66  * var map = {"frame1":["actor1","actor2"], "frame2":["actor3","actor4"]}
 67  * @private
 68  */
 69 vxlFrameAnimation.prototype._setup = function(map){
 70 	this.activeFrame = 1;
 71 
 72 	for (var f in map){
 73 		var actorList = map[f];
 74 		var frame = parseInt(f.substr(5,f.length));
 75 		for(var i=0, max = actorList.length; i < max; i+=1){
 76 			this.addActorToFrame(frame,actorList[i]);
 77 		}
 78 	}
 79 	vxl.go.console('FrameAnimation: Setup finished.');
 80 };
 81 
 82 /**
 83  * Starts the animation loop
 84  * @param {Number} rate the framerate for the animation (optional)
 85  */
 86 vxlFrameAnimation.prototype.start = function(rate){
 87 	if (this.scene == null) throw 'FrameAnimation: the animation is not associated with any scene. Please use scene.setFrameAnimation method';
 88 
 89     this._startDate = new Date().getTime();
 90     this._time  = 0;
 91     this._running = true;
 92     
 93     if (rate != undefined && rate >=0){
 94     	this.renderRate = rate;
 95     }
 96     
 97     this._timeUp();
 98 };
 99 
100 
101 /**
102  * Implements a self-adjusting timer
103  * @see http://www.sitepoint.com/creating-accurate-timers-in-javascript/
104  * @private 
105  */
106 vxlFrameAnimation.prototype._timeUp = function(){
107     if (!this._running) return;
108     
109     this.nextFrame();
110     
111     if (this._time == this.renderRate * 100){  //for long running animations
112         this._time = 0;
113         this._startDate = new Date().getTime();
114     }
115     
116     this._time += this.renderRate;
117 
118     var diff = (new Date().getTime() - this._startDate) - this._time;
119     
120     if (diff > this.renderRate) diff = 0; //ignore it
121     
122     setTimeout((function(self){
123         return function(){
124             self._timeUp();
125         }
126     })(this), this.renderRate - diff);
127     
128 };
129 
130 /**
131  * Stops the animation loop
132  */
133 vxlFrameAnimation.prototype.stop = function(){
134     this._running = false;
135 };
136 
137 vxlFrameAnimation.prototype.setFrameRate = function(rate){
138 	if (rate <=0) return;
139 	this.stop();
140 	this.renderRate = rate;
141 	this.start();
142 };

145 /**
146  * Selects the actors that will be visible in the current frame
147  */
148 vxlFrameAnimation.prototype.update = function(){
149 	
150 	if (this.scene == null) throw 'FrameAnimation: the animation is not associated with any scene. Please use scene.setFrameAnimation method';
151 	
152 	
153 	//we hide them all first
154 	this.scene.setPropertyForAll('visible', false);  
155 	 
156 	//then we decide which ones are visible
157 	var currentActors = this.actorByFrameMap[this.activeFrame]
158 	var NUM = currentActors.length;
159 	for (var i=0; i<NUM; i++){
160 		var actor = this.scene.getActorByName(currentActors[i]);
161 		if (actor != null){
162 			actor.setVisible(true);
163 		}
164 	}
165 };
166 
167 /**
168  * Verifies if the frame number passed as parameter is in the range of the current animation
169  * @param f a frame number
170  * @returns true if the number passed as parameter is a valid frame number, false otherwise
171  */
172 vxlFrameAnimation.prototype.isValidFrame = function(f){
173  return (f>=1 && f<= this.frameCount);
174 };
175 
176 /**
177  * Moves the animation to the next valid frame. If the activeFrame is the last frame in the animation, then
178  * the animation is reset to the first frame.
179  */
180 vxlFrameAnimation.prototype.nextFrame = function(){
181 	if (this.activeFrame < this.frameCount){
182 		this.activeFrame++;
183 	}
184 	else{
185 		this.activeFrame = 1;
186 	}
187 };
188 
189 /**
190  * Gets the next n valid frames. Works as a circular buffer.
191  */
192 vxlFrameAnimation.prototype.getNextFrames = function(n){
193 	var list = [];
194 	var c = this.activeFrame;
195 	if (n> this.frameCount) n = this.frameCount;
196 	for (var i=1; i <=n; i++){
197 		var next = c + i;
198 		if (this.isValidFrame(next)){
199 			list.push(next);
200 		}
201 		else{
202 			list.push(next-this.frameCount);
203 		}
204 	}
205 	vxl.go.console('Animation: next frames: ' + list);
206 	return list;
207 };
208 
209 /**
210  * Gets the previous n frames. Works as a circular buffer.
211  */
212 vxlFrameAnimation.prototype.getPreviousFrames = function(n){
213 	var list = [];
214 	var c = this.activeFrame;
215 	if (n> this.frameCount) n = this.frameCount;
216 	for (var i=1; i <=n; i++){
217 		var previous = c - i;
218 		if (this.isValidFrame(previous)){
219 			list.push(previous);
220 		}
221 		else{
222 			list.push(this.frameCount+previous);
223 		}
224 	}
225 	vxl.go.console('Animation: previous frames: ' + list);
226 	return list;
227 };
228 
229 /**
230  * Sets f as the active frame
231  * @param f the frame to set as active
232  */
233 vxlFrameAnimation.prototype.setFrame = function (f){
234 	if (f>=1 && f <= this.frameCount){
235 		this.activeFrame = f;
236 		this.render();
237 	}
238 };
