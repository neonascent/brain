 18 vxlTrackballCameraInteractor.prototype = new vxlCameraInteractor();

 20 function vxlTrackballCameraInteractor(c){
 21 	this.MOTION_FACTOR = 10.0;
 22 	this.task = vxl.def.camera.task.NONE;
 23 	this.x = 0;
 24 	this.y = 0;
 25 	this.lastX = 0;
 26 	this.lastY = 0;
 27 	this.ctrlPressed = false;
 28 	this.altPressed = false;
 29 	this.keyPressed = 0;
 30 	this.button = -1;
 31 	this.dragging = false;
 32 	this.connectTo(c);
 33 }

 50 vxlTrackballCameraInteractor.prototype.onMouseDown = function(ev){
 51 	this.x = ev.clientX;
 52 	this.y = ev.clientY;
 53 	this.dragging = true;
 54 	this.button = ev.button;
 55 };
 56 
 57 vxlTrackballCameraInteractor.prototype.onMouseMove = function(ev){
 58 
 59 	this.lastX = this.x;
 60 	this.lastY = this.y;
 61 	
 62 	this.x = ev.clientX;
 63     this.y = ev.clientY;
 64 	
 65 
 66 	if (!this.dragging) return;
 67 	
 68 	
 69 	this.ctrlPressed = ev.ctrlKey;
 70 	this.altPressed = ev.altKey;
 71 	
 72 	var dx = this.x - this.lastX;
 73 	var dy = this.y - this.lastY;
 74 	
 75 	if (this.button == 0) { 
 76 		if(this.ctrlPressed){
 77 			this.dolly(dy);
 78 		}
 79 		else{ 
 80 			this.rotate(dx,dy);
 81 		}
 82 	}
 83 
 84 	this.lastX = this.x;
 85     this.lastY = this.y; 
 86 
 87 };

 89 vxlTrackballCameraInteractor.prototype.onKeyDown = function(ev){
 90 	var camera = this.camera;
 91 	
 92 	this.keyPressed = ev.keyCode;
 93 	this.ctrlPressed = ev.ctrlKey;
 94 	
 95 	if (!this.ctrlPressed){
 96 		if (this.keyPressed == 38){
 97 			camera.changeElevation(10);
 98 			camera.status('elevation up');
 99 		}
100 		else if (this.keyPressed == 40){
101 			camera.changeElevation(-10);
102 			camera.status('elevation down');
103 		}
104 		else if (this.keyPressed == 37){
105 			camera.changeAzimuth(-10);
106 			camera.status('azimuth left');
107 		}
108 		else if (this.keyPressed == 39){
109 			camera.changeAzimuth(10);
110 			camera.status('azimuth right');
111 		}
112 		//just to try picking. later on do it better
113 		else if (this.keyPressed = 80) {
114 			this.picking =!camera.view.actorManager.picking;
115 			camera.view.actorManager.setPicking(this.picking);
116 			
117 		}
118 	}
119 	else if(this.ctrlPressed && this.keyPressed !=17) {
120 		var px = 0;
121 		var py = 0;
122 		vxl.go.console(ev);
123 		if (this.keyPressed == 38){
124 			py = 10;
125 		}
126 		else if (this.keyPressed == 40){
127 			py = -10;
128 		}
129 		else if (this.keyPressed == 37){
130 			px = -10;
131 		}
132 		else if (this.keyPressed == 39){
133 			px = 10;
134 		}
135 		if(px != 0 || py !=0){
136 			this.pan(px,py);
137 		}
138 	}
139 };

147 vxlTrackballCameraInteractor.prototype.dolly = function(value){
148 	this.task = vxl.def.camera.task.DOLLY;
149 	var camera = this.camera;
150 	var dv = 2 * this.MOTION_FACTOR * value / camera.view.canvas.height;
151 	
152 	camera.dolly(Math.pow(1.1,dv));
153 	
154 
155 	camera.refresh();
156 	
157 };
158 
159 vxlTrackballCameraInteractor.prototype.rotate = function(dx, dy){
160 	this.task = vxl.def.camera.task.ROTATE;
161 	
162 	var camera = this.camera;
163 	var canvas = camera.view.canvas;
164 	
165 	var delta_elevation = -20.0 / canvas.height;
166 	var delta_azimuth = -20.0 / canvas.width;
167 				
168 	var nAzimuth = dx * delta_azimuth * this.MOTION_FACTOR;
169 	var nElevation = dy * delta_elevation * this.MOTION_FACTOR;
170 	
171 	camera.changeAzimuth(nAzimuth);
172 	camera.changeElevation(nElevation);
173 	camera.refresh();
174 	
175 };
176 
177 vxlTrackballCameraInteractor.prototype.pan = function(dx,dy){
178 	this.task = vxl.def.camera.task.PAN;
179 	
180 	var camera = this.camera;
181 	var canvas = camera.view.canvas;
182 	var scene = camera.view.scene;
183 	
184 	var dimMax = Math.max(canvas.width, canvas.height);
185 	
186 	var deltaX = 1 / dimMax;
187 	var deltaY = 1 / dimMax;
188 				
189 	var ndx = dx * deltaX * this.MOTION_FACTOR * scene.bb.max();
190 	var ndy = dy * deltaY * this.MOTION_FACTOR * scene.bb.max();
191 
192 	camera.pan(ndx,ndy);
193 	camera.refresh();
194 	
195 };
