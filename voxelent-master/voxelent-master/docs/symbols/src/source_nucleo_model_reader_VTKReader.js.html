 17 /**
 18  * Reads a VTK file and creates the respective JSON(s) file(s).
 19  * @class Reads VTK files (ASCII)  and creates JSON objects that can be used as models
 20  * @constructor 
 21  */
 22 function vxlVTKReader(scene){
 23 
 24     this.scene      = scene; //the scene that will be updated
 25     this.ARRAY_SIZE = 65536*3;
 26     this.vertices   = [];
 27     this.indices    = [];
 28     this.normals    = [];
 29     this.scalars    = [];
 30     this.colors     = [];
 31     
 32 
 33     this.mode = "SOLID";
 34 
 35     this.tags = {
 36         NOWHERE             : 0,    
 37         POINTS              : 1,
 38         LINES               : 2,
 39         POLYGONS            : 3,
 40         POINT_DATA          : 4,
 41         NORMALS             : 5,
 42         CELL_DATA           : 6,
 43         TEXTURE_COORDINATES : 7,
 44         SCALARS             : 8,
 45         LOOKUP_TABLE        : 9,
 46         COLOR_SCALARS       : 10
 47     };
 48     
 49     this.parts = [];
 50     
 51     vxl.go.notifier.publish(vxl.events.READER_DONE, this);
 52 };

 54 /**
 55  * Verifies if the HTML5 API is available. 
 56  */
 57 vxlVTKReader.prototype.isSupported = function(){
 58     return (window.File && window.FileReader && window.FileList && window.Blob);
 59 };

 62 /**
 63  * @param {File} file an HTML5 File object
 64  */
 65 vxlVTKReader.prototype.read = function(file){
 66     var vtkReader = this;
 67     var filename = file.name;
 68     
 69     if (filename.length - 4 !== filename.indexOf('.vtk')){
 70         throw 'vxlVTKReader.read: the file '+file+' is not a VTK file';
 71     }
 72     
 73     modelname = filename.replace(/\.[^/.]+$/, "");
 74     var reader = new FileReader();
 75      
 76     reader.onload = function(event){
 77         document.body.style.cursor = 'default';
 78         var contents = event.target.result.trim();
 79         var lines = contents.split(/\r\n|\r|\n/);
 80         vtkReader._parse(lines);  
 81         vtkReader._processIndexBlocks(modelname);
 82         vxl.go.notifier.fire(vxl.events.READER_DONE, vtkReader);
 83     };
 84     
 85     document.body.style.cursor = 'wait';
 86     reader.readAsText(file);
 87 };

 90 /**
 91  * If the VTK file exists in memory as a String (text) then we can
 92  * use this method to parse it. 
 93  * 
 94  * @param {String} name the name of the object
 95  * @param {String} text the retrieved object contents
 96  */
 97 vxlVTKReader.prototype.parseText = function(name, text){
 98     document.body.style.cursor = 'wait'; 
 99     var lines = text.split(/\r\n|\r|\n/);
100     this._parse(lines);  
101     this._processIndexBlocks(name);
102     vxl.go.notifier.fire(vxl.events.READER_DONE, this);
103     document.body.style.cursor = 'default';
104 };

106 /**
107  * Reads the file line by line and populates the respective arrays
108  */
109 vxlVTKReader.prototype._parse = function(lines){
110    this.outputfile = '';
111    this.numBlocks = 0;
112    this.location = 'NOWHERE';
113    var linenumber = 0;
114    for(var linenumber=0; linenumber<lines.length; linenumber++)
115    {
116     
117        try
118             {
119                 if (lines[linenumber].indexOf('POINTS') == 0)
120                 {
121                     console.log(lines[linenumber]);
122                     this.location = this.tags.POINTS;
123                     continue;
124                 }
125                 else if (lines[linenumber].indexOf('LINES') == 0)
126                 {
127                     console.log(lines[linenumber]);
128                     this.location = this.tags.LINES;
129                     this.mode = "LINES";
130                     continue;
131                 }
132                 else if (lines[linenumber].indexOf('POLYGONS')==0)
133                 {
134                     console.log(lines[linenumber]);
135                     this.location = this.tags.POLYGONS;
136                     continue;
137                 }
138                 else if (lines[linenumber].indexOf('POINT_DATA')==0)
139                 {                   
140                     this.location = this.tags.POINT_DATA;
141                     continue;
142                 }
143                 else if (lines[linenumber].indexOf('NORMALS')==0)
144                 {      
145                     console.log(lines[linenumber]);             
146                     this.location = this.tags.NORMALS;
147                     continue;
148                 }
149                 else if (lines[linenumber].indexOf('CELL_DATA')==0)
150                 {      
151                     console.log(lines[linenumber]);             
152                     this.location = this.tags.CELL_DATA;
153                     continue;
154                 }
155                 else if (lines[linenumber].indexOf('TEXTURE_COORDINATES')==0)
156                 {      
157                     console.log(lines[linenumber]);             
158                     this.location = this.tags.TEXTURE_COORDINATES;
159                     continue;
160                 }
161                 else if (lines[linenumber].indexOf('SCALARS')==0)
162                 {      
163                     console.log(lines[linenumber]);             
164                     this.location = this.tags.SCALARS;
165                     continue;
166                 }
167                 else if (lines[linenumber].indexOf('LOOKUP_TABLE')==0)
168                 {      
169                     console.log(lines[linenumber]);             
170                     this.location = this.tags.LOOKUP_TABLE;
171                     continue;
172                 }   
173                 else if (lines[linenumber].indexOf('COLOR_SCALARS')==0)
174                 {      
175                     console.log(lines[linenumber]);             
176                     this.location = this.tags.COLOR_SCALARS;
177                     continue;
178                 }           
179                 // -------------------
180                 else if(this.location == this.tags.POINTS)
181                 {
182                     var v = lines[linenumber].trim().split(' ');
183                     if (v == "") continue;
184                     for (var i=0; i<v.length; i++)
185                     {
186                          this.vertices.push(parseFloat(v[i]));
187                     }
188                 }
189                 else if(this.location == this.tags.LINES)
190                 {
191                     var tt = lines[linenumber].trim().split(' ');
192                     if (tt == "") continue;
193                     if(tt.length>0 && tt[0] == '2')
194                     {
195                         this.indices.push(parseInt(tt[1]));
196                         this.indices.push(parseInt(tt[2]));
197                     }
198                 }
199                else if(this.location == this.tags.POLYGONS) //they have to be triangles
200                 {
201                     var tt = lines[linenumber].trim().split(' ');
202                     if (tt=="") continue; 
203                     if(tt.length>0 && tt[0] != '3')
204                     {
205                         throw "Not triangles here";
206                     }
207                     for(var i= 1; i<tt.length; i++)
208                     {
209                         this.indices.push(parseInt(tt[i]));
210                     }
211                 }
212                 else if(this.location == this.tags.LOOKUP_TABLE)
213                 {
214                     if(lines[linenumber].indexOf('LOOKUP_TABLE')==0)
215                         continue;
216                     else
217                     {
218                         var pd = lines[linenumber].trim().split(' ');
219                         if (pd=="") continue;
220                         for(var i=0; i<pd.length; i++)
221                         {
222                             this.scalars.push(parseFloat(pd[i]));
223                         }
224                    }
225                 }
226                 else if(this.location == this.tags.COLOR_SCALARS)
227                 {
228                     var n = lines[linenumber].trim().split(' ');
229                     if (n=="") continue;
230                     for(var i=0; i<n.length; i++)
231                     {
232                         this.colors.push(parseFloat(n[i]));
233                     }
234                 }
235                 else if(this.location == this.tags.NORMALS)
236                 {
237                     var n = lines[linenumber].trim().split(' ');
238                     if (n=="") continue;
239                     for(var i=0; i<n.length; i++)
240                     {
241                         this.normals.push(parseFloat(n[i]));
242                     }
243                 }
244            } // end try
245         catch(err)
246             {
247             console.log('Error while processing line '+ linenumber.toString());
248             //console.log(lines) // what is this for??
249             throw err;
250             }
251     }// end foor loop
252 };

254 /**
255  * Divides the calculated indices into blocks
256  * @private
257  */
258 vxlVTKReader.prototype._processIndexBlocks = function(filename){
259    this.numBlocks = 0;
260    var v_count =  this.vertices.length/3;
261    var n_count =  this.normals.length/3;
262    var ii_count = this.indices.length;
263    var i_count =  ii_count/3;
264    var c_count =  this.colors.length/3;
265    var pd_count = this.scalars.length;
266  
267    console.log('vertices:\t' + v_count.toString() +'\nnormals:\t' + n_count.toString() + '\nindices:\t' + ii_count.toString() +'\ntriangles:\t' + i_count.toString() +
268              '\nscalars:\t' + pd_count.toString() + '\ncolors:\t'+ c_count.toString()+'\n');
269  
270     this.numBlocks = parseInt(ii_count/this.ARRAY_SIZE);
271     if (ii_count % this.ARRAY_SIZE != 0)
272         this.numBlocks  = this.numBlocks + 1;
273     console.log( 'Number of Blocks: ' + this.numBlocks.toString());
274     
275     for(var i=0; i<this.numBlocks; i++){
276         this._processBlock(i, filename);
277     }
278 };

280 /**
281  * Process the selected block of indices
282  * @private
283  */
284 vxlVTKReader.prototype._processBlock = function(blockID, filename){
285     var fid = (blockID + 1).toString();
286     var blockname = "";
287     
288     if (this.numBlocks == 1)
289         blockname = filename ;
290     else
291         blockname = filename + '_' + fid;
292         
293     var block  = this._weaveBlock(blockID);
294     this._writeJSON(blockname, block);
295     console.log('Block ['+ fid +'] processed,  output: '+ blockname);
296 };
297 
298 /**
299  * Calculates the new index array for the block in question
300  */
301 vxlVTKReader.prototype._weaveBlock = function(blockID){
302 
303     var lowerBound = this.ARRAY_SIZE*blockID;
304     var upperBound = this.ARRAY_SIZE*(blockID+1);
305 
306     if (upperBound > this.indices.length){
307         upperBound = this.indices.length;
308     }
309     
310     var newindex = {};
311     
312     var block = {
313         vertices: [],   //new vertex array
314         indices : [],   //new index array
315         scalars : [],   //new scalar array
316         colors  : []   //new color array
317     };
318     
319     var hasPointData = (this.scalars.length>0);
320     var hasColorData = (this.colors.length>0);
321     
322     
323     // Set of indices to be processed
324     var aux = this.indices.slice(lowerBound, upperBound);
325     
326     taux = aux.length;
327     var nidx = -1
328     
329     
330     console.log('Weaving block #' + (blockID+1) + '  ['+ lowerBound+','+upperBound+']');
331     
332     //for each index to be processed
333     for (var i=0; i<taux; i+=1){ 
334         //if index hasn't been mapped
335         var oidx = aux[i];
336         if (newindex[oidx] == undefined){
337             nidx++;
338             // create new index for the old index (incrementally)
339             block.indices.push(nidx);
340             // save in the map for posterior searches
341             newindex[oidx] = nidx;
342             // multiply by three to find the right starting point in the vertex array
343             var index = oidx * 3
344             // add the correspondent vertex into the new position in the new vertex array
345             block.vertices.push(this.vertices[index]);
346             block.vertices.push(this.vertices[index+1]);
347             block.vertices.push(this.vertices[index+2]);
348             // add the correspondent point data if any
349             if (hasPointData){
350                 block.scalars.push(this.scalars[oidx]);
351             }
352             if (hasColorData){
353                 block.colors.push(this.colors[index]);
354                 block.colors.push(this.colors[index+1]);
355                 block.colors.push(this.colors[index+2]);
356             }
357         }
358         else{
359             // if the index was mapped then use it in the new index array
360             block.indices.push(newindex[oidx]);
361         }
362     }
363     delete aux;
364     return block;
365 
366 };

368 /**
369  * Creates the JSON object
370  * @private 
371  */
372 vxlVTKReader.prototype._writeJSON = function(fname,block){
373     
374      var jsonPart = new Object();
375      
376      jsonPart["name"]        = fname;
377      jsonPart["vertices"]    = block.vertices.slice(0);
378      jsonPart["indices"]     = block.indices.slice(0);
379      
380      if (block.scalars.length>0){
381         jsonPart["scalars"]  = block.scalars.slice(0);
382      }
383      
384      if (block.colors.length >0){
385         jsonPart["colors"]   = block.colors.slice(0);
386      }
387      
388      jsonPart["mode"]        = this.mode;
389      
390      this.parts.push(jsonPart);
391 };
