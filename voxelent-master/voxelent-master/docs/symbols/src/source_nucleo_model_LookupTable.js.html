1 /**
  2  * @class
  3  * @constructor
  4  */
  5 function vxlLookupTable(){
  6 	this.ID = null;
  7 	this.map = null;
  8 	this.max = Number.MIN_VALUE;
  9 	this.min = Number.MAX_VALUE;
 10 }
 11 
 12 vxlLookupTable.prototype.load = function(ID,payload){
 13 	this.ID = ID;
 14 	this.map = payload;
 15 	for (var key in this.map) {
 16 		var n = Number(key);
 17 		if (n < this.min) {this.min = n;}
 18 		else if (n >= this.max) {this.max = n;}
 19     }
 20 }
 21 
 22 vxlLookupTable.prototype.getColor = function(value){
 23 	var l = this;
 24 	var key = Math.round(value);
 25 	if (key >= l.min && key <= l.max){
 26 	    var c = [l.map[key][0],l.map[key][1],l.map[key][2],1.0];
 27 		return c;
 28 	}
 29 	else if (key <l.min) { //truncate to min value
 30 			return  [l.map[l.min][0],l.map[l.min][1],l.map[l.min][2],1.0];
 31 	}
 32 	else if (key>l.max){ //truncate to max value
 33 		return  [l.map[l.max][0],l.map[l.max][1],l.map[l.max][2],1.0];
 34 	}
 35 	else{
 36 		alert('assertion error in getColor routine');
 37 		return  [l.map[l.min][0],l.map[l.min][1],l.map[l.min][2],1.0];
 38 	}
 39 		
 40 }
 41 
 42 /**
 43 *
 44 *	@param s array with scalar data
 45 *	@param max range
 46 *	@param min range
 47 *	@returns unpacked colors translated through this lookup table 
 48 */
 49 vxlLookupTable.prototype.getColors = function(s,min,max){
 50 	var c = [];
 51 	
 52 	for(var i=0;i<s.length;i++){
 53 		var value = s[i] * this.max / max;
 54 		var cc = this.getColor(value);
 55 		c.push(cc[0]);
 56 		c.push(cc[1]);
 57 		c.push(cc[2]);
 58 		c.push(cc[3]);
 59 	}
 60 	
 61 	return c;
 62 }
