 17 /**
 18  * <p>Presents a simple interface to communicate with a ESSL (GLSL) program
 19  * This class is responsible for creating, compiling and linking any ESSL program.
 20  * It also has methods to query and set uniforms and attributes belonging to the program
 21  * that is being currently executed in the GPU</P>
 22  * 
 23  * <p>a vxlProgram maintains a database of the programs that have been linked to the GPU. 
 24  * This way, program switching is easier as it is not necessary to go through the 
 25  * compilation and linking process every time</p>
 26  * @class
 27  * @constructor
 28  */
 29 function vxlProgram (gl) {
 30     this._gl                = gl;
 31     this._codebase          = {};
 32 
 33     this._program           = {};
 34     this._attributeList     = {};
 35     this._enabledAttributes = [];
 36     this._uniformList       = {};
 37     this._uniformType       = {};
 38 
 39     this._uniform_cache     = {};
 40     
 41     this._currentWebGLProgram     =  null;
 42     this._currentProgramID        = "";
 43     this._currentUniformLocation  = {};
 44     this._defaults         = [];
 45 };
 46

 70 /**
 71  * Loads a program
 72  * @param {String} ID the id of the program to load
 73  */
 74 vxlProgram.prototype.loadProgram = function(ID){
 75     
 76     var code = this._codebase[ID];
 77     
 78     var gl   = this._gl;
 79     var WEB_GL_PROGRAM  = gl.createProgram();
 80     
 81     
 82     if (code.VERTEX_SHADER){
 83         var vs = this._createShader(vxl.def.glsl.VERTEX_SHADER,code.VERTEX_SHADER);
 84         gl.attachShader(WEB_GL_PROGRAM, vs);
 85     }
 86     
 87     if (code.FRAGMENT_SHADER){
 88         var fs = this._createShader(vxl.def.glsl.FRAGMENT_SHADER,code.FRAGMENT_SHADER);
 89         gl.attachShader(WEB_GL_PROGRAM, fs);
 90     }
 91     
 92     //fix for version 0.89.2 Making sure that the vertex array is ALWAYS the attribute with location 0
 93     gl.bindAttribLocation(WEB_GL_PROGRAM, 0 , vxl.def.glsl.VERTEX_ATTRIBUTE);
 94     
 95     gl.linkProgram(WEB_GL_PROGRAM);
 96      
 97     if (!gl.getProgramParameter(WEB_GL_PROGRAM, gl.LINK_STATUS)) {
 98         alert("Program: Could not link the shading program");
 99     }
100     else{
101         //vxl.go.console("Program: the program "+ID+" has been loaded");
102     }
103     
104     this._program[ID] = WEB_GL_PROGRAM;
105   
106 };

117 /**
118  * Uses a program from the database.
119  * If you are not sure if the program you want to use is in the database then call vxlRenderer.setProgram instead
120  * @param {String} ID the program id
121  * @see vxlRenderer#setProgram
122  */
123 vxlProgram.prototype.useProgram = function(ID){
124 
125     var gl = this._gl;
126     var WEB_GL_PROGRAM = this._program[ID];
127     
128     if (WEB_GL_PROGRAM != undefined && WEB_GL_PROGRAM != null){
129         
130         if (WEB_GL_PROGRAM == this._currentWebGLProgram) return;
131         
132         //gl.linkProgram(WEB_GL_PROGRAM);
133         gl.useProgram (WEB_GL_PROGRAM);
134         
135         this._currentWebGLProgram = WEB_GL_PROGRAM;
136         this._currentProgramID = ID;
137         this._parseUniforms();
138         
139         //vxl.go.console('Program: the program '+ID+' has been linked and is the current program');
140     }
141     else{
142         alert("Program: the program " + ID + " has NOT been loaded");
143     }
144 };

147 /**
148  * Loads the uniform defaults for the current program
149  */
150 vxlProgram.prototype.loadDefaults = function(){
151     var code = this._codebase[this._currentProgramID];
152    
153     if ('DEFAULTS' in code){
154     
155         var defaults = code.DEFAULTS;
156         for(var u in defaults){
157             this.setUniform(u,defaults[u]);
158         }
159     }
160     //overriding defaults
161     var defaults = this._defaults[this._currentProgramID];
162     if (defaults != undefined){
163   
164         for (var u in defaults){
165             this.setUniform(u,defaults[u])
166         }
167     }
168    
169 };
170 
171 /**
172  * Overrides defaults by hand 
173  */
174 vxlProgram.prototype.setDefault = function(programID, uniformName, value){
175     
176     if (this._defaults[programID] == undefined){
177         this._defaults[programID] = {}
178     }
179   
180     this._defaults[programID][uniformName] = value;
181     
182     //Overriding behaviour
183     if (programID == this._currentProgramID){
184         this.setUniform(uniformName, value);
185     }
186 };

189 /**
190  * Overrides defaults by hand 
191  */
192 vxlProgram.prototype.getDefault = function(programID, uniformName){
193     
194     if (this._defaults[programID] == undefined){
195         return undefined;
196     }
197   
198     return this._defaults[programID][uniformName];

201 /**

203  * @param {Object} an object containing uniform names and values. Every property of this object

212 /**
213  * Sets a uniform.
214  * Uses polymorphism to make the programmers life happier
215  * @param {String} uniform name
216  * @param {Object} the value
217  */
218 vxlProgram.prototype.setUniform = function(uniformID, value, hint){
219     
220     var webGLProgram 		= this._currentWebGLProgram;
221     var uniformList 		= this._uniformList[this._currentProgramID];
222     var uniformLoc  		= this._currentUniformLocation;
223     var uniform_cache 		= this._uniform_cache;
224     
225     if (uniformList.hasObject(uniformID)){
226         uniformLoc[uniformID] = this._gl.getUniformLocation(webGLProgram,uniformID);
227         
228     }
229     else{
230     	throw('Program: the uniform '+uniformID+' is not defined for the program '+this._currentProgramID);
231         return;
232     }
233     
234     uniform_cache[uniformID] = value;
235     this._setPolymorphicUniform(uniformID, uniformLoc[uniformID], value, hint);
236 };

262 /**
263  * Enables a vertex attribute array
264  * @param {String} name the name of the attribute array to enable
265  */
266 vxlProgram.prototype.enableAttribute = function(name){
267     
268    if (this._enabledAttributes.indexOf(name) != -1) return; //Speeds up
269      
270    var a = this._getAttributeLocation(name);
271    this._gl.enableVertexAttribArray(a);
272    this._enabledAttributes.push(name);
273 };
274 
275 /**
276  * Disables a vertex attribute array
277  * @param {String} name the name of the attribute array to disable
278  * 
279  */
280 vxlProgram.prototype.disableAttribute = function(name){
281     
282     var idx = this._enabledAttributes.indexOf(name); 
283     if (idx != -1) { //so it is enabled
284         var a = this._getAttributeLocation(name);
285         this._gl.disableVertexAttribArray(a);
286         this._enabledAttributes.splice(idx,1);
287     }
288 };

290 /**
291  * Creates a WebGL shader
292  * 
293  * @private This method is private.
294  */
295 vxlProgram.prototype._createShader = function(type,code){
296     var gl      = this._gl;
297     var shader = null;
298     
299     if (type == vxl.def.glsl.VERTEX_SHADER){
300         shader = gl.createShader(gl.VERTEX_SHADER);
301     }
302     else if (type == vxl.def.glsl.FRAGMENT_SHADER){
303         shader = gl.createShader(gl.FRAGMENT_SHADER);
304     }
305     
306     if (code == undefined || code == null){
307         alert('Error getting the code for shader of type ' + type);
308     }
309     
310     gl.shaderSource(shader, code);
311     gl.compileShader(shader);
312 
313     if (!gl.getShaderParameter(shader, gl.COMPILE_STATUS)) {
314         alert(gl.getShaderInfoLog(shader));
315     }
316     
317     return shader;
318 };

321 /**
322  * Parses uniforms
323  * This method is private
324  * @private
325  * 
326  */
327 vxlProgram.prototype._parseUniforms = function(id){
328     
329     vs = this._codebase[this._currentProgramID].VERTEX_SHADER;
330     fs = this._codebase[this._currentProgramID].FRAGMENT_SHADER;
331     /*@TODO: look for a way to retrieve uNames directly from the parsing of the shaders
332     this should simplify the structure of the JSON file representing the program*/
333     uNames = this._codebase[this._currentProgramID].UNIFORMS;
334     
335     uTypes = {};
336     
337     
338     for (var i=0;i< uNames.length; i++){
339         var uniformID = uNames[i];
340         var rex = new RegExp('uniform\\s+\\w+\\s'+uniformID,'g');
341         
342         if (vs.search(rex) != -1){
343             uTypes[uniformID] = vs.substring(vs.search(rex),vs.length).substring(0,vs.indexOf(';')).split(' ')[1];
344         }
345         
346         else if(fs.search(rex) != 1){
347             uTypes[uniformID] = fs.substring(fs.search(rex),fs.length).substring(0,fs.indexOf(';')).split(' ')[1];
348         }
349         
350         else{
351             alert('Program: In the program '+this._currentProgramID+' the uniform '+uniformID+' is listed but not used');
352         }
353     }
354     
355     
356     this._uniformList[this._currentProgramID] = uNames;
357     this._uniformType[this._currentProgramID] = uTypes; 
358 };

360 /**
361  * Obtains an attribute location
362  * This method is private
363  * @param {String} name
364  * @private
365  */
366 vxlProgram.prototype._getAttributeLocation = function(name){
367 
368     if(!(name in this._attributeList)){
369         this._attributeList[name] = this._gl.getAttribLocation(this._currentWebGLProgram,name);
370     }
371 
372     return this._attributeList[name];
373 };

375 /**
376  * This is one of the jewels of Voxelent. Based on the information contained in the 
377  * program database, it will do the appropriate gl call to set the uniform
378  * This method is private. Use setUniform instead.
379  * @see vxlProgram#setUniform
380  * @private 
381  */
382 vxlProgram.prototype._setPolymorphicUniform = function(uniformID, locationID,value,hint){
383 
384 	//In the extend of what it is reasonable,
385 	//We cross check GLSL type information with actual javascript variable types 
386 	//to make the right calls
387 	//hint allows better casting of int and float values. If not specified default is float
388     
389     var gl = this._gl;
390     var glslType = this._uniformType[this._currentProgramID][uniformID];
391     
392     if (glslType == 'bool'){
393     	//if (typeof value != 'boolean') { 
394     	//	vxl.go.console('Program: the uniform '+uniformID+' is defined as bool in GLSL. However the JS variable is not');
395     	//}/
396         gl.uniform1i(locationID,value);
397         return;
398     }
399     
400     else if (glslType == 'float'){
401     	gl.uniform1f(locationID,value);
402     	return;
403     }
404     
405     else if (glslType == 'int' || glslType == 'sampler2D'){
406         gl.uniform1i(locationID,value);
407         return;
408     }
409     
410     else if (glslType == 'mat4'){    
411         gl.uniformMatrix4fv(locationID,false,value);
412         return;
413     }
414     
415     
416     else if (value instanceof Array){
417         if (hint  == 'int'){
418             switch(value.length){
419                 case 1: { gl.uniform1iv(locationID,value); break; };
420                 case 2: { gl.uniform2iv(locationID,value); break; };
421                 case 3: { gl.uniform3iv(locationID,value); break; };
422                 case 4: { gl.uniform4iv(locationID,value); break; };
423                 default: alert('ERROR');
424             }
425        }
426        else{
427             switch(value.length){
428                 case 1 : { gl.uniform1fv(locationID,value); break; }
429                 case 2 : { gl.uniform2fv(locationID,value); break; }
430                 case 3 : { gl.uniform3fv(locationID,value); break; }
431                 case 4 : { gl.uniform4fv(locationID,value); break; }
432                 default: alert('ERROR');
433             }
434        }
435     }
436     
437     else {
438     	alert('Program: ERROR. The uniform  '+uniformID+ ' could not be mapped');
439     }
440 };
