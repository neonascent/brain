172  /**
173   * Loads 3D models, textures and other models to a Voxelent's scene.
174   * 
175   * This method is very flexible. It can load one or multiple models depending on the type of the 
176   * first parameter. If it is a String, it will try to find the file with that name in Voxelent's data folder
177   * (voxdata by default). Otherwise, if  the parameter 'arguments' is an Array, load will iterate
178   * through it and will try to load every element of this list. Every element being the file name.
179   * 
180   * Nucleo supports three different loading modes which are defined in 
181   * vxl.def.model.loadingMode:
182   * 
183   * LIVE: As each asset is loaded it is added immediately into the scene by creating the corresponding actor
184   * 
185   * LATER: All the models are loaded first THEN the actors are created. 
186   * This is useful when you want to display a full scene instead of showing incremental updates.
187   * 
188   * DETACHED: The models are loaded into the vxlModelManager object but actors are never created.
189   * This allows background loading.
190   * 
191   * @param {String|Array} arguments the name of the asset or the list of models (each element being the file name).
192   * @param {String} path the path that will be concatenated to the list of files (optional).
193   * @param {vxl.def.model.loadingMode} mode the loading mode
194   * @param {vxlScene} scene the scene in case we do not want to load these models in the current one
195   * 
196   * @see {vxl#def#asset#loadingMode}
197   * @see {vxlAssetManager}
198   * @see {vxlScene#setLoadingMode}
199   * 
200   */
201  load :  function(arguments,path,mode,scene){
202  	
203  	var scene = scene==null?vxl.c.scene:scene;
204  	var models = [];
205  	
206  	var p = vxl.util.getPath(path);
207  	
208  	if (typeof(arguments) == 'string' || arguments instanceof String){
209  		models.push(p  + arguments);
210  	}
211  	else if (arguments instanceof Array){
212  		for(var i=0; i<arguments.length;i++){
213 			models.push(p + arguments[i]);
214 		}
215  	}
216  	if (mode != undefined && mode != null){
217 		scene.setLoadingMode(mode);
218 	}
219 	
220 	vxl.go.modelManager.loadList(models, scene);
221  },
222  
223  /**
224   * Loads a sequence
225   * @param {String} prefix the shared name by all the files. For instance part in part1, part2, ...
226   * @param {Number} start start number in the sequence
227   * @param {Number} end end number in the sequence
228   * @param {String} path relative path to the webpage where the files are located
229   * @param {String} mode a loading mode (See load method)
230   * @param {String} scene the scene where the generated actors should be placed
231   */
232  loadSequence :  function(prefix, start, end, path,mode,scene){
233     
234     var scene = scene==null?vxl.c.scene:scene;
235     var models = [];
236     
237     var p = vxl.util.getPath(path);
238     
239     for(var i=start;i<=end; i+=1){
240         models.push(p + prefix+i+'.json');
241     }
242     if (mode != undefined && mode != null){
243         scene.setLoadingMode(mode);
244     }
245     
246     vxl.go.modelManager.loadList(models, scene);
247  },
