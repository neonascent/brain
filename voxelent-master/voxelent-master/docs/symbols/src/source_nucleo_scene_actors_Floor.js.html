 18 vxl.def.model.floor = new vxlModel();
 19 vxl.def.model.floor.load('floor',{"vertices":[],"indices":[],"diffuse":[1.0,1.0,1.0,1.0]});
 20 
 21 vxlFloor.prototype = new vxlActor();
 22 vxlFloor.prototype.constructor = vxlFloor;
 23 
 24 /**
 25  * @class
 26  * @constructor
 27  * @extends vxlActor
 28  */
 29 function vxlFloor(){
 30 	vxlActor.call(this, vxl.def.model.floor);
 31 	this.mode 		= vxl.def.actor.mode.WIREFRAME;
 32     this.visible 	= false;
 33     this.toy    	= true;
 34 };
 35 
 36 /**
 37  * Creates the grid
 38  */
 39 vxlFloor.prototype.setGrid =function(dimension, spacing){
 40 
 41 	var dim = dimension;
 42     var lines = 2*dim/spacing;
 43     var inc = 2*dim/lines;
 44     var v = [];
 45     var i = [];
 46 
 47     for(var l=0;l<=lines;l++){
 48         v[6*l] = -dim; 
 49         v[6*l+1] = 0;
 50         v[6*l+2] = -dim+(l*inc);
 51         
 52         v[6*l+3] = dim;
 53         v[6*l+4] = 0;
 54         v[6*l+5] = -dim+(l*inc);
 55         
 56         v[6*(lines+1)+6*l] = -dim+(l*inc); 
 57         v[6*(lines+1)+6*l+1] = 0;
 58         v[6*(lines+1)+6*l+2] = -dim;
 59         
 60         v[6*(lines+1)+6*l+3] = -dim+(l*inc);
 61         v[6*(lines+1)+6*l+4] = 0;
 62         v[6*(lines+1)+6*l+5] = dim;
 63         
 64         i[2*l] = 2*l;
 65         i[2*l+1] = 2*l+1;
 66         i[2*(lines+1)+2*l] = 2*(lines+1)+2*l;
 67         i[2*(lines+1)+2*l+1] = 2*(lines+1)+2*l+1;        
 68     }
 69     this.model.vertices = v.slice(0);
 70     this.model.wireframe = i.slice(0);
 71     this.visible = true;
 72 };
