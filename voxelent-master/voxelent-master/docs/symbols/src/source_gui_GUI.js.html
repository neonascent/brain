 27 function vxlGUI(view){
 28     var e = vxl.events;
 29     vxl.go.notifier.subscribe([
 30         e.MODEL_NEW,
 31         e.MODELS_LOADING,
 32         e.MODELS_LOADED,
 33     ], this);
 34     
 35     this.view = view;
 36     this.canvas           = $('canvas#'+this.view.name); 
 37 
 38     this.jqOverlay        = undefined;
 39     this.jqProgressBar    = undefined;
 40     this.jqMessage        = undefined;
 41 
 42     this.count            = 0;
 43     this.loaded           = 0;
 44     this.createGUI();
 45 };
 46 
 47 /**
 48  * 
 49  */
 50 vxlGUI.prototype.createGUI = function(){
 51     
 52     var pbdiv = document.createElement('div');
 53     pbdiv.id = this.view.name+'-progressbar-container';
 54     pbdiv.style.position  = 'absolute';
 55     pbdiv.style.width     = this.canvas.width();
 56     pbdiv.style.height    = this.canvas.height();
 57     pbdiv.style.top       = '4px';
 58     pbdiv.style.fontFamily  = 'Arial, Helvetica';
 59     pbdiv.style.fontSize    = '13px';
 60     pbdiv.style.display     = 'table-cell';
 61     pbdiv.style.verticalAlign  = 'middle';
 62     pbdiv.style.textAlign      = 'center';
 63     pbdiv.style.backgroundImage = "url('http://voxelent.com/html/demos/css/img/vox-gui-logo.png')";
 64     pbdiv.style.backgroundRepeat = 'no-repeat';
 65     pbdiv.style.backgroundPosition = 'center '+Math.round(this.canvas.height()/3);
 66     pbdiv.style.borderColor  = '#ccc'
 67     pbdiv.style.borderStyle = 'solid';
 68     pbdiv.style.borderWidth = '1px';
 69     
 70     this.jqOverlay = $(pbdiv);
 71     
 72     
 73     
 74     var pb = document.createElement('div');
 75     pb.div = this.view.name+'-progressbar';
 76     pb.style.position  = 'relative';
 77     pb.style.display   = 'inline-block';
 78     pb.style.textAlign = 'center';
 79     pb.style.width     = '80%';
 80     pb.style.height   =   '25px';
 81     pb.style.top      = this.canvas.height()/2;
 82     this.jqProgressBar = $(pb);
 83     
 84     var msg = document.createElement('div');
 85     msg.style.position   = 'relative';
 86     msg.style.textAlign   = 'center';
 87     msg.style.top         = this.canvas.height()/2 + 20;
 88     this.jqMessage = $(msg);
 89     
 90     this.jqOverlay.append(this.jqProgressBar); 
 91     this.jqOverlay.append(this.jqMessage);
 92      
 93 };
 94 
 95 
 96 vxlGUI.prototype.showOverlay = function(){
 97     this.canvas.hide();
 98     this.canvas.parent().append(this.jqOverlay);
 99 };
100 
101 vxlGUI.prototype.initProgressBar = function(mmanager){
102     this.count = mmanager.toLoad.length;
103     this.jqProgressBar.progressbar({value:0});
104     this.jqMessage.html('<p>Loading...</p>');
105 };
106 
107 
108 vxlGUI.prototype.updateProgressBar = function(mmanager){
109     this.loaded = this.count - mmanager.toLoad.length;
110     var percentage = Math.round((1- (mmanager.toLoad.length / this.count)) * 100);
111     this.jqMessage.html('<p>Loading: ' + this.loaded + ' out of '+ this.count+' ('+percentage+'%)</p>');
112     this.jqProgressBar.progressbar({value:percentage});
113 };    
114 
115 
116 vxlGUI.prototype.hideOverlay = function(){
117     this.jqProgressBar.progressbar({value:100});
118     this.jqOverlay.fadeOut(300).hide();
119     this.canvas.fadeIn(300);
120 };
121 
122 vxlGUI.prototype.handleEvent = function(event, source){
123     var e = vxl.events;
124     switch(event){
125         case e.MODELS_LOADING: this.showOverlay(); 
126                                this.initProgressBar(source); 
127                                break;
128         case e.MODEL_NEW:      this.updateProgressBar(source); 
129                                break;
130         case e.MODELS_LOADED:  this.hideOverlay(source);  
131                                break;
132     }
133 };
