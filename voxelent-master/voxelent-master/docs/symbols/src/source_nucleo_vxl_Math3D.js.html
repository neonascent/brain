  7 vxl.vec3.dot = function(a,b){
  8 	if (!(a instanceof vxlVector3) || !(b instanceof vxlVector3)) {
  9 		alert('vxl.math.vector.dot ERROR');
 10 		return null;
 11 	}
 12 	return ((a.x*b.x) + (a.y*b.y) + (a.z*b.z));

 15 vxl.vec3.cross = function(a,b){
 16 	if (!(a instanceof vxlVector3) || !(b instanceof vxlVector3)) {
 17 		alert('vxl.math.vector.cross ERROR');
 18 		return null;
 19 	}
 20 	
 21 	var Zx = a.y * b.z - a.z * b.y; 
 22 	var Zy = a.z * b.x - a.x * b.z;
 23 	var Zz = a.x * b.y - a.y * b.x;
 24 	return new vxlVector3(Zx,Zy,Zz);

 55 vxl.vec3.round = function(v,n){
 56 	var v = this;
 57 	if (n == null || n == undefined || n <0) {
 58 		v.x = Math.round(v.x);
 59 		v.y = Math.round(v.y);
 60 		v.z = Math.round(v.z);

 62 	v.x = Math.round(v.x*10*n)/(10*n);

 67 vxl.vec3.normalize = function(v,n){
 68 	var le = vxl.vec3.length(v);
 69 	
 70 	if (le ==0) throw('ERROR: normalizing a vector by a zero norm');
 71 	
 72 	if (n == undefined){
 73 	   v.x = v.x / le;
 74 	   v.y = v.y / le;
 75 	   v.z = v.z / le;
 76 	}
 77 	else{
 78 	    n.x = v.x / le;
 79 	    n.y = v.y / le;
 80 	    n.z = v.z / le;
 81 	}

110 	
111 	var v = this;
112 	var name = 'vector ';
113 	
114 	if (isNaN(v.x) || isNaN(v.y) || isNaN(v.z)){
115 		return 'vector ' +n+' does not have numeric values';
116 	}
117 	
118 	if (n != null){
119 		name = n;
120 	}
121 	
122 	if (p == null){
123 		return name +': (' + v.x +','+v.y+','+v.z+')';
124 	}
125 	else if (p>0){
126 		return name+':(' + v.x.toFixed(p) +','+v.y.toFixed(p)+','+v.z.toFixed(p)+')';
127 	}

289 vxl.mat4.inverse = function(m,dest){
290 
291 	  // Calculate the 4x4 determinant
292     // If the determinant is zero, 
293     // then the inverse matrix is not unique.
294     var det = vxl.mat4._determinant4x4(m);
295 
296     if (Math.abs(det) < 1e-8){
297         return null;
298     }
299     vxl.mat4._makeAdjoint(m);
300 
301     // Scale the adjoint matrix to get the inverse
302     m.m11 /= det;
303     m.m12 /= det;
304     m.m13 /= det;
305     m.m14 /= det;
306     
307     m.m21 /= det;
308     m.m22 /= det;
309     m.m23 /= det;
310     m.m24 /= det;
311     
312     m.m31 /= det;
313     m.m32 /= det;
314     m.m33 /= det;
315     m.m34 /= det;
316     
317     m.m41 /= det;
318     m.m42 /= det;
319     m.m43 /= det;
320     m.m44 /= det;
321 };

396 vxl.mat4.rotate = function(m,angle,v){
397 
398     // angles are in degrees. Switch to radians
399     angle = angle / 180 * Math.PI;
400     
401     angle /= 2;
402     var sinA = Math.sin(angle);
403     var cosA = Math.cos(angle);
404     var sinA2 = sinA * sinA;
405     
406     // normalize
407     var length = Math.sqrt(v.x * v.x + v.y * v.y + v.z * v.z);
408     if (length == 0) {
409         // bad vector, just use something reasonable
410         v.x = 0;
411         v.y = 0;
412         v.z = 1;
413     } else if (length != 1) {
414         v.x /= length;
415         v.y /= length;
416         v.z /= length;
417     }
418     
419     var mat = new vxlMatrix4x4();
420 
421     // optimize case where axis is along major axis
422     if (v.x == 1 && v.y == 0 && v.z == 0) {
423         mat.m11 = 1;
424         mat.m12 = 0;
425         mat.m13 = 0;
426         mat.m21 = 0;
427         mat.m22 = 1 - 2 * sinA2;
428         mat.m23 = 2 * sinA * cosA;
429         mat.m31 = 0;
430         mat.m32 = -2 * sinA * cosA;
431         mat.m33 = 1 - 2 * sinA2;
432         mat.m14 = mat.m24 = mat.m34 = 0;
433         mat.m41 = mat.m42 = mat.m43 = 0;
434         mat.m44 = 1;
435     } else if (v.x == 0 && v.y == 1 && v.z == 0) {
436         mat.m11 = 1 - 2 * sinA2;
437         mat.m12 = 0;
438         mat.m13 = -2 * sinA * cosA;
439         mat.m21 = 0;
440         mat.m22 = 1;
441         mat.m23 = 0;
442         mat.m31 = 2 * sinA * cosA;
443         mat.m32 = 0;
444         mat.m33 = 1 - 2 * sinA2;
445         mat.m14 = mat.m24 = mat.m34 = 0;
446         mat.m41 = mat.m42 = mat.m43 = 0;
447         mat.m44 = 1;
448     } else if (v.x == 0 && v.y == 0 && v.z == 1) {
449         mat.m11 = 1 - 2 * sinA2;
450         mat.m12 = 2 * sinA * cosA;
451         mat.m13 = 0;
452         mat.m21 = -2 * sinA * cosA;
453         mat.m22 = 1 - 2 * sinA2;
454         mat.m23 = 0;
455         mat.m31 = 0;
456         mat.m32 = 0;
457         mat.m33 = 1;
458         mat.m14 = mat.m24 = mat.m34 = 0;
459         mat.m41 = mat.m42 = mat.m43 = 0;
460         mat.m44 = 1;
461     } else {
462         var x2 = v.x*v.x;
463         var y2 = v.y*v.y;
464         var z2 = v.z*v.z;
465     
466         mat.m11 = 1 - 2 * (y2 + z2) * sinA2;
467         mat.m12 = 2 * (v.x * v.y * sinA2 + v.z * sinA * cosA);
468         mat.m13 = 2 * (v.x * v.z * sinA2 - v.y * sinA * cosA);
469         mat.m21 = 2 * (v.y * v.x * sinA2 - v.z * sinA * cosA);
470         mat.m22 = 1 - 2 * (z2 + x2) * sinA2;
471         mat.m23 = 2 * (v.y * v.z * sinA2 + v.x * sinA * cosA);
472         mat.m31 = 2 * (v.z * v.x * sinA2 + v.y * sinA * cosA);
473         mat.m32 = 2 * (v.z * v.y * sinA2 - v.x * sinA * cosA);
474         mat.m33 = 1 - 2 * (x2 + y2) * sinA2;
475         mat.m14 = mat.m24 = mat.m34 = 0;
476         mat.m41 = mat.m42 = mat.m43 = 0;
477         mat.m44 = 1;
478     }
479 	vxl.mat4.multRight(m,mat);
480 };

483 vxl.mat4.rotateX = function(mat, angle, dest) {
484         var s = Math.sin(angle);
485         var c = Math.cos(angle);
486         
487         // Cache the matrix values (makes for huge speed increases!)
488         var a10 = mat.m21, a11 = mat.m22, a12 = mat.m23, a13 = mat.m24;
489         var a20 = mat.m31, a21 = mat.m32, a22 = mat.m33, a23 = mat.m34;
490 
491         if(!dest) { 
492                 dest = mat;
493         } else if(mat != dest) { // If the source and destination differ, copy the unchanged rows
494                 dest.m11 = mat.m11;
495                 dest.m12 = mat.m12;
496                 dest.m13 = mat.m13;
497                 dest.m14 = mat.m14;
498                 
499                 dest.m41 = mat.m41;
500                 dest.m42 = mat.m42;
501                 dest.m43 = mat.m43;
502                 dest.m44 = mat.m44;
503         }
504         
505         // Perform axis-specific matrix multiplication
506         dest.m21 = a10*c + a20*s;
507         dest.m22 = a11*c + a21*s;
508         dest.m23 = a12*c + a22*s;
509         dest.m24 = a13*c + a23*s;
510         
511         dest.m31 = a10*-s + a20*c;
512         dest.m32 = a11*-s + a21*c;
513         dest.m33 = a12*-s + a22*c;
514         dest.m34 = a13*-s + a23*c;
515         return dest;
516 };

527                 dest = mat;
528         }

536                 dest.m42 = mat.m42;
537                 dest.m43 = mat.m43;
538                 dest.m44 = mat.m44;

540         
541         // Perform axis-specific matrix multiplication
542         dest.m11 = a00*c + a20*-s;

544         dest.m13 = a02*c + a22*-s;

548         dest.m32 = a01*s + a21*c;
549         dest.m33 = a02*s + a22*c;

613 vxl.mat4.multLeft = function(m,mat){

619                + mat.m13 * m.m33 + mat.m14 * m.m43);
620     var m14 = (mat.m11 * m.m14 + mat.m12 * m.m24
621                + mat.m13 * m.m34 + mat.m14 * m.m44);
622 
623     var m21 = (mat.m21 * m.m11 + mat.m22 * m.m21
624                + mat.m23 * m.m31 + mat.m24 * m.m41);
625     var m22 = (mat.m21 * m.m12 + mat.m22 * m.m22
626                + mat.m23 * m.m32 + mat.m24 * m.m42);

641     var m41 = (mat.m41 * m.m11 + mat.m42 * m.m21
642                + mat.m43 * m.m31 + mat.m44 * m.m41);
643     var m42 = (mat.m41 * m.m12 + mat.m42 * m.m22

645     var m43 = (mat.m41 * m.m13 + mat.m42 * m.m23
646                + mat.m43 * m.m33 + mat.m44 * m.m43);
647     var m44 = (mat.m41 * m.m14 + mat.m42 * m.m24
648                + mat.m43 * m.m34 + mat.m44 * m.m44);
649     
650     m.m11 = m11;
651     m.m12 = m12;
652     m.m13 = m13;

671 vxl.mat4.ortho = function(m,left, right, bottom, top, near, far){
672     var tx = (left + right) / (left - right);
673     var ty = (top + bottom) / (top - bottom);
674     var tz = (far + near) / (far - near);
675     
676     var matrix = new vxlMatrix4x4();
677     matrix.m11 = 2 / (left - right);
678     matrix.m12 = 0;
679     matrix.m13 = 0;
680     matrix.m14 = 0;
681     matrix.m21 = 0;
682     matrix.m22 = 2 / (top - bottom);
683     matrix.m23 = 0;
684     matrix.m24 = 0;
685     matrix.m31 = 0;
686     matrix.m32 = 0;
687     matrix.m33 = -2 / (far - near);
688     matrix.m34 = 0;
689     matrix.m41 = tx;
690     matrix.m42 = ty;
691     matrix.m43 = tz;
692     matrix.m44 = 1;
693     
694     vxl.mat4.multRight(m,matrix);
695 };

697 vxl.mat4.frustum = function(m,left, right, bottom, top, near, far){
698     var matrix = new vxlMatrix4x4();
699     var A = (right + left) / (right - left);
700     var B = (top + bottom) / (top - bottom);
701     var C = -(far + near) / (far - near);
702     var D = -(2 * far * near) / (far - near);
703     
704     matrix.m11 = (2 * near) / (right - left);
705     matrix.m12 = 0;
706     matrix.m13 = 0;
707     matrix.m14 = 0;
708     
709     matrix.m21 = 0;
710     matrix.m22 = 2 * near / (top - bottom);
711     matrix.m23 = 0;
712     matrix.m24 = 0;
713     
714     matrix.m31 = A;
715     matrix.m32 = B;
716     matrix.m33 = C;
717     matrix.m34 = -1;
718     
719     matrix.m41 = 0;
720     matrix.m42 = 0;
721     matrix.m43 = D;
722     matrix.m44 = 0;
723     
724     vxl.mat4.multRight(m,matrix);
725 };

728     var top = Math.tan(fovy * Math.PI / 360) * zNear;
729     var bottom = -top;
730     var left = aspect * bottom;
731     var right = aspect * top;
732     vxl.mat4.frustum(m,left, right, bottom, top, zNear, zFar);

735 vxl.mat4.lookat = function(m,eyex, eyey, eyez, centerx, centery, centerz, upx, upy, upz){
736     var matrix = new vxlMatrix4x4();
737     
738     // Make rotation matrix
739 
740     // Z vector
741     var zx = eyex - centerx;
742     var zy = eyey - centery;
743     var zz = eyez - centerz;
744     var mag = Math.sqrt(zx * zx + zy * zy + zz * zz);
745     if (mag) {
746         zx /= mag;
747         zy /= mag;
748         zz /= mag;
749     }
750 
751     // Y vector
752     var yx = upx;
753     var yy = upy;
754     var yz = upz;
755 
756     // X vector = Y cross Z
757     xx =  yy * zz - yz * zy;
758     xy = -yx * zz + yz * zx;
759     xz =  yx * zy - yy * zx;
760 
761     // Recompute Y = Z cross X
762     yx = zy * xz - zz * xy;
763     yy = -zx * xz + zz * xx;
764     yx = zx * xy - zy * xx;
765 
766     // cross product gives area of parallelogram, which is < 1.0 for
767     // non-perpendicular unit-length vectors; so normalize x, y here
768 
769     mag = Math.sqrt(xx * xx + xy * xy + xz * xz);
770     if (mag) {
771         xx /= mag;
772         xy /= mag;
773         xz /= mag;
774     }
775 
776     mag = Math.sqrt(yx * yx + yy * yy + yz * yz);
777     if (mag) {
778         yx /= mag;
779         yy /= mag;
780         yz /= mag;
781     }
782 
783     matrix.m11 = xx;
784     matrix.m12 = xy;
785     matrix.m13 = xz;
786     matrix.m14 = 0;
787     
788     matrix.m21 = yx;
789     matrix.m22 = yy;
790     matrix.m23 = yz;
791     matrix.m24 = 0;
792     
793     matrix.m31 = zx;
794     matrix.m32 = zy;
795     matrix.m33 = zz;
796     matrix.m34 = 0;
797     
798     matrix.m41 = 0;
799     matrix.m42 = 0;
800     matrix.m43 = 0;
801     matrix.m44 = 1;
802     matrix.translate(-eyex, -eyey, -eyez);
803     
804     vxl.mat4.multRight(m,matrix);
805 };

818 vxl.mat4._determinant4x4 = function(m){
819     var a1 = m.m11;
820     var b1 = m.m12; 
821     var c1 = m.m13;
822     var d1 = m.m14;
823 
824     var a2 = m.m21;
825     var b2 = m.m22;

828

835     var b4 = m.m42;

840          - b1 * vxl.mat4._determinant3x3(a2, a3, a4, c2, c3, c4, d2, d3, d4)

845 vxl.mat4._makeAdjoint = function(m){
846     var a1 = m.m11;
847     var b1 = m.m12; 
848     var c1 = m.m13;
849     var d1 = m.m14;
850 
851     var a2 = m.m21;
852     var b2 = m.m22; 
853     var c2 = m.m23;
854     var d2 = m.m24;
855 
856     var a3 = m.m31;
857     var b3 = m.m32; 
858     var c3 = m.m33;
859     var d3 = m.m34;
860 
861     var a4 = m.m41;
862     var b4 = m.m42; 
863     var c4 = m.m43;
864     var d4 = m.m44;
865 
866     // Row column labeling reversed since we transpose rows & columns
867     m.m11  =   vxl.mat4._determinant3x3(b2, b3, b4, c2, c3, c4, d2, d3, d4);
868     m.m21  = - vxl.mat4._determinant3x3(a2, a3, a4, c2, c3, c4, d2, d3, d4);
869     m.m31  =   vxl.mat4._determinant3x3(a2, a3, a4, b2, b3, b4, d2, d3, d4);
870     m.m41  = - vxl.mat4._determinant3x3(a2, a3, a4, b2, b3, b4, c2, c3, c4);
871         
872     m.m12  = - vxl.mat4._determinant3x3(b1, b3, b4, c1, c3, c4, d1, d3, d4);
873     m.m22  =   vxl.mat4._determinant3x3(a1, a3, a4, c1, c3, c4, d1, d3, d4);
874     m.m32  = - vxl.mat4._determinant3x3(a1, a3, a4, b1, b3, b4, d1, d3, d4);
875     m.m42  =   vxl.mat4._determinant3x3(a1, a3, a4, b1, b3, b4, c1, c3, c4);
876         
877     m.m13  =   vxl.mat4._determinant3x3(b1, b2, b4, c1, c2, c4, d1, d2, d4);
878     m.m23  = - vxl.mat4._determinant3x3(a1, a2, a4, c1, c2, c4, d1, d2, d4);
879     m.m33  =   vxl.mat4._determinant3x3(a1, a2, a4, b1, b2, b4, d1, d2, d4);
880     m.m43  = - vxl.mat4._determinant3x3(a1, a2, a4, b1, b2, b4, c1, c2, c4);
881         
882     m.m14  = - vxl.mat4._determinant3x3(b1, b2, b3, c1, c2, c3, d1, d2, d3);
883     m.m24  =   vxl.mat4._determinant3x3(a1, a2, a3, c1, c2, c3, d1, d2, d3);
884     m.m34  = - vxl.mat4._determinant3x3(a1, a2, a3, b1, b2, b3, d1, d2, d3);
