526     /**
527      *Returns a RGB color based on an integer (0..16 millions?)

529     int2rgb: function(i){
530         return [((i >> 16) & 0xFF)/256,((i >> 8) & 0xFF)/256,(i & 0xFF)/256];

532     /**
533      * Formats Arrays, vec3 and vec4 for display
534      * 
535      * @param {Array, vec3, vec4} arr the object to format
536      * @param {Number} digits the number of decimal figures
537      */
538 	format: function(arr, digits){
539 		var p = Math.pow(10,digits);
540 		if (typeof(arr) == 'object'){
541 			
542 			var result = '['; 
543 			for (var i=0; i < arr.length-1; i+=1){
544 				result  += Math.round(arr[i] * p) / p + ', '; 
545 			}
546 			result += Math.round(arr[arr.length-1] * p) / p  + ']'
547 		}
548 		else if (typeof(arr) == 'number'){
549 			result = '[' + Math.round(arr * p) / p  + ']';
550 		}
551 		return result;
552 	},
553 	
554 	/**
555 	 * Creates a vector from a set of parameters
556 	 * @param {Array, vec3, Number} x it can be an Array, a vec3 or a number
557 	 * @param {Number} y if x is a number, this parameter corresponds to the y-component
558 	 * @param {Number} z if x is a number, this parameter corresponds to the z-component
559 	 */
560 	createVec3: function(x,y,z){
561 	    var vvv = vec3.create();
562 	    if (x instanceof Array){
563             vec3.set(vec3.create(x), vvv)
564         }
565         else if (x instanceof determineMatrixArrayType()){
566             vec3.set(x, vvv)
567         }
568         else{
569             vec3.set(vec3.createFrom(x,y,z), vvv);
570         }
571         return vvv;
572 	},
573 	/**
