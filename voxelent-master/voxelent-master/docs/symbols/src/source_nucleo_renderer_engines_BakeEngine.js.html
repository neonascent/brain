 17 /**
 18  * @constructor
 19  * @class
 20  * Implements a basic rendering strategy that works with the following programs:
 21  * 
 22  * vxl.def.essl.bake
 23  * TODO: Does not deal with model scalars well...
 24  * 
 25  */
 26 function vxlBakeEngine(renderer){
 27     this.renderer = renderer;
 28     
 29     var gl = renderer.gl;
 30     
 31     this._calls     = {}
 32     
 33     this._gl_buffers  = {
 34         index       : gl.createBuffer(),
 35         baked       : gl.createBuffer(),
 36         position    : gl.createBuffer(),
 37         scale       : gl.createBuffer(),
 38         shading     : gl.createBuffer()
 39     };  
 40     
 41     this._data        = {
 42         index       : [],
 43         baked       : [],
 44         position    : [],
 45         scale       : [],
 46         shading     : []
 47         
 48     }; 
 49      
 50     this._allocated   = []; //if an actor is here, do not allocate
 51     
 52     this._offsets     = {
 53        position :{},
 54        scale    :{},
 55        shading  :{},
 56        baked    :{} 
 57     }; 
 58     
 59     
 60     this.glsl = vxl.util.extend(
 61         vxl.def.essl,{
 62         POSITION: "aPosition",
 63         SCALE:    "aScale",
 64         SHADING:  "aShading"
 65     });
 66 
 67 };

 70 /**
 71  * Creates an array of a defined size populated with the indicated value
 72  * @private
 73  */
 74 vxlBakeEngine.prototype._populate = function(value, size){
 75     var a = [];
 76     
 77     for (var i = 0; i < size; i+=1){
 78         if (value instanceof Array || value instanceof Float32Array){
 79             for (var j = 0; j < value.length; j+=1){
 80                 a.push(value[j]);
 81             }
 82         }
 83         else{
 84             a.push(value);
 85         }
 86     }
 87     return a;
 88 };

 91 /**
 92  * Computes the number of required calls to render one scene
 93  * @private
 94  */
 95 vxlBakeEngine.prototype._computeRequiredCalls = function(scene){
 96 
 97     var elements    = scene._actors; //.concat(scene.toys.list);
 98     var NUM         = elements.length;
 99     var accum = 0;
100     var count = 1;
101     
102     for(var i = 0; i < NUM; i+=1){
103         accum += elements[i].model.indices.length;
104         if (accum > 65000){ //66560?
105             accum = 0;
106             count ++;
107         }
108     }
109     return count;
110 };

113 /**
114  * Receives one actor and returns the GL buffers
115  * @private
116  */
117 vxlBakeEngine.prototype._allocateActor = function(actor){
118     
119     if (this._allocated.indexOf(actor.UID) != -1) return; //this actor has been allocated
120     var data            = this._data;
121     var offsets         = this._offsets;
122     var gl              = this.renderer.gl;
123     var model           = actor.model;
124     var NUM_VERTICES    = model.vertices.length;
125     
126         
127     var color = [], normal = [];
128     
129     //Taking care of colors
130     if (actor.material.diffuse) {
131         color = this._populate(actor.material.diffuse, NUM_VERTICES/3);
132     }
133     else{
134         color = this._populate([0.7,0.7,0.7], NUM_VERTICES/3);
135     }
136     
137     //Taking care of normals
138     if (model.normals){
139         normal = model.normals;
140     }
141     else{
142         normal = this._populate(0, NUM_VERTICES);
143     }
144     
145     offsets.baked[actor.UID]    =data.baked.length;
146     
147     for (var i=0;i<NUM_VERTICES; i+=3){
148         
149         data.baked.push(model.vertices[i]);
150         data.baked.push(model.vertices[i+1]);
151         data.baked.push(model.vertices[i+2]);
152         
153         data.baked.push(normal[i]);
154         data.baked.push(normal[i+1]);
155         data.baked.push(normal[i+2]);
156         
157         data.baked.push(color[i]);
158         data.baked.push(color[i+1]);
159         data.baked.push(color[i+2]);
160     }
161     
162     offsets.position[actor.UID] = data.position.length;
163     offsets.scale[actor.UID]    = data.scale.length;
164     offsets.shading[actor.UID]  = data.shading.length;
165     
166     data.position = data.position.concat(this._populate(actor._position, NUM_VERTICES/3));
167     data.scale    = data.scale.concat(this._populate(actor._scale, NUM_VERTICES/3));
168     data.shading = data.shading.concat(this._populate(actor.material.opacity, NUM_VERTICES/3));
169     
170 
171     var ind = model.indices.slice(0);
172     if (data.index.length > 0){
173         var max = data.index.max()+1;
174         var NUM_INDICES = model.indices.length;
175         for (var i=0; i<NUM_INDICES; i+=1){
176             ind[i] += max;
177         }
178         data.index = data.index.concat(ind);
179     }
180     else{
181         data.index = ind;
182     }
183     
184     this._allocated.push(actor.UID);
185 };

187 /**
188  * @private
189  */
190 vxlBakeEngine.prototype._updateActorPosition = function(actor){
191     
192     var data = this._data;
193     var offset = this._offsets.position[actor.UID];
194     
195     if (offset == undefined) return;
196     
197     var LEN = actor.model.vertices.length + offset;
198     for(var i =offset;i<LEN;i+=3){
199         data.position[i]   = actor._position[0];
200         data.position[i+1] = actor._position[1];
201         data.position[i+2] = actor._position[2];
202     }
203 };
204 
205 /**
206  * @private
207  */
208 vxlBakeEngine.prototype._updateActorScale = function(actor){
209     var data = this._data;
210     var offset = this._offsets.scale[actor.UID];
211     
212     if (offset == undefined) return;
213     
214     var LEN = actor.model.vertices.length + offset;
215     for(var i =offset;i<LEN;i+=3){
216         data.scale[i]   = actor._scale[0];
217         data.scale[i+1] = actor._scale[1];
218         data.scale[i+2] = actor._scale[2];
219     }
220 };

223 /**
224  * @private
225  */
226 vxlBakeEngine.prototype._updateActorColor = function(actor){
227     var data = this._data;
228     var offset = this._offsets.baked[actor.UID];
229     
230     if (offset == undefined) return;
231     
232     var LEN = (actor.model.vertices.length*3) + offset;
233     for(var i =offset;i<LEN;i+=9){
234         data.baked[i+6] = actor.material.diffuse[0];
235         data.baked[i+7] = actor.material.diffuse[1];
236         data.baked[i+8] = actor.material.diffuse[2];
237     }
238 };
239 
240 /**
241  * @private
242  */
243 vxlBakeEngine.prototype._updateActorShading = function(actor){
244     var data = this._data;
245     var offset = this._offsets.shading[actor.UID];
246     
247     if (offset == undefined) return;
248     
249     var LEN = (actor.model.vertices.length/3) + offset;
250     var val = 0.0;
251     
252     if (actor.material.opacity){
253         val = 1.0;
254     }
255     
256     for(var i =offset;i<LEN;i+=1){
257        data.shading[i]   = val;
258     }
259 };

270 /**
271  * Implements basic allocation of memory. Creates the WebGL buffers for the actor
272  * @param {vxlScene} scene the scene to allocate memory for
273  * @returns an object that contains the allocated WebGL buffers
274  */
275 vxlBakeEngine.prototype.allocate = function(scene){
276     
277     if (this._computeRequiredCalls(scene) > 1) {
278         throw 'Not renderable yet. Working on it. The number of indices exceeds the 65K limit';
279     }
280     
281     var elements    = scene._actors; //.concat(scene.toys.list);
282     var NUM         = elements.length;
283     var r           = this.renderer;
284     var buffers     = this._gl_buffers;
285     var data        = this._data;
286 
287     var pm         = r.pm;
288     var gl          = r.gl;
289     var essl        = this.glsl;
290     var mode        = gl.STATIC_DRAW;
291     
292     
293     
294     for(var i = 0; i < NUM; i+=1){
295         this._allocateActor(elements[i]);
296     }
297     
298     pm.enableAttribute(essl.VERTEX_ATTRIBUTE);
299     pm.enableAttribute(essl.NORMAL_ATTRIBUTE);
300     pm.enableAttribute(essl.COLOR_ATTRIBUTE);
301     
302     pm.enableAttribute(essl.POSITION);
303     pm.enableAttribute(essl.SCALE);
304     pm.enableAttribute(essl.SHADING);
305     
306     gl.bindBuffer(gl.ARRAY_BUFFER, buffers.baked);
307     gl.bufferData(gl.ARRAY_BUFFER, new Float32Array(data.baked), mode);
308     pm.setAttributePointer(essl.VERTEX_ATTRIBUTE, 3, gl.FLOAT, false, 36, 0);
309     pm.setAttributePointer(essl.NORMAL_ATTRIBUTE, 3, gl.FLOAT, false, 36, 12);
310     pm.setAttributePointer(essl.COLOR_ATTRIBUTE, 3, gl.FLOAT, false, 36, 24);
311     
312     gl.bindBuffer(gl.ARRAY_BUFFER, buffers.position);
313     gl.bufferData(gl.ARRAY_BUFFER, new Float32Array(data.position), mode);
314     pm.setAttributePointer(essl.POSITION, 3, gl.FLOAT, false,12,0);
315     
316     gl.bindBuffer(gl.ARRAY_BUFFER, buffers.scale);
317     gl.bufferData(gl.ARRAY_BUFFER, new Float32Array(data.scale), mode);
318     pm.setAttributePointer(essl.SCALE, 3, gl.FLOAT, false,12,0);
319     
320     gl.bindBuffer(gl.ARRAY_BUFFER, buffers.shading);
321     gl.bufferData(gl.ARRAY_BUFFER, new Float32Array(data.shading), mode);
322     pm.setAttributePointer(essl.SHADING, 1, gl.FLOAT, false,4,0);
323     
324 
325     gl.bindBuffer(gl.ELEMENT_ARRAY_BUFFER, buffers.index);
326     gl.bufferData(gl.ELEMENT_ARRAY_BUFFER, new Uint16Array(data.index), mode);
327 };

332 /**
333  * Renders the actors one by one
334  * @param {vxlScene} scene the scene to render
335  */
336 vxlBakeEngine.prototype.render = function(scene){
337 
338     //Updates the perspective matrix and passes it to the program
339     var r       = this.renderer;
340     var trx     = r.transforms
341     var pm     = r.pm;
342     var gl      = r.gl;
343     var essl    = vxl.def.essl;
344     var data    = this._data;
345     
346     trx.update();
347 
348     pm.setUniform(essl.PERSPECTIVE_MATRIX, trx.pMatrix);
349     pm.setUniform(essl.MODEL_VIEW_MATRIX,  trx.mvMatrix);
350     pm.setUniform(essl.NORMAL_MATRIX,      trx.nMatrix);
351     pm.setUniform(essl.MVP_MATRIX,         trx.mvpMatrix);
352     
353     for(var i = 0, N = scene._actors.length; i<N; i+=1){
354         var actor = scene._actors[i];
355         this._updateActorPosition(actor);   
356         this._updateActorScale(actor);     
357         this._updateActorShading(actor);  
358         this._updateActorColor(actor);     
359     }
360     
361     
362     gl.drawElements(gl.TRIANGLES, data.index.length, gl.UNSIGNED_SHORT,0);
363     gl.bindBuffer(gl.ARRAY_BUFFER, null);
364     gl.bindBuffer(gl.ELEMENT_ARRAY_BUFFER, null);
365 };
