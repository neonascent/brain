 18 /**
 19  * <p>
 20  * An actor is a representation of a model in Voxelent. Actors can cache model properties 
 21  * and modified them. This is useful when there are several actors based in the same model
 22  * but each one of them needs to have a different version of any given model property (i.e. color)
 23  * </p>
 24  * <p> 
 25  * To propagate one change for all the actors based in the same model, the setProperty method
 26  * should be invoked by setting the third parameter (scope) like this:
 27  * </p>
 28  * 
 29  * <pre class="prettyprint">
 30  * var actor = vxl.c.scene.getActorByName('example');
 31  * actor.setProperty('color',[1.0,0.0,0.0], vxl.def.model)
 32  * </pre>
 33  * 
 34  * <p>If the change should be local (for just that actor)  then you should write:</p>
 35  * 
 36  * <pre class="prettyprint">
 37  * var actor = vxl.c.scene.getActorByName('example');
 38  * actor.setProperty('color',[1.0,0.0,0.0], vxl.def.actor)
 39  * </pre>
 40  * 
 41  * <p> Or simply </p>
 42  *  
 43  * <pre class="prettyprint">
 44  * var actor = vxl.c.scene.getActorByName('example');
 45  * actor.setProperty('color',[1.0,0.0,0.0])
 46  * </pre>
 47  * @class Actors represent models (assets) in a Scene
 48  * @constructor
 49  */
 50 function vxlActor(model){
 51   
 52   this._bb = [0, 0, 0, 0, 0, 0];
 53   this._position 	= vec3.create([0, 0, 0]);
 54   this._translation = vec3.create([0, 0, 0]);
 55   this._centre      = vec3.create([0, 0, 0]);
 56   this._scale 		= vec3.create([1, 1, 1]);
 57   this._rotation 	= vec3.create([0, 0, 0]);
 58   this._matrix      = mat4.identity();
 59   this._renderers   = [];
 60   this._gl_buffers  = [];
 61   this._picking     = vxl.def.actor.picking.DISABLED;
 62   this._pickingCallback = undefined;
 63   this._pickingColor    = undefined; //used when the picking is vxl.def.actor.picking.OBJECT
 64   this._trackingCameras = [];
 65   
 66   this.UID = vxl.util.generateUID();
 67   this.scene        = undefined;
 68   this.clones       = 0;
 69   
 70   this.mode         = vxl.def.actor.mode.SOLID;
 71   this.cull         = vxl.def.actor.cull.NONE;
 72   this.visible      = true;
 73  
 74   this.mesh         = undefined;
 75   
 76   this.material     = new vxlMaterial();
 77   this.renderable   = undefined;
 78   
 79   if (model){
 80   	this.model 	    = model;
 81   	this.name 	    = model.name;
 82   	this.mode       = model.mode;
 83   	this._bb        = model.bb.slice(0);
 84   	this._centre    = vec3.set(model.centre, vec3.create());
 85   	this.material.getFrom(model);
 86   	
 87   	if (model.type == vxl.def.model.type.BIG_DATA){
 88   	    this.renderable = new vxlRenderable(this);
 89   	}
 90   }
 91   else{
 92       this.model = new vxlModel();
 93   }
 94   
 95   var e = vxl.events;
 96   vxl.go.notifier.publish(
 97       [
 98         e.ACTOR_MOVED,
 99         e.ACTOR_SCALED,
100         e.ACTOR_ROTATED,
101         e.ACTOR_CHANGED_COLOR,
102         e.ACTOR_CHANGED_SHADING,
103       ], this);
104 };

116 /**
117  * Sets the position of this actor. 
118  * 
119  * 
120  * @param {Number, Array, vec3} x it can be the x coordinate, a 3-dimensional Array or a vec3 (glMatrix)
121  * @param {Number} y if x is a number, then this parameter corresponds to the y-coordinate
122  * @param {Number} z if x is a number, then this parameter corresponds to the z-coordinate
123  */
124 vxlActor.prototype.setPosition = function (x,y,z){
125     var np = vxl.util.createVec3(x,y,z);
126     vec3.subtract(np, this._position, this._translation);
127     this._position = np;  
128     
129     var m = this._matrix;
130     m[12] = this._position[0];
131     m[13] = this._position[1];
132     m[14] = this._position[2];
133     m[15] = 1;
134     this._computeBoundingBox();
135     this._notifyTrackingCameras();
136     vxl.go.notifier.fire(vxl.events.ACTOR_MOVED, this);
137     
138     return this;
139 };

142 /**
143  * Translates the actor by a given vector 
144  * 
145  * @param {Number, Array, vec3} x it can be the x coordinate, a 3-dimensional Array or a vec3 (glMatrix)
146  * @param {Number} y if x is a number, then this parameter corresponds to the y-coordinate
147  * @param {Number} z if x is a number, then this parameter corresponds to the z-coordinate
148  */
149 vxlActor.prototype.translate = function (x,y,z){
150     this._translation = vxl.util.createVec3(x,y,z); 
151     
152     var m = this._matrix;
153     mat4.translate(m,this._translation);
154     this._position = vec3.createFrom(m[12], m[13], m[14]);
155     this._computeBoundingBox();
156     this._notifyTrackingCameras();
157     vxl.go.notifier.fire(vxl.events.ACTOR_MOVED, this);
158     
159     return this;
160 };

209 /**
210  * Scales this actor. 
211  * @param {Number, Array, vec3} s the scaling factor. The scaling factor is applied in all axes.
212  *
213  */
214 vxlActor.prototype.setScale = function(s,a,b){
215     if (s == 0 && a == undefined && b==undefined) return;
216     
217     
218     if (typeof(s)=="number" && a == undefined && b == undefined){
219         this._scale = vxl.util.createVec3(s,s,s);
220     }
221     else{
222         this._scale = vxl.util.createVec3(s,a,b);
223     }
224     
225     var m = this._matrix;
226     mat4.scale(m,this._scale);
227     this._computeBoundingBox();
228     vxl.go.notifier.fire(vxl.events.ACTOR_SCALED, this);
229     
230     return this;
231 };

233 /**
234  * Adds a tracking camera
235  * 
236  * @param{vxlCamera} camera the tracking camera
237  */
238 vxlActor.prototype.addTrackingCamera = function(camera){
239     if (camera instanceof vxlCamera && camera.type !=vxl.def.camera.type.TRACKING){
240         throw "vxlActor._addTrackingCamera ERROR: the selected camera is not set to tracking"
241     }
242     else if (camera instanceof vxlCamera){
243         this._trackingCameras.push(camera);
244     }
245     else{
246         throw "vxlActor._addTrackingCamera ERROR: the object passed as a parameter is not a vxlCamera"
247     }
248 };

250 /**
251  * Removes a tracking camera
252  * @param{vxlCamera} camera the tracknig camera
253  */
254 vxlActor.prototype.removeTrackingCamera = function(camera){
255     var idx = this._trackingCameras.indexOf(camera);
256     this._trackingCameras.splice(idx,1);
257 };

259 /**
260  * Notifies all the tracking cameras of the actor translation 
261  * @param {Object} camera
262  */
263 vxlActor.prototype._notifyTrackingCameras = function(){
264     for (var i=0, N = this._trackingCameras.length; i<N;i+=1){
265         this._trackingCameras[i].updateWithActor(this);
266     }

269 /**
270  * Computes the current bounding box for this actor.
271  * This method is called on demand by the scene or any other object. 
272  * The bounding box is NOT automatically recalculated when 
273  * moving or scaling an actor for performance reasons.
274  *  
275  */
276 vxlActor.prototype._computeBoundingBox = function(){
277 
278     var vs  = this.model.vertices;
279     var vsT = [];
280     var T = this._matrix;
281     
282     for(var i=0;i<vs.length;i=i+3){
283         var x = vxl.util.createVec3(vs[i],vs[i+1],vs[i+2]);
284         mat4.multiplyVec3(T, x);
285         vsT.push(x[0], x[1], x[2]);    
286     }
287     
288     var bbA = [vsT[0],vsT[1],vsT[2],vsT[0],vsT[1],vsT[2]];
289     
290     for(var i=0;i<vsT.length;i=i+3){
291         bbA[0] = Math.min(bbA[0],vsT[i]);
292         bbA[1] = Math.min(bbA[1],vsT[i+1]);
293         bbA[2] = Math.min(bbA[2],vsT[i+2]);
294         bbA[3] = Math.max(bbA[3],vsT[i]);
295         bbA[4] = Math.max(bbA[4],vsT[i+1]);
296         bbA[5] = Math.max(bbA[5],vsT[i+2]);
297     }
298     
299      
300     this._bb = bbA;
301 };

303 /**
304  * Returns an array with the bounding box vertices. Good for rendering the transformed
305  * bounding box (after geometric transformations) 
306  * @returns {Array} a 8-element array with the vertices that constitute the actor bounding box
307  */
308 vxlActor.prototype.getBoundingBoxVertices = function(){
309     var b = this._bb; 
310     return [
311         b[0], b[1], b[2],
312         b[0], b[4], b[2],
313         b[3], b[4], b[2],
314         b[3], b[1], b[2],
315         b[0], b[1], b[5],
316         b[0], b[4], b[5],
317         b[3], b[4], b[5],
318         b[3], b[1], b[5] 
319         ];
320 };

397 /**
398  * If the property exists, then it updates it
399  * @param {String} property name of the property 
400  * @param {Object} value  value to be set
401  * @param {String} scope indicates if the change is made at the actor level or at the model level
402  * valid values for scope are vxl.def.model and vxl.def.actor
403  * @TODO: if the property is position or scale then call the respective methods from here
404  */
405 vxlActor.prototype.setProperty = function(property, value, scope){
406     
407     
408     if (scope == vxl.def.actor || scope == undefined || scope == null){
409 
410         switch (property){
411             case 'position': this.setPosition(value);  break;
412             case 'scale':    this.setScale(value);     break;
413             case 'color':    this.setColor(value);     break;
414             case 'shading':  this.setShading(value);   break;
415             case 'texture':  this.setTexture(value);   break;
416             case 'opacity':  this.setOpacity(value);   break;
417             case 'shininess':this.setShininess(value); break;
418             default: this[property] = value; break;
419         }
420   
421 		vxl.go.console('Actor: The actor '+this.name+' has been updated. ['+property+' = '+value+']');
422 	}
423 	else if(scope == vxl.def.model){
424 		this.model[property] = value;
425 		vxl.go.console('vxlActor: The model '+this.model.namname+' has been updated. ['+property+' = '+value+']');
426 	}
427 	else{
428 		throw('vxlActor.setProperty. Scope:' + scope +' is not valid');
429 	}
430 	
431 	return this;
432 	
433 };

447 /**
448  * Returns an actor property if that property exists in the actor. Otherwise it will search 
449  * in the model. This method is used by the renderer. There are some cases where actors have local changes
450  * that are not reflected in the model. In these cases the renderer should pick the actor property
451  * over the model property
452  * @param {String} property the property name
453  * @returns {Object} the property or undefined if the property is not found 
454  */
455 vxlActor.prototype.getProperty = function(property){
456     
457     if (property == 'color'){
458         return this.materia.diffuse; //there's no real 'color' property.
459     }
460 	else if (this.hasOwnProperty(property)) {
461 		return this[property];
462 	}
463 	else if (this.material.hasOwnProperty(property)){
464 	   return this.material[property];   
465 	}
466 	else if (this.model.hasOwnProperty(property)){
467 		return this.model[property];
468 	}
469 	else {
470 		return undefined;
471 	}
472 };

474 /**
475  * Estimates the current position as
476  * the center of the current bounding box. 
477  * This method does not update the internal position of the actor
478  * it only returns an estimate based on the location of its bounding box.
479  */
480 vxlActor.prototype.computePosition = function(){
481 	bb = this._bb;
482 	var cc = this._position;
483 	
484 	cc[0] = (bb[3] + bb[0]) /2;
485 	cc[1] = (bb[4] + bb[1]) /2;
486 	cc[2] = (bb[5] + bb[2]) /2;
487 		
488 	cc[0] = Math.round(cc[0]*1000)/1000;
489 	cc[1] = Math.round(cc[1]*1000)/1000;
490 	cc[2] = Math.round(cc[2]*1000)/1000;
491 	
492 	return vec3.create(cc); 
493 };

519 /**
520 * Sets the lookup table for this actor.
521 * This method will only succeed if the model that this actor represents has scalars 
522 * @param {String} lutID the lookup table id. See {@link vxl.def.lut} for currently supported ids.
523 * @param {Number} min lowest value for interpolation
524 * @param {Number} max highest value for interpolation
525 */
526 vxlActor.prototype.setLookupTable = function(lutID,min,max){
527 	
528 	if (this.model.scalars == undefined) return;
529 	
530 	var self = this;
531 	
532 	function scheduledSetLookupTable(scalars){
533         var lut = vxl.go.lookupTableManager.get(lutID);
534         self.material.colors  = lut.getColors(self.model.scalars,min,max);
535         
536         //if(this.mesh){
537         //    this.mesh. //update mesh with vertex colors it may require access to the original index array
538        // }
539        
540        if (self.model.type == vxl.def.model.type.BIG_DATA){
541            self.renderable.update();
542        }
543 	}
544 	
545 	//Given that obtaining the colors can be a time consume op, it is deferred here.
546 	setTimeout(function(){scheduledSetLookupTable()},0);
547 	
548 	return this; 
549 };

579 /**
580  * Duplicates this actor.
581  * 
582  * Properties copied by REFERENCE:
583  * model,
584  * buffers,
585  * renderers
586  * 
587  * Everything else is copied by VALUE.
588  * 
589  * This method is fundamental to replicate objects in the scene, without having to duplicate
590  * the shared model. A cloned actor however can have different position, colors, properties, etc.
591  * 
592  * If a cloned actor modifies his internal model, any other actor that shares the model will be
593  * affected. 
594  * 
595  * The returned actor is not added to the scene automatically. It is up the
596  * programmer to determine the scene the cloned actor needs to be added to if any.
597  * 
598  * @see vxlModel
599  * @returns {vxlActor} an actor 
600  */
601 vxlActor.prototype.clone = function(){
602     this.clones++;
603 	
604 	var duplicate = new vxlActor(this.model);
605 	duplicate.setScene(this.scene);
606 	
607 	//TODO: duplicate.setMatrix(this._matrix); this method should update all the other variables (position, scale, rot, etc).
608      	
609 	//Now to save us some memory, let's SHARE the WebGL buffers that the current actor has already allocated'
610 	//duplicate.renderers = this._renderers;
611 	//duplicate.buffers   = this._gl_buffers;
612 
613 	duplicate.name     += '-'+this.clones; 
614 	return duplicate;
615 };

617 /**
618  * 
619  * @param {String} type one of the possible values for vxl.def.actor.picking
620  * @param {Object} callback a function that is invoked when a picking event occurs. This parameter is 
621  * required if the type (first argument) is different from vxl.def.actor.picking.DISABLED 
622  * the callback receives an actor object to operate over it.
623  */
624 vxlActor.prototype.setPicker = function(type, callback){
625     this._picking = type;
626     
627     switch(type){
628         case vxl.def.actor.picking.DISABLED: 
629             this._pickingCallback = undefined;
630             this.mesh = undefined;
631             this.renderable = undefined;
632             this.setVisualizationMode(vxl.def.actor.mode.SOLID); 
633             break;
634         
635         case vxl.def.actor.picking.CELL: 
636             
637             if (this.mesh == undefined){
638                 this.mesh = new vxlMesh(this); 
639                 this.mesh.setColor(this.material.diffuse);
640                 this.renderable = new vxlRenderable(this);
641                 this.setVisualizationMode(vxl.def.actor.mode.FLAT);           
642             };
643             
644             this._pickingCallback = callback;
645              
646             break;
647         case vxl.def.actor.picking.OBJECT:
648             this._pickingColor = vxl.go.picker.getColorFor(this);
649             this._pickingCallback = callback;
650             break;
651     }
652     
653     if (this.isPickable()){
654         for(var i=0, N = this.scene.views.length; i<N; i+=1){
655             var r = this.scene.views[i].renderer;
656             if(!r.isOffscreenEnabled()){
657                 r.enableOffscreen();
658             }
659         }
660     }
661     //@TODO: Disable when there are no pickable actors in the scene.
662     
663     return this;
664 };
