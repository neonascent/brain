 21 /**
 22  * 
 23  * Implements a rendering strategy that works with the vxl.def.glsl.lambert program. 
 24  * A rendering strategy allows decoupling rendering specific code (i.e. code that access and 
 25  * communicates with the program) from the actor.
 26  * @constructor
 27  * @class a vxlLambertStrategy object is selected by default as the strategy that an actor
 28  * will use for rendering.
 29  * 
 30  * @extends vxlBasicStrategy
 31  * @author Diego Cantor
 32  * 
 33  */
 34 function vxlLambertStrategy(renderer){
 35 	vxlBasicStrategy.call(this,renderer);
 36 
 37 	this._offscreen = false;
 38 	this._target    = undefined;
 39 	this._onPickingBuffer  = false;
 40 }

 43 /**
 44  * Renders the actors one by one
 45  * @param {vxlScene} scene the scene to render
 46  */
 47 vxlLambertStrategy.prototype.render = function(scene){
 48 
 49     //Updates the perspective matrix and passes it to the program
 50     var r       = this.renderer,
 51     trx     = r.transforms,
 52     prg     = r.prg,
 53     glsl    = vxl.def.glsl,
 54     gl      = r.gl;
 55     
 56     trx.calculatePerspective();
 57     trx.calculateModelView();
 58     prg.setUniform(glsl.PERSPECTIVE_MATRIX, trx.pMatrix);
 59     
 60     var elements = scene._actors.concat(scene.toys.list);
 61     var NUM = elements.length;
 62     
 63     if (scene.frameAnimation != undefined){
 64         scene.frameAnimation.update();
 65     }
 66     
 67     this._handlePicking(scene._actors);
 68     
 69 
 70     for(var i = 0; i < NUM; i+=1){
 71             this._renderActor(elements[i]);
 72     }
 73 };
 74 
 75 /**
 76  * @private
 77  */
 78 vxlLambertStrategy.prototype._handlePicking = function(actors){
 79     
 80     if (this._target == undefined) return; //quick fail if the target has not been defined
 81     
 82     if (this._offscreen){
 83         
 84         var gl = this.renderer.gl;
 85         
 86         this._onPickingBuffer = true;
 87         gl.bindFramebuffer(gl.FRAMEBUFFER, this._target.framebuffer);
 88         gl.clear(gl.COLOR_BUFFER_BIT | gl.DEPTH_BUFFER_BIT);
 89         gl.clearColor(0,0,0,1);
 90         
 91         for(var i = 0, NUM = actors.length; i < NUM; i+=1){
 92             if (actors[i].isPickable()){
 93                 this._renderActor(actors[i]);
 94             }
 95         }
 96         
 97         this._onPickingBuffer = false;
 98         gl.bindFramebuffer(gl.FRAMEBUFFER, null);
 99         var cc = this.renderer._clearColor;
100         this.renderer.clearColor(cc);
101     }
102 };

104 /**
105  * @private
106  */
107 vxlLambertStrategy.prototype._enableNormals = function(actor){
108     
109     var model = actor.model;
110     var gl = this.renderer.gl;
111     var prg = this.renderer.prg;
112     var buffers = this._gl_buffers[actor.UID];
113     var glsl    = vxl.def.glsl; 
114     
115     if(model.normals && actor.shading){
116         try{
117             gl.bindBuffer(gl.ARRAY_BUFFER, buffers.normal);
118             gl.bufferData(gl.ARRAY_BUFFER, new Float32Array(model.normals), gl.STATIC_DRAW);
119             
120             prg.enableAttribute(glsl.NORMAL_ATTRIBUTE);
121             prg.setAttributePointer(glsl.NORMAL_ATTRIBUTE,3,gl.FLOAT, false, 0,0);
122             
123         }
124         catch(err){
125             alert('There was a problem while rendering the actor ['+actor.name+']. The problem happened while handling the normal buffer. Error =' +err.description);
126             throw('There was a problem while rendering the actor ['+actor.name+']. The problem happened while handling the normal buffer. Error =' +err.description);
127         }
128     }
129 };

131 /**
132  * @private
133  */
134 vxlLambertStrategy.prototype._enableColors = function(actor){
135     
136     var model = actor.model;
137     var gl = this.renderer.gl;
138     var prg = this.renderer.prg;
139     var buffers = this._gl_buffers[actor.UID];
140     var glsl    = vxl.def.glsl;

142     if (actor.colors && actor.colors.length == actor.model.vertices.length){    
143         try{
144             prg.setUniform("uUseVertexColors", true);
145             gl.bindBuffer(gl.ARRAY_BUFFER, buffers.color);
146             gl.bufferData(gl.ARRAY_BUFFER, new Float32Array(actor.colors), gl.STATIC_DRAW);
147             
148             prg.enableAttribute(glsl.COLOR_ATTRIBUTE);
149             prg.setAttributePointer(glsl.COLOR_ATTRIBUTE, 3, gl.FLOAT, false, 0, 0);
150         }
151         catch(err){
152             alert('There was a problem while rendering the actor ['+actor.name+']. The problem happened while handling the color buffer. Error =' +err.description);
153             throw('There was a problem while rendering the actor ['+actor.name+']. The problem happened while handling the color buffer. Error =' +err.description);
154         }
155     }
156 };

158 /**
159  * Renders a solid actor
160  * @private
161  */
162 vxlLambertStrategy.prototype._renderSolid = function(actor){
163     
164     var buffers = this._gl_buffers[actor.UID];
165     var gl      = this.renderer.gl;
166     
167     this._enableNormals(actor);
168     this._enableColors(actor);    
169 
170     
171     gl.bindBuffer(gl.ELEMENT_ARRAY_BUFFER, buffers.index);
172     gl.bufferData(gl.ELEMENT_ARRAY_BUFFER, new Uint16Array(actor.model.indices), gl.STATIC_DRAW);
173     gl.drawElements(gl.TRIANGLES, actor.model.indices.length, gl.UNSIGNED_SHORT,0);
174 };

176 /**
177  * Renders an actor as a wireframe
178  */
179 vxlLambertStrategy.prototype._renderWireframe = function(actor){
180     
181 
182     var buffers = this._gl_buffers[actor.UID];
183     var gl      = this.renderer.gl;
184     var prg     = this.renderer.prg;
185     var glsl    = vxl.def.glsl;
186     
187    
188     this._enableNormals(actor);
189     
190     if (actor.name == 'floor'){
191         prg.disableAttribute(glsl.NORMAL_ATTRIBUTE); //floor does not have reflections. Always mate color!
192     }
193     
194     gl.bindBuffer(gl.ELEMENT_ARRAY_BUFFER, buffers.wireframe);
195     gl.bufferData(gl.ELEMENT_ARRAY_BUFFER, new Uint16Array(actor.model.wireframe), gl.STATIC_DRAW);
196     gl.drawElements(gl.LINES, actor.model.wireframe.length, gl.UNSIGNED_SHORT,0);
197 };

426 /**
427  * @private
428  */
429 vxlLambertStrategy.prototype._renderTextured = function(actor){
430     
431     var model   = actor.model;
432     var buffers = this._gl_buffers[actor.UID];
433     var texture = this._gl_textures[actor.UID]; 
434     var gl      = this.renderer.gl;
435     var prg     = this.renderer.prg;
436     var glsl    = vxl.def.glsl;   
437     
438     
439     if (model.texcoords){
440         try{
441             prg.setUniform("uUseTextures", true);
442             gl.bindBuffer(gl.ARRAY_BUFFER, buffers.texcoords);
443             gl.bufferData(gl.ARRAY_BUFFER, new Float32Array(model.texcoords), gl.STATIC_DRAW);
444             prg.setAttributePointer(glsl.TEXCOORD_ATTRIBUTE, 2, gl.FLOAT,false, 0,0);
445             prg.enableAttribute(glsl.TEXCOORD_ATTRIBUTE);
446         }
447         catch(err){
448             alert('There was a problem while rendering the actor ['+actor.name+']. The problem happened while handling the texture buffer. Error =' +err.description);
449             throw('There was a problem while rendering the actor ['+actor.name+']. The problem happened while handling the texture buffer. Error =' +err.description);
450         }
451     }
452     else{
453         //@TODO: Be more specific
454         throw('error');
455     }
456             
457     gl.activeTexture(gl.TEXTURE0);
458     gl.bindTexture(gl.TEXTURE_2D, texture);
459     gl.texParameteri(gl.TEXTURE_2D, gl.TEXTURE_WRAP_S, gl.CLAMP_TO_EDGE);
460     gl.texParameteri(gl.TEXTURE_2D, gl.TEXTURE_WRAP_T, gl.CLAMP_TO_EDGE);
461     gl.texParameteri(gl.TEXTURE_2D, gl.TEXTURE_MAG_FILTER, actor.texture.getMagFilter(gl));
462     gl.texParameteri(gl.TEXTURE_2D, gl.TEXTURE_MIN_FILTER, actor.texture.getMinFilter(gl));
463     prg.setUniform("uSampler", 0);
464     
465     gl.bindBuffer(gl.ELEMENT_ARRAY_BUFFER, buffers.index);
466     gl.drawElements(gl.TRIANGLES, model.indices.length, gl.UNSIGNED_SHORT,0);
467 };

469 /**
470  * @private
471  */
472 vxlLambertStrategy.prototype._handleCulling = function(actor){
473     var gl = this.renderer.gl;
474     
475     gl.disable(gl.CULL_FACE);
476     if (actor.cull != vxl.def.actor.cull.NONE){
477         gl.enable(gl.CULL_FACE);
478         switch (actor.cull){
479             case vxl.def.actor.cull.BACK: gl.cullFace(gl.BACK); break;
480             case vxl.def.actor.cull.FRONT: gl.cullFace(gl.FRONT); break;
481         }
482     } 
483     
484            
485 };

487 /**
488  * @private
489  */
490 vxlLambertStrategy.prototype._renderActor = function(actor){
491 
492     if (!actor.visible) return; //Quick and simple
493     if (this._renderpass == true && actor.name == 'floor') return;
494     
495 	var model 	= actor.model;
496     var buffers = this._gl_buffers[actor.UID];
497     var texture = this._gl_textures[actor.UID]; 
498     var gl      = this.renderer.gl;
499     var prg     = this.renderer.prg;
500 	var glsl    = vxl.def.glsl;   
501     var diffuse = [actor.color[0], actor.color[1], actor.color[2],actor.opacity];
502     
503     
504 	prg.disableAttribute(glsl.TEXCOORD_ATTRIBUTE);
505     prg.disableAttribute(glsl.COLOR_ATTRIBUTE);
506     prg.disableAttribute(glsl.NORMAL_ATTRIBUTE);
507     prg.disableAttribute(glsl.PICKING_COLOR_ATTRIBUTE);
508     prg.enableAttribute(glsl.VERTEX_ATTRIBUTE);
509     
510     prg.setUniform("uUseVertexColors", false);
511     prg.setUniform("uUseTextures", false);
512     prg.setUniform("uUseShading",actor.shading);
513     prg.setUniform("uMaterialDiffuse",diffuse);
514     
515     
516     this._handleCulling(actor);
517     this._applyActorTransform(actor);
518     
519     
520     if(this._onPickingBuffer){
521         this._renderPickingBuffer(actor);
522         return;
523     }
524     
525 		
526 	if (actor.mode != vxl.def.actor.mode.FLAT){
527     	try{
528             gl.bindBuffer(gl.ARRAY_BUFFER, buffers.vertex);
529             gl.bufferData(gl.ARRAY_BUFFER, new Float32Array(model.vertices.slice(0)), gl.STATIC_DRAW);
530     		prg.setAttributePointer(glsl.VERTEX_ATTRIBUTE, 3, gl.FLOAT, false, 0, 0);
531     	}
532         catch(err){
533             alert('There was a problem while rendering the actor ['+actor.name+']. The problem happened while handling the vertex buffer. Error =' +err.description);
534     		throw('There was a problem while rendering the actor ['+actor.name+']. The problem happened while handling the vertex buffer. Error =' +err.description);
535         }
536     }
537     
538     
539     var am = vxl.def.actor.mode;
540     
541     switch(actor.mode){
542         case am.SOLID:           this._renderSolid(actor);               break;
543         case am.WIREFRAME:       this._renderWireframe(actor);           break;
544         case am.POINTS:          this._renderPoints(actor);              break;
545         case am.LINES:           this._renderLines(actor);               break;
546         case am.BOUNDING_BOX:    this._renderBoundingBox(actor);         break;
547         case am.BB_AND_SOLID:    this._renderBoundingBoxAndSolid(actor); break;
548         case am.WIRED_AND_SOLID: this._renderWiredAndSolid(actor);       break;
549         case am.FLAT:            this._renderFlat(actor);                break;
550         case am.TEXTURED:        this._renderTextured(actor);            break;
551         default:
552             alert('There was a problem while rendering the actor ['+actor.name+']. The visualization mode: '+actor.mode+' is not valid.');
553             throw('There was a problem while rendering the actor ['+actor.name+']. The visualization mode: '+actor.mode+' is not valid.'); 
554     }
555     
556     //Cleaning up
557     gl.bindBuffer(gl.ARRAY_BUFFER, null);
558     gl.bindBuffer(gl.ELEMENT_ARRAY_BUFFER, null);
559 };
