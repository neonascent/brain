1 /**
  2  * Manages the lookup table files. The constructor will try to load all
  3  * the lookup tables defined in vxl.def.luts at once.
  4  * 
  5  * @class
  6  * @constructor
  7  */
  8 function vxlLookupTableManager(){
  9 	this.lutTimerID = 0;
 10 	this.tables = [];
 11 	vxl.go.notifier.addSource(vxl.events.DEFAULT_LUT_LOADED,this);
 12 	this.loadAll();
 13 };
 14 
 15 /**
 16  * Load a lookup table file
 17  * @param {String} name the filename of the lookup table to load
 18  */
 19 vxlLookupTableManager.prototype.load = function(name){
 20 		var self = this;
 21 		if (this.isLoaded(name)) return;
 22 
 23 	    var request = new XMLHttpRequest();
 24 	    request.open("GET", vxl.def.lut.folder+'/'+name+'.lut');
 25 	    request.onreadystatechange = function() {
 26 	      if (request.readyState == 4) {
 27 		    if(request.status == 404) {
 28 				alert (name + ' does not exist');
 29 				vxl.go.console('LookupTableManager: '+name + ' does not exist');
 30 			 }
 31 			else {
 32 				self.handle(name,JSON.parse(request.responseText));
 33 			}
 34 		  }
 35 	    };
 36 		request.send();
 37 };
 38 /**
 39  * Once the lookup table file is retrieved, this method adds it to the lookup table manager
 40  */
 41 vxlLookupTableManager.prototype.handle = function (ID, payload) {
 42 	var lut = new vxlLookupTable();
 43 	lut.load(ID,payload);
 44 	this.tables.push(lut);
 45 	
 46 	if (lut.ID == vxl.def.lut.main){
 47 		vxl.go.notifier.fire(vxl.events.DEFAULT_LUT_LOADED);
 48 	}
 49 };
 50 /**
 51  * Check if a lookup table has been loaded by this lookup table manager
 52  * @param {String} ID the id of the table to check
 53  */
 54 vxlLookupTableManager.prototype.isLoaded = function(ID){
 55 	for(var i=0;i<this.tables.length;i++){
 56 		if (this.tables[i].ID == ID) return true;
 57 	}
 58 	return false;
 59 };
 60 
 61 /**
 62  * Retrieves a lookup table
 63  * @param {String} ID id of the lookup table to retrieve
 64  */
 65 vxlLookupTableManager.prototype.get = function(ID){
 66 	for(var i=0;i<this.tables.length;i++){
 67 		if (this.tables[i].ID == ID) return this.tables[i];
 68 	}
 69 	return null;
 70 };
 71 
 72 /**
 73  * Returns a list with the names of all of the lookup tables that have been loaded.
 74  * @returns {Array} an array with the names of the lookup tables that have been loaded
 75  */
 76 vxlLookupTableManager.prototype.getAllLoaded = function(){
 77     var tablenames = [];
 78     for(var i=0;i<this.tables.length;i++){
 79         tablenames[i] = this.tables[i].ID;
 80     }
 81     return tablenames;
 82 };
 83 
 84 /**
 85  * Checks if all the lookup tables have been loaded
 86  */
 87 vxlLookupTableManager.prototype.allLoaded = function(){
 88 	//@TODO: think of a timeout to alter this state in the case not all tables are loaded (can this happen?)
 89 	return (vxl.def.lut.list.length == this.tables.length);
 90 };
 91 
 92 /**
 93  * Loads all the lookup tables defined in vxl.def.luts
 94  */
 95 vxlLookupTableManager.prototype.loadAll = function(){
 96 	for(var i=0;i<vxl.def.lut.list.length;i++){
 97 		this.load(vxl.def.lut.list[i]);
 98 	}
 99 };
100 
101 /**
102  * Creates the global lookup table manager and load all the lookup tables at once
103  */
104 vxl.go.lookupTableManager = new vxlLookupTableManager();
