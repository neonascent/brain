 18 vxlTrackerInteractor.prototype = new vxlViewInteractor();
 19 vxlTrackerInteractor.prototype.constructor = vxlViewInteractor;
 20 /**
 21  * @class Determines the application behaviour originated by mouse and keyboard events. 
 22  * @constructor
 23  * Interprets mouse and keyboard events and translate them to camera actions
 24  * @augments vxlViewInteractor
 25  */
 26 function vxlTrackerInteractor(){
 27     vxlViewInteractor.call(this);    
 28 	this.MOTION_FACTOR = 10.0;
 29 	this.task = vxl.def.interactor.task.NONE;
 30 	this.x = 0;
 31 	this.y = 0;
 32 	this.lastX = 0;
 33 	this.lastY = 0;
 34 	this.lastClickedX = 0;
 35 	this.lastClickedY = 0;
 36 	this.ctrlPressed = false;
 37 	this.altPressed = false;
 38 	this.keyPressed = 0;
 39 	this.button = -1;
 40 	this.dragging = false;
 41 	this.dragndrop = false;
 42 	this.isMac = vxl.util.isMac();
 43 	
 44 };

 53 /**
 54  * Reacts to the window onKeyDown event 
 55  */
 56 vxlTrackerInteractor.prototype.onKeyDown = function(ev){
 57     this.keyPressed = ev.keyCode;
 58     this.altPressed = ev.altKey;
 59     this.shiftPressed = ev.shiftKey;
 60     
 61     var camera = this.camera;
 62     
 63     
 64     if (!this.altPressed && !this.shiftPressed){
 65         switch(this.keyPressed){
 66               case 38:camera.changeElevation(10);  break;
 67               case 40:camera.changeElevation(-10); break;
 68               case 37:camera.changeAzimuth(-10);   break;
 69               case 39:camera.changeAzimuth(10);    break;
 70               case 70: this.view.fullscreen(true); break;
 71               case 88: this.view.fullscreen(false); break;
 72               default: break;
 73         }
 74     }
 75     //PANNING
 76     else if(this.shiftPressed && this.keyPressed !=17) {
 77         var px = 0;
 78         var py = 0;
 79         switch(this.keyPressed){
 80             case 38:py = 10; break;
 81             case 40:py = -10;break;
 82             case 37:px = -10;break;
 83             case 39:px = 10; break;
 84             default: break;
 85         }
 86         if(px != 0 || py !=0){
 87             this.pan(px,py);
 88         }
 89     }
 90     this.camera.refresh();
 91 };

122 /**
123  *Reacts to the canvas onmousemove event 
124  */
125 vxlTrackerInteractor.prototype.onMouseMove = function(ev){
126 
127 	this.lastX         = this.x;
128 	this.lastY         = this.y;
129 	this.x             = ev.clientX;
130     this.y             = ev.clientY;
131 
132 	if (!this.dragging) return;
133     if (this.button !=0) return;  
134 	
135 	this.ctrlPressed 	= ev.ctrlKey;
136 	
137 	if (this.isMac && ev.metaKey){
138 	    this.ctrlPressed = true;
139 	}
140 	
141 	this.altPressed 	= ev.altKey;
142 	this.shiftPressed 	= ev.shiftKey;
143 	
144 	var rx = this.x - this.lastX;
145 	var ry = this.y - this.lastY;
146 		 
147     if (this.ctrlPressed && !this.shiftPressed){ 
148 		this.dolly(ry);
149 	}
150 	else if (this.shiftPressed && !this.ctrlPressed){
151 		this.pan(rx,ry);
152 	}
153 	else if (this.ctrlPressed && this.shiftPressed){
154 	    this.roll(ry);
155 	}
156 	else {
157 
158         this.rotate(rx,ry);
159     }
160 	
161 	this.camera.refresh();
162 };

165 /**
166  *  Implements the behaviour of the tracker on the ondragover mouse event
167  */
168 vxlTrackerInteractor.prototype.onDragOver = function(event){
169     event.stopPropagation();
170     event.preventDefault();
171     
172         
173     if (this.view.dragndrop){
174         if (!this.dragndrop){
175             this.bgcolor = this.view.backgroundColor.slice(0);
176             this.dragndrop = true;
177         }
178         event.dataTransfer.dropEffect = 'copy';
179         this.view.setBackgroundColor(0,0.514,0.678);
180     }
181 };
182 
183 /**
184  * Implements the behaviour of the tracker on the ondragleave mouse event 
185  */
186 vxlTrackerInteractor.prototype.onDragLeave = function(event){
187     event.stopPropagation();
188     event.preventDefault();
189 
190     if (this.view.dragndrop){
191         event.dataTransfer.dropEffect = 'copy';
192         this.view.setBackgroundColor(this.bgcolor);
193         this.dragndrop = false;
194     }
195     
196 };

198 /**
199  * Implements the behaviour of the tracker on the ondrop mouse event
200  * @TODO: this method only works for VTK ascii files. Review other formats 
201  */
202 vxlTrackerInteractor.prototype.onDrop = function(event){
203     event.stopPropagation();
204     event.preventDefault();
205     if (!this.view.dragndrop) return; //the view is configured to not accept dnd
206     this.dragndrop = false;
207     this.view.setBackgroundColor(this.bgcolor);
208     
209     var files = event.dataTransfer.files;
210     var reader = new vxlVTKReader(this.view.scene);
211     if (reader.isSupported()){
212         reader.read(files[0]);
213     }
214     else {
215         throw 'vxlTrackerInteractor.drop: File API is not supported on this browser';
216     }
217 };

238 /**
239  * Internal method used by this tracker to perform rolling
240  * @param {Number} value the rolling angle
241  */
242 vxlTrackerInteractor.prototype.roll = function(value){
243     
244     this.task = vxl.def.interactor.task.ROLL;
245     
246     var canvas = this.camera.view.canvas;
247     
248     var dy = -20.0 / canvas.width;
249     
250     var rotY = value * dy * this.MOTION_FACTOR;
251     
252     this.camera.rotate(0,0,rotY);
253 };

257 /**    this.dragndrop = false;
258  * Internal method used by this tracker to rotate the camera.
259  * @param {Number} dx the rotation on the X axis (elevation)
260  * @param {Number} dy the rotation on the Y axis (azimuth)
261  */
262 vxlTrackerInteractor.prototype.rotate = function(rx, ry){
263 	
264 	this.task = vxl.def.interactor.task.ROTATE;
265 	
266 	var canvas = this.camera.view.canvas;
267 	var dx = -20.0 / canvas.height;
268 	var dy = -20.0 / canvas.width;
269 	var motionFactorX = this.MOTION_FACTOR;
270 	var motionFactorY = this.MOTION_FACTOR;
271 	if (rx*rx > 2 * ry *ry){
272 	    motionFactorY *= 0.5;
273 	}
274 	else if (ry*ry > 2* rx*rx){
275 	    motionFactorX *= 0.5;
276 	}
277 	
278 	var rotX = rx * dx * motionFactorX;
279 	var rotY = ry * dy * motionFactorY;
280 	
281 	this.camera.rotate(rotX, rotY);
282 };

284 /**
285  * Internal method used by this tracker to perform panning 
286  * @param {Object} dx
287  * @param {Object} dy
288  */
289 vxlTrackerInteractor.prototype.pan = function(dx,dy){
290 
291 	this.task = vxl.def.interactor.task.PAN;
292 	
293 	var camera = this.camera;
294 	var canvas = camera.view.canvas;
295 	var scene = camera.view.scene;    this.dragndrop = false;
296 	var dimMax = Math.max(canvas.width, canvas.height);
297 	var deltaX = 1 / dimMax;
298 	var deltaY = 1 / dimMax;
299 	var max = scene.bb.max();
300     var ndx = dx * deltaX * this.MOTION_FACTOR * max / 2;
301 	var ndy = -dy * deltaY * this.MOTION_FACTOR * max / 2;
302 
303 	camera.pan(ndx,ndy);
304 };
