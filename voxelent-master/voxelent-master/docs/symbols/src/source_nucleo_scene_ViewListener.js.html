 16 /**
 17  * @class
 18  * @constructor
 19  */
 20 function vxlViewListener(vw){
 21 	this.view = vw;
 22 	this.interactor = new vxlTrackballCameraInteractor();
 23 	this.picker = new vxlPickerInteractor();
 24 	this.update();
 25 }
 26 
 27 vxlViewListener.prototype.setInteractor = function(i){
 28 	this.interactor = i;
 29 	this.update();
 30 }
 31 
 32 
 33 vxlViewListener.prototype.update = function(){
 34 	var self = this;
 35 	var canvas = this.view.canvas;
 36 	
 37 	var camera = this.view.cameraman.active;
 38 	var interactor = this.interactor;
 39 	
 40 	interactor.connectTo(camera);
 41 
 42 	canvas.onmousedown = function(ev) {
 43 		interactor.onMouseDown(ev);
 44     };
 45 
 46     canvas.onmouseup = function(ev) {
 47 		interactor.onMouseUp(ev);
 48     };
 49 	
 50 	canvas.onmousemove = function(ev) {
 51 		interactor.onMouseMove(ev);
 52     };
 53 	
 54 	window.onkeydown = function(ev){
 55 		interactor.onKeyDown(ev);
 56 	}
 57 	
 58 	window.onkeyup = function(ev){
 59 		interactor.onKeyUp(ev);
 60 	}
 61 }
