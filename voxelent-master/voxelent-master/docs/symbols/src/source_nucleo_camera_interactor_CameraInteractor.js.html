 18 	this.camera = null;
 19 }

 22 vxlCameraInteractor.prototype.connectTo = function(c){
 23 	if (c != null){
 24 		if (c instanceof vxlCamera){
 25 			this.camera = c;
 26 			this.camera.interactor = this;
 27 		}
 28 		else {
 29 			throw(' The object '+c+' is not a valid camera');
 30 		}
 31 	}
 32 }
