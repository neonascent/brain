 17 /**
 18  * @constructor
 19  * @class
 20  * Implements the default rendering strategy 
 21  * 
 22  * 
 23  * A vxlRenderStrategy object is selected by default as the strategy that an actor
 24  * will use for rendering.
 25  * 
 26  * A rendering strategy allows decoupling rendering specific code (i.e. code that access and 
 27  * 	communicates with the program) from the actor.
 28  * @author Diego Cantor
 29  * 
 30  */
 31 function vxlRenderStrategy(renderer){
 32 	this.renderer = renderer;
 33 	this._gl_buffers  = {};
 34 	this._gl_textures = {};
 35 	this._offscreen = false;
 36     this._target    = undefined;
 37     this._onPickingBuffer  = false;
 38     this._debug_picking_flag = false; //used only for debuggin purposes
 39 	
 40 	if (renderer){
 41 	var gl = this.renderer.gl;
 42 	
 43 	gl.enable(gl.BLEND);
 44 	gl.blendFunc(gl.SRC_ALPHA, gl.ONE_MINUS_SRC_ALPHA);
 45 	gl.enable(gl.DEPTH_TEST);
 46 	gl.depthFunc(gl.LESS);
 47 	gl.clearDepth(1.0);
 48 	gl.disable(gl.CULL_FACE);
 49     }	
 50 }

 52 /**
 53  * Implements basic allocation of memory. Creates the WebGL buffers for the actor
 54  * @param {vxlScene} scene the scene to allocate memory for
 55   */
 56 vxlRenderStrategy.prototype.allocate = function(scene){
 57     var elements = scene._actors.concat(scene.toys.list);
 58     var NUM = elements.length;
 59     
 60     for(var i = 0; i < NUM; i+=1){
 61         this._allocateActor(elements[i]);
 62     }
 63 };

 66  * @param {vxlScene} scene the scene to deallocate memory from

 69     //DO NOTHING. THE DESCENDANTS WILL.
 70 };

 72 /**
 73  * Receives one actor and returns the GL buffers
 74  */
 75 vxlRenderStrategy.prototype._allocateActor = function(actor){
 76    
 77     if (this._gl_buffers[actor.UID] != undefined){
 78         if (actor.dirty){ //@TODO: hmmmm frowning on dirty property this is a hack
 79             this._reallocateActor(actor); 
 80         }
 81         return;
 82     }
 83    	
 84 	var gl = this.renderer.gl;
 85 	var model = actor.model;
 86     var buffers = {};
 87 	
 88 	//Vertex Buffer
 89 	buffers.vertex = gl.createBuffer();
 90 	
 91 	
 92 	//Index Buffer
 93 	if (model.indices != undefined){
 94 		buffers.index = gl.createBuffer();
 95 	}
 96 	
 97 	//Normals Buffer
 98 	if (model.normals){
 99 		buffers.normal = gl.createBuffer();
100 	}
101 	
102 	//Color Buffer for scalars
103 	if (model.scalars != undefined || model.colors != undefined){
104 		buffers.color = gl.createBuffer(); //we don't BIND values or use the buffers until the lut is loaded and available
105 	}
106 	
107 	//Wireframe Buffer 
108 	if (model.wireframe != undefined){
109 		buffers.wireframe = gl.createBuffer();
110 	}
111 	
112 	//Bounding box Buffer 
113     buffers.bb = gl.createBuffer();
114 	
115 	//Texture Coords Buffer
116 	if (model.texture){
117 	    buffers.texcoords = gl.createBuffer();
118 	}
119 	   
120     //Cleaning up
121     gl.bindBuffer(gl.ARRAY_BUFFER, null);
122     gl.bindBuffer(gl.ELEMENT_ARRAY_BUFFER, null);
123 	
124 	
125 	this._gl_buffers[actor.UID] = buffers;
126 	this._reallocateActor(actor);
127 };

130 /**
131  * Receives one actor and returns the GL buffers
132  */
133 vxlRenderStrategy.prototype._reallocateActor = function(actor){
134    
135     
136     vxlRenderStrategy
137     var gl = this.renderer.gl;
138     var model = actor.model;
139     var buffers = this._gl_buffers[actor.UID];
140     
141     gl.bindBuffer(gl.ARRAY_BUFFER, buffers.vertex);
142     gl.bufferData(gl.ARRAY_BUFFER, new Float32Array(model.vertices), gl.STATIC_DRAW);
143     
144     //Index Buffer
145     if (model.indices != undefined){    
146     //Cleaning up
147     gl.bindBuffer(gl.ARRAY_BUFFER, null);
148     gl.bindBuffer(gl.ELEMENT_ARRAY_BUFFER, null);
149         gl.bindBuffer(gl.ELEMENT_ARRAY_BUFFER, buffers.index);
150         gl.bufferData(gl.ELEMENT_ARRAY_BUFFER, new Uint16Array(model.indices), gl.STATIC_DRAW);
151     }
152     
153     //Normals Buffer
154     if (model.normals){
155         gl.bindBuffer(gl.ARRAY_BUFFER, buffers.normal);
156         gl.bufferData(gl.ARRAY_BUFFER, new Float32Array(model.normals), gl.STATIC_DRAW);
157     }
158     
159     //Color Buffer for scalars
160     if (model.scalars != undefined || model.colors != undefined){
161         buffers.color = gl.createBuffer(); //we don't BIND values or use the buffers until the lut is loaded and available
162     }
163     
164     //Wireframe Buffer 
165     if (model.wireframe != undefined){
166         gl.bindBuffer(gl.ELEMENT_ARRAY_BUFFER, buffers.wireframe);
167         gl.bufferData(gl.ELEMENT_ARRAY_BUFFER, new Uint16Array(model.wireframe), gl.STATIC_DRAW);
168     }   
169     //Cleaning up
170     gl.bindBuffer(gl.ARRAY_BUFFER, null);
171     gl.bindBuffer(gl.ELEMENT_ARRAY_BUFFER, null);
172     
173     //Bounding box Buffer 
174     gl.bindBuffer(gl.ELEMENT_ARRAY_BUFFER, buffers.bb);
175     gl.bufferData(gl.ELEMENT_ARRAY_BUFFER, new Uint16Array(vxlModel.BB_INDICES), gl.STATIC_DRAW);
176     
177     
178     //Texture Coords Buffer
179     if (model.texture && actor.material.texture.loaded){
180         gl.bindBuffer(gl.ARRAY_BUFFER, buffers.texcoords);
181         gl.bufferData(gl.ARRAY_BUFFER, new Float32Array(model.texcoords), gl.STATIC_DRAW);
182         
183         this._gl_textures[actor.UID] = gl.createTexture();
184         gl.bindTexture(gl.TEXTURE_2D, this._gl_textures[actor.UID]);
185         gl.texImage2D(gl.TEXTURE_2D, 0, gl.RGBA, gl.RGBA, gl.UNSIGNED_BYTE, actor.texture.image);
186         gl.generateMipmap(gl.TEXTURE_2D);
187         
188         gl.bindTexture(gl.TEXTURE_2D, null);
189     }
190     
191     
192     //Cleaning up
193     gl.bindBuffer(gl.ARRAY_BUFFER, null);
194     gl.bindBuffer(gl.ELEMENT_ARRAY_BUFFER, null);
195 };
196 /**
197  * Passes the matrices to the shading program
198  * @param {vxlActor} the actor 
199  * we will update each Model-View matrix of each renderer according to
200  * the actor position,scale and rotation.
201  * @private
202  */
203 vxlRenderStrategy.prototype._applyActorTransform = function(actor){
204     
205     var r		= this.renderer;
206     var trx 	= r.transforms
207     var	prg 	= r.prg;
208     var glsl 	= vxl.def.glsl;
209 
210     trx.push();
211         mat4.multiply(trx.mvMatrix, actor._matrix);
212 	    prg.setUniform(glsl.MODEL_VIEW_MATRIX,	r.transforms.mvMatrix);
213 
214 	    trx.calculateModelViewPerspective();
215         prg.setUniform(glsl.MVP_MATRIX, r.transforms.mvpMatrix);
216         
217         trx.calculateNormal(); 
218         prg.setUniform(glsl.NORMAL_MATRIX, r.transforms.nMatrix);
219     
220     trx.pop();
221  };

224  /**
225  * Passes the matrices to the shading program
226  * @param {vxlActor} the actor 
227  * we will update each Model-View matrix of each renderer according to
228  * the actor position,scale and rotation.
229  * @private
230  */
231 vxlRenderStrategy.prototype._applyGlobalTransform = function(){
232     
233     var r       = this.renderer;
234     var trx     = r.transforms
235     var prg     = r.prg;
236     var glsl    = vxl.def.glsl;
237 
238     prg.setUniform(glsl.MODEL_VIEW_MATRIX,  r.transforms.mvMatrix);
239     trx.calculateModelViewPerspective();
240     prg.setUniform(glsl.MVP_MATRIX, r.transforms.mvpMatrix);
241     trx.calculateNormal(); 
242     prg.setUniform(glsl.NORMAL_MATRIX, r.transforms.nMatrix);
243     
244  };

247  /**
248  * Renders the actors one by one
249  * @param {vxlScene} scene the scene to render
250  */
251 vxlRenderStrategy.prototype.render = function(scene){
252 
253     //Updates the perspective matrix and passes it to the program
254     var r       = this.renderer,
255     trx     = r.transforms,
256     prg     = r.prg,
257     glsl    = vxl.def.glsl,
258     gl      = r.gl;
259     
260     trx.calculatePerspective();
261     trx.calculateModelView();
262     
263     
264     var elements = scene._actors.concat(scene.toys.list);
265     var NUM = elements.length;
266     
267     //is this supposed to be here?
268     //----------------------------------------------------//
269     if (scene.frameAnimation != undefined){
270         scene.frameAnimation.update();
271     }
272     //----------------------------------------------------//
273     
274     this._handlePicking(scene._actors);
275     
276     if (this._debug_picking_flag){
277         gl.clearColor(0,0,0,1);
278         gl.clear(gl.COLOR_BUFFER_BIT | gl.DEPTH_BUFFER_BIT);
279     };
280     
281 
282     for(var i = 0; i < NUM; i+=1){
283             this._renderActor(elements[i]);
284     }
285 };

287 /**
288  * @private
289  */
290 vxlRenderStrategy.prototype._handlePicking = function(actors){
291     
292     if (this._target == undefined) return; //quick fail if the target has not been defined
293     
294     if (this._offscreen){
295         
296         var gl = this.renderer.gl;
297         
298         this._onPickingBuffer = true;
299         
300         gl.bindFramebuffer(gl.FRAMEBUFFER, this._target.framebuffer);
301         
302         
303         gl.clearColor(0,0,0,1);
304         gl.clear(gl.COLOR_BUFFER_BIT | gl.DEPTH_BUFFER_BIT);
305         
306         for(var i = 0, NUM = actors.length; i < NUM; i+=1){
307             if (actors[i].isPickable()){
308                 this._renderActor(actors[i]);
309             }
310         }
311         
312         this._onPickingBuffer = false;
313         
314         gl.bindFramebuffer(gl.FRAMEBUFFER, null);
315         
316     }
317 };

319 /**
320  * @private
321  */
322 vxlRenderStrategy.prototype._enableNormals = function(actor){
323     
324     var model = actor.model;
325     var gl = this.renderer.gl;
326     var prg = this.renderer.prg;
327     var buffers = this._gl_buffers[actor.UID];
328     var glsl    = vxl.def.glsl; 
329     
330     if(model.normals && actor.material.shading){
331         try{
332             gl.bindBuffer(gl.ARRAY_BUFFER, buffers.normal);
333             gl.bufferData(gl.ARRAY_BUFFER, new Float32Array(model.normals), gl.STATIC_DRAW);
334             
335             prg.enableAttribute(glsl.NORMAL_ATTRIBUTE);
336             prg.setAttributePointer(glsl.NORMAL_ATTRIBUTE,3,gl.FLOAT, false, 0,0);
337             
338         }
339         catch(err){
340             alert('There was a problem while rendering the actor ['+actor.name+']. The problem happened while handling the normal buffer. Error =' +err.description);
341             throw('There was a problem while rendering the actor ['+actor.name+']. The problem happened while handling the normal buffer. Error =' +err.description);
342         }
343     }
344 };

373 /**
374  * Renders a solid actor
375  * @private
376  */
377 vxlRenderStrategy.prototype._renderSolid = function(actor){
378     
379     var buffers = this._gl_buffers[actor.UID];
380     var gl      = this.renderer.gl;
381     
382     this._enableNormals(actor);
383     this._enableColors(actor);    
384 
385     
386     gl.bindBuffer(gl.ELEMENT_ARRAY_BUFFER, buffers.index);
387     gl.bufferData(gl.ELEMENT_ARRAY_BUFFER, new Uint16Array(actor.model.indices), gl.STATIC_DRAW);
388     gl.drawElements(gl.TRIANGLES, actor.model.indices.length, gl.UNSIGNED_SHORT,0);
389 };

566     }
567     catch(err){
568         alert('There was a problem while rendering the actor ['+actor.name+']. The problem happened while handling the vertex buffer. Error =' +err.description);
569         throw('There was a problem while rendering the actor ['+actor.name+']. The problem happened while handling the vertex buffer. Error =' +err.description);
570     }
571     
572    if (r.colors){    
573             try{
574                 prg.setUniform("uUseVertexColors", true);
575                 gl.bindBuffer(gl.ARRAY_BUFFER, buffers.color);
576                 gl.bufferData(gl.ARRAY_BUFFER, new Float32Array(r.colors), gl.STATIC_DRAW);
577

583                 throw('There was a problem while rendering the actor ['+actor.name+']. The problem happened while handling the color buffer. Error =' +err.description);
584             }
585     }
586     
587     gl.bindBuffer(gl.ELEMENT_ARRAY_BUFFER, buffers.index);
588     gl.bufferData(gl.ELEMENT_ARRAY_BUFFER, new Uint16Array(r.indices), gl.STATIC_DRAW);
589     gl.drawElements(gl.TRIANGLES, r.indices.length, gl.UNSIGNED_SHORT,0);
590

594 /**
595  * This method takes care of rendering the scene in the background buffer. Only the 
596  * actors that are pickable reach this method (see _handlePicking) so the isPickable 
597  * validation (actor.isPickable) is not necessary here.
598  * 
599  * This method makes sure the correct version of the object is rendered in the background
600  * buffer
601  * 
602  * if the object picking method is OBJECT then the same object is rendered but using
603  * the picking color (actor._pickingColor) as the diffuse material uniform. 
604  *
605  * Otherwise, when the object picking mode is CELL, then the mesh renderable is rendered. 
606  * The mesh renderable colors every cell differently (drawback in performance)
607  *    
608  * @private
609  */
610 vxlRenderStrategy.prototype._renderPickingBuffer = function(actor){
611     
612     var model   = actor.model;
613     var buffers = this._gl_buffers[actor.UID];
614     var texture = this._gl_textures[actor.UID]; 
615     var gl      = this.renderer.gl;
616     var prg     = this.renderer.prg;
617     var glsl    = vxl.def.glsl;
618     
619     prg.setUniform("uUseShading",false);
620     
621     if (actor._picking == vxl.def.actor.picking.CELL){
622     
623       
624         if (actor.mesh == undefined){
625             return; // we just fail safely. We will render it whenever the actor is ready ;-)
626                     // the mesh is generated inside actor.setPicker (see actor.setPicker method)
627         }
628         
629         /******************************************/
630         // THIS IS THE MAGIC TA DA!
631          r = actor.mesh.renderable;
632          if (r == undefined) return; 
633         /******************************************/
634         
635         
636         try{
637             gl.bindBuffer(gl.ARRAY_BUFFER, buffers.vertex);
638             gl.bufferData(gl.ARRAY_BUFFER, new Float32Array(r.vertices), gl.STATIC_DRAW);
639             prg.setAttributePointer(glsl.VERTEX_ATTRIBUTE, 3, gl.FLOAT, false, 0, 0);
640         }
641         catch(err){
642             alert('There was a problem while rendering the actor ['+actor.name+']. The problem happened while handling the vertex buffer. Error =' +err.description);
643             throw('There was a problem while rendering the actor ['+actor.name+']. The problem happened while handling the vertex buffer. Error =' +err.description);
644         }
645         
646         if (r.pickingColors && r.pickingColors.length == r.vertices.length){    
647             try{
648                 prg.setUniform("uUseVertexColors", true);
649                 gl.bindBuffer(gl.ARRAY_BUFFER, buffers.color);
650                 gl.bufferData(gl.ARRAY_BUFFER, new Float32Array(r.pickingColors), gl.STATIC_DRAW);
651                 
652                 prg.enableAttribute(glsl.COLOR_ATTRIBUTE);
653                 prg.setAttributePointer(glsl.COLOR_ATTRIBUTE, 3, gl.FLOAT, false, 0, 0);
654             }
655             catch(err){
656                 alert('There was a problem while rendering the actor ['+actor.name+']. The problem happened while handling the color buffer. Error =' +err.description);
657                 throw('There was a problem while rendering the actor ['+actor.name+']. The problem happened while handling the color buffer. Error =' +err.description);
658             }
659         }
660         
661         gl.bindBuffer(gl.ELEMENT_ARRAY_BUFFER, buffers.index);
662         gl.bufferData(gl.ELEMENT_ARRAY_BUFFER, new Uint16Array(r.indices), gl.STATIC_DRAW);
663         gl.drawElements(gl.TRIANGLES, r.indices.length, gl.UNSIGNED_SHORT,0);
664     
665     }
666     else if (actor._picking == vxl.def.actor.picking.OBJECT){
667         
668         prg.setUniform("uMaterialDiffuse",actor._pickingColor.concat(1.0));
669         try{
670             gl.bindBuffer(gl.ARRAY_BUFFER, buffers.vertex);
671             gl.bufferData(gl.ARRAY_BUFFER, new Float32Array(model.vertices), gl.STATIC_DRAW);
672             prg.setAttributePointer(glsl.VERTEX_ATTRIBUTE, 3, gl.FLOAT, false, 0, 0);
673         }
674         catch(err){
675             alert('There was a problem while rendering the actor ['+actor.name+']. The problem happened while handling the vertex buffer. Error =' +err.description);
676             throw('There was a problem while rendering the actor ['+actor.name+']. The problem happened while handling the vertex buffer. Error =' +err.description);
677         }
678         gl.bindBuffer(gl.ELEMENT_ARRAY_BUFFER, buffers.index);
679         gl.bufferData(gl.ELEMENT_ARRAY_BUFFER, new Uint16Array(model.indices), gl.STATIC_DRAW);
680         gl.drawElements(gl.TRIANGLES, model.indices.length, gl.UNSIGNED_SHORT,0);
681     }
682     
683     //Cleaning up
684     gl.bindBuffer(gl.ARRAY_BUFFER, null);
685     gl.bindBuffer(gl.ELEMENT_ARRAY_BUFFER, null);
686      
687 };

689 /**
690  * @private
691  */
692 vxlRenderStrategy.prototype._renderTextured = function(actor){
693     
694     var model   = actor.model;
695     var buffers = this._gl_buffers[actor.UID];
696     var texture = this._gl_textures[actor.UID]; 
697     var gl      = this.renderer.gl;
698     var prg     = this.renderer.prg;
699     var glsl    = vxl.def.glsl;   
700     
701     if (!actor.material.texture.loaded){
702         this._renderSolid(actor);
703         return;
704     }
705     
706     if (model.texcoords){
707         try{
708             prg.setUniform("uUseTextures", true);
709             gl.bindBuffer(gl.ARRAY_BUFFER, buffers.texcoords);
710             gl.bufferData(gl.ARRAY_BUFFER, new Float32Array(model.texcoords), gl.STATIC_DRAW);
711             prg.setAttributePointer(glsl.TEXCOORD_ATTRIBUTE, 2, gl.FLOAT,false, 0,0);
712             prg.enableAttribute(glsl.TEXCOORD_ATTRIBUTE);
713         }
714         catch(err){
715             alert('There was a problem while rendering the actor ['+actor.name+']. The problem happened while handling the texture buffer. Error =' +err.description);
716             throw('There was a problem while rendering the actor ['+actor.name+']. The problem happened while handling the texture buffer. Error =' +err.description);
717         }
718     }
719     else{
720         //@TODO: Be more specific
721         throw('error');
722     }
723             
724     gl.activeTexture(gl.TEXTURE0);
725     gl.bindTexture(gl.TEXTURE_2D, texture);
726     gl.texParameteri(gl.TEXTURE_2D, gl.TEXTURE_WRAP_S, gl.CLAMP_TO_EDGE);
727     gl.texParameteri(gl.TEXTURE_2D, gl.TEXTURE_WRAP_T, gl.CLAMP_TO_EDGE);
728     gl.texParameteri(gl.TEXTURE_2D, gl.TEXTURE_MAG_FILTER, actor.texture.getMagFilter(gl));
729     gl.texParameteri(gl.TEXTURE_2D, gl.TEXTURE_MIN_FILTER, actor.texture.getMinFilter(gl));
730     prg.setUniform("uSampler", 0);
731     
732     gl.bindBuffer(gl.ELEMENT_ARRAY_BUFFER, buffers.index);
733     gl.drawElements(gl.TRIANGLES, model.indices.length, gl.UNSIGNED_SHORT,0);
734 };

736 /**
737  * @private
738  */
739 vxlRenderStrategy.prototype._handleCulling = function(actor){
740     var gl = this.renderer.gl;
741     
742     gl.disable(gl.CULL_FACE);
743     if (actor.cull != vxl.def.actor.cull.NONE){
744         gl.enable(gl.CULL_FACE);
745         switch (actor.cull){
746             case vxl.def.actor.cull.BACK: gl.cullFace(gl.BACK); break;
747             case vxl.def.actor.cull.FRONT: gl.cullFace(gl.FRONT); break;
748         }
749     } 
750     
751            
752 };

754 /**
755  * @private
756  */
757 vxlRenderStrategy.prototype._renderActor = function(actor){
758 
759     if (!actor.visible) return; //Quick and simple
760 
761     var model   = actor.model;
762     var buffers = this._gl_buffers[actor.UID];
763     var texture = this._gl_textures[actor.UID]; 
764     var gl      = this.renderer.gl;
765     var prg     = this.renderer.prg;
766     var glsl    = vxl.def.glsl;   
767     var trx     = this.renderer.transforms;
768     var diffuse = [actor.material.diffuse[0], 
769                    actor.material.diffuse[1], 
770                    actor.material.diffuse[2],
771                    actor.material.opacity];
772     
773     
774     /**
775      * Setting the program has to be done BEFORE setting ANY uniform or attribute
776      */
777     if(actor.shininess > 0){
778         this.renderer.setProgram(vxl.def.glsl.phong);
779         prg.setUniform("uShininess", actor.material.shininess);
780     }
781     else{
782         this.renderer.setProgram(vxl.def.glsl.lambert);
783     }
784     
785     
786     prg.disableAttribute(glsl.TEXCOORD_ATTRIBUTE);
787     prg.disableAttribute(glsl.COLOR_ATTRIBUTE);
788     prg.disableAttribute(glsl.NORMAL_ATTRIBUTE);
789     prg.disableAttribute(glsl.PICKING_COLOR_ATTRIBUTE);
790     prg.enableAttribute(glsl.VERTEX_ATTRIBUTE);
791     
792     prg.setUniform(glsl.PERSPECTIVE_MATRIX, trx.pMatrix);
793     prg.setUniform("uUseVertexColors", false);
794     prg.setUniform("uUseTextures", false);
795     prg.setUniform("uUseShading",actor.material.shading);
796     prg.setUniform("uMaterialDiffuse",diffuse);
797     
798     
799     this._handleCulling(actor);
800     this._applyActorTransform(actor);
801     
802     
803     if(this._onPickingBuffer || this._debug_picking_flag){
804         this._renderPickingBuffer(actor);
805         return;
806     }
807     
808         
809     if (actor.mode != vxl.def.actor.mode.FLAT){
810         try{
811             gl.bindBuffer(gl.ARRAY_BUFFER, buffers.vertex);
812             gl.bufferData(gl.ARRAY_BUFFER, new Float32Array(model.vertices.slice(0)), gl.STATIC_DRAW);
813             prg.setAttributePointer(glsl.VERTEX_ATTRIBUTE, 3, gl.FLOAT, false, 0, 0);
814         }
815         catch(err){
816             alert('There was a problem while rendering the actor ['+actor.name+']. The problem happened while handling the vertex buffer. Error =' +err.description);
817             throw('There was a problem while rendering the actor ['+actor.name+']. The problem happened while handling the vertex buffer. Error =' +err.description);
818         }
819     }
820     
821     
822     var am = vxl.def.actor.mode;
823     
824     switch(actor.mode){
825         case am.SOLID:           this._renderSolid(actor);               break;
826         case am.WIREFRAME:       this._renderWireframe(actor);           break;
827         case am.POINTS:          this._renderPoints(actor);              break;
828         case am.LINES:           this._renderLines(actor);               break;
829         case am.BOUNDING_BOX:    this._renderBoundingBox(actor);         break;
830         case am.BB_AND_SOLID:    this._renderBoundingBoxAndSolid(actor); break;
831         case am.WIRED_AND_SOLID: this._renderWiredAndSolid(actor);       break;
832         case am.FLAT:            this._renderFlat(actor);                break;
833         case am.TEXTURED:        this._renderTextured(actor);            break;
834         default:
835             alert('There was a problem while rendering the actor ['+actor.name+']. The visualization mode: '+actor.mode+' is not valid.');
836             throw('There was a problem while rendering the actor ['+actor.name+']. The visualization mode: '+actor.mode+' is not valid.'); 
837     }
838     
839     //Cleaning up
840     gl.bindBuffer(gl.ARRAY_BUFFER, null);
841     gl.bindBuffer(gl.ELEMENT_ARRAY_BUFFER, null);
842 };
